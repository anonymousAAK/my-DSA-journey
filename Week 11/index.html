🔗 Singly Linked List

A singly linked list is a sequence of nodes, where each node contains data and a pointer to the next node. The last node points to null.

Structure

HEAD→
1→
2→
3→
4→
5→
NULL

Advantages over Arrays

  • O(1) insertion/deletion at head -- no shifting required
  • Dynamic size -- no pre-allocation or resizing needed

Disadvantages

  • O(n) access by index -- no random access (must traverse)
  • Extra memory for each next pointer
  • Poor cache performance -- nodes are scattered in memory

➕ Insertion Operations

Insert at Head -- O(1)

Before: HEAD → [1] → [2] → [3] → NULL Insert 0 at head: 1. Create node [0] 2. Set [0].next = HEAD 3. HEAD = [0] After: HEAD → [0] → [1] → [2] → [3] → NULL

Insert at Tail -- O(n)

Before: HEAD → [1] → [2] → [3] → NULL Insert 4 at tail: 1. Traverse to last node [3] 2. Set [3].next = new node [4] After: HEAD → [1] → [2] → [3] → [4] → NULL
SinglyLinkedList.java
void insertAtHead(int data) { Node newNode = new Node(data); newNode.next = head; head = newNode; } void insertAtTail(int data) { Node newNode = new Node(data); if (head == null) { head = newNode; return; } Node curr = head; while (curr.next != null) curr = curr.next; curr.next = newNode; } void insertAtIndex(int idx, int data) { if (idx == 0) { insertAtHead(data); return; } Node curr = head; for (int i = 0; i < idx - 1 && curr != null; i++) curr = curr.next; if (curr == null) throw new IndexOutOfBoundsException(); Node newNode = new Node(data); newNode.next = curr.next; curr.next = newNode; }
linked_lists.cpp
ListNode* insertAtHead(ListNode* head, int val) { return new ListNode(val, head); } ListNode* insertAtTail(ListNode* head, int val) { ListNode* newNode = new ListNode(val); if (!head) return newNode; ListNode* curr = head; while (curr->next) curr = curr->next; curr->next = newNode; return head; } ListNode* insertAtPosition(ListNode* head, int val, int pos) { if (pos == 0) return insertAtHead(head, val); ListNode* curr = head; for (int i = 0; i < pos - 1 && curr; ++i) curr = curr->next; if (!curr) return head; ListNode* newNode = new ListNode(val, curr->next); curr->next = newNode; return head; }
linked_lists.py
def insert_head(self, val: int) -> None: """Insert a node at the head. O(1)""" self.head = ListNode(val, self.head) def insert_tail(self, val: int) -> None: """Insert a node at the tail. O(n)""" new_node = ListNode(val) if not self.head: self.head = new_node return curr = self.head while curr.next: curr = curr.next curr.next = new_node
linked_lists.rs
fn push_front(&mut self, value: i32) { // self.head.take() moves the current head out, leaving None. let new_node = Box::new(Node { value, next: self.head.take(), }); self.head = Some(new_node); self.size += 1; } fn push_back(&mut self, value: i32) { let new_node = Box::new(Node { value, next: None }); let mut current = &mut self.head; while let Some(ref mut node) = current { if node.next.is_none() { node.next = Some(new_node); self.size += 1; return; } current = &mut node.next; } self.head = Some(new_node); self.size += 1; }

❌ Deletion Operations

Delete value 3 from: HEAD → [1] → [2] → [3] → [4] → NULL 1. Traverse until curr.next.data == 3 2. Set curr.next = curr.next.next (skip node [3]) After: HEAD → [1] → [2] → [4] → NULL
SinglyLinkedList.java
boolean deleteByValue(int val) { if (head == null) return false; if (head.data == val) { head = head.next; return true; } Node curr = head; while (curr.next != null) { if (curr.next.data == val) { curr.next = curr.next.next; return true; } curr = curr.next; } return false; }
linked_lists.cpp
ListNode* deleteByValue(ListNode* head, int val) { if (!head) return nullptr; if (head->val == val) { ListNode* tmp = head->next; delete head; return tmp; } ListNode* curr = head; while (curr->next && curr->next->val != val) { curr = curr->next; } if (curr->next) { ListNode* tmp = curr->next; curr->next = tmp->next; delete tmp; } return head; }
linked_lists.py
def delete(self, val: int) -> bool: if not self.head: return False if self.head.val == val: self.head = self.head.next return True curr = self.head while curr.next: if curr.next.val == val: curr.next = curr.next.next return True curr = curr.next return False

🔄 Reverse a Linked List

Iterative Approach -- O(n) time, O(1) space

Maintain three pointers: prev, curr, and next. At each step, reverse curr.next to point to prev, then advance all three pointers.

Step-by-step reversal: prev=NULL curr=[1]→[2]→[3]→NULL Step 1: next = [2], [1].next = NULL, prev = [1], curr = [2] NULL←[1] [2]→[3]→NULL Step 2: next = [3], [2].next = [1], prev = [2], curr = [3] NULL←[1]←[2] [3]→NULL Step 3: next = NULL, [3].next = [2], prev = [3], curr = NULL NULL←[1]←[2]←[3] HEAD = prev = [3] Result: [3]→[2]→[1]→NULL
SinglyLinkedList.java
// Iterative -- O(n) time, O(1) space void reverseIterative() { Node prev = null, curr = head, next; while (curr != null) { next = curr.next; curr.next = prev; prev = curr; curr = next; } head = prev; } // Recursive -- O(n) time, O(n) space (call stack) Node reverseRec(Node node) { if (node == null || node.next == null) return node; Node newHead = reverseRec(node.next); node.next.next = node; node.next = null; return newHead; }
linked_lists.cpp
ListNode* reverseIterative(ListNode* head) { ListNode* prev = nullptr; ListNode* curr = head; while (curr) { ListNode* next = curr->next; curr->next = prev; prev = curr; curr = next; } return prev; } ListNode* reverseRecursive(ListNode* head) { if (!head || !head->next) return head; ListNode* newHead = reverseRecursive(head->next); head->next->next = head; head->next = nullptr; return newHead; }
linked_lists.py
def reverse_iterative(self) -> None: prev, curr = None, self.head while curr: nxt = curr.next curr.next = prev prev = curr curr = nxt self.head = prev def reverse_recursive(self) -> None: def _reverse(node): if not node or not node.next: return node new_head = _reverse(node.next) node.next.next = node node.next = None return new_head self.head = _reverse(self.head)
linked_lists.rs
fn reverse(&mut self) { let mut prev: Option<Box<Node>> = None; let mut current = self.head.take(); while let Some(mut node) = current { let next = node.next.take(); node.next = prev; prev = Some(node); current = next; } self.head = prev; }
Rust Note
Linked lists are notoriously difficult in Rust due to ownership rules. The .take() method is key -- it moves the value out of an Option and replaces it with None, satisfying the borrow checker. For production code, use std::collections::LinkedList or VecDeque.

🎯 Find Middle Node

Use Floyd's slow/fast pointer technique: slow moves one step at a time, fast moves two steps. When fast reaches the end, slow is at the middle.

[1] → [2] → [3] → [4] → [5] → NULL Step 0: slow=[1], fast=[1] Step 1: slow=[2], fast=[3] Step 2: slow=[3], fast=[5] Step 3: fast.next = NULL → STOP Middle node = 3

🔀 Cycle Detection -- Floyd's Algorithm

Also known as the tortoise and hare algorithm. If there is a cycle, the fast pointer will eventually lap the slow pointer and they will meet. If fast reaches null, there is no cycle.

List with cycle: [1] → [2] → [3] → [4] → [5] ↑ | +←←←←←←←←←←←←↓ Phase 1: Detect cycle slow and fast start at head slow moves 1 step, fast moves 2 steps If they meet → cycle exists! Phase 2: Find cycle start Reset slow to head Both move 1 step at a time They meet at the cycle start node = [3]
SinglyLinkedList.java
boolean hasCycle() { Node slow = head, fast = head; while (fast != null && fast.next != null) { slow = slow.next; fast = fast.next.next; if (slow == fast) return true; // cycle detected } return false; }
linked_lists.cpp
ListNode* detectCycle(ListNode* head) { ListNode* slow = head; ListNode* fast = head; while (fast && fast->next) { slow = slow->next; fast = fast->next->next; if (slow == fast) { // Phase 2: find cycle start slow = head; while (slow != fast) { slow = slow->next; fast = fast->next; } return slow; } } return nullptr; }
linked_lists.py
def has_cycle(self) -> bool: slow = fast = self.head while fast and fast.next: slow = slow.next fast = fast.next.next if slow is fast: return True return False

🔗 Merge Two Sorted Linked Lists

Given two sorted linked lists, merge them into one sorted list by re-linking existing nodes. Use a dummy head (sentinel node) to simplify edge cases.

L1: [1] → [3] → [5] → NULL L2: [2] → [4] → [6] → NULL Compare heads: 1 < 2 → take [1] Compare: 3 > 2 → take [2] Compare: 3 < 4 → take [3] Compare: 5 > 4 → take [4] Compare: 5 < 6 → take [5] Remaining: [6] Merged: [1] → [2] → [3] → [4] → [5] → [6] → NULL
MergeSortedListsAndLRU.java
static ListNode mergeSortedLists(ListNode l1, ListNode l2) { ListNode dummy = new ListNode(0); // sentinel head ListNode curr = dummy; while (l1 != null && l2 != null) { if (l1.val <= l2.val) { curr.next = l1; l1 = l1.next; } else { curr.next = l2; l2 = l2.next; } curr = curr.next; } curr.next = (l1 != null) ? l1 : l2; return dummy.next; }
linked_lists.cpp
ListNode* mergeSortedLists(ListNode* l1, ListNode* l2) { ListNode dummy(0); ListNode* tail = &dummy; while (l1 && l2) { if (l1->val <= l2->val) { tail->next = l1; l1 = l1->next; } else { tail->next = l2; l2 = l2->next; } tail = tail->next; } tail->next = l1 ? l1 : l2; return dummy.next; }
linked_lists.py
def merge_sorted_lists(l1, l2): dummy = ListNode(0) tail = dummy while l1 and l2: if l1.val <= l2.val: tail.next = l1; l1 = l1.next else: tail.next = l2; l2 = l2.next tail = tail.next tail.next = l1 if l1 else l2 return dummy.next
linked_lists.rs
// Rust uses Vec-based merge for simplicity fn merge_sorted_lists(a: &[i32], b: &[i32]) -> Vec<i32> { let mut result = Vec::with_capacity(a.len() + b.len()); let (mut i, mut j) = (0, 0); while i < a.len() && j < b.len() { if a[i] <= b[j] { result.push(a[i]); i += 1; } else { result.push(b[j]); j += 1; } } result.extend_from_slice(&a[i..]); result.extend_from_slice(&b[j..]); result }
Key Insight
The dummy/sentinel node pattern eliminates edge cases for handling the head of the merged list. Instead of special-casing "which list provides the first node?", we always append to dummy.next and return dummy.next at the end.

💾 LRU Cache

A Least Recently Used cache evicts the least recently accessed item when full. Both get and put must be O(1).

Data Structure: Doubly Linked List + Hash Map

  • Hash Map: key → node (for O(1) lookup)
  • Doubly Linked List: maintains access order (head = most recent, tail = least recent)
  • On get or put: move the accessed node to the head
  • On eviction: remove the tail node
LRU Cache (capacity=3): HashMap: { 1:node1, 3:node3, 4:node4 } Doubly Linked List (MRU → LRU): DUMMY_HEAD ↔ [4:40] ↔ [3:30] ↔ [1:10] ↔ DUMMY_TAIL get(1) → move node1 to front: DUMMY_HEAD ↔ [1:10] ↔ [4:40] ↔ [3:30] ↔ DUMMY_TAIL put(5, 50) → evict LRU (key 3), insert 5: DUMMY_HEAD ↔ [5:50] ↔ [1:10] ↔ [4:40] ↔ DUMMY_TAIL
Real-World Usage
LRU caches are used everywhere: CPU caches, browser history, Redis eviction policies, database buffer pools, and CDN edge caches. This is a classic system design interview question.
MergeSortedListsAndLRU.java
static class LRUCache { static class DLNode { int key, val; DLNode prev, next; DLNode(int k, int v) { key = k; val = v; } } private final int capacity; private final Map<Integer, DLNode> map; private final DLNode head, tail; LRUCache(int capacity) { this.capacity = capacity; this.map = new HashMap<>(); head = new DLNode(0, 0); tail = new DLNode(0, 0); head.next = tail; tail.prev = head; } private void remove(DLNode node) { node.prev.next = node.next; node.next.prev = node.prev; } private void insertAfterHead(DLNode node) { node.next = head.next; node.prev = head; head.next.prev = node; head.next = node; } int get(int key) { if (!map.containsKey(key)) return -1; DLNode node = map.get(key); remove(node); insertAfterHead(node); return node.val; } void put(int key, int value) { if (map.containsKey(key)) { DLNode node = map.get(key); node.val = value; remove(node); insertAfterHead(node); } else { if (map.size() == capacity) { DLNode lru = tail.prev; remove(lru); map.remove(lru.key); } DLNode newNode = new DLNode(key, value); insertAfterHead(newNode); map.put(key, newNode); } } }

class LRUCache { int capacity_; list<pair<int, int>> cache_; unordered_map<int, list<pair<int, int>>::iterator> map_; public: LRUCache(int capacity) : capacity_(capacity) {} int get(int key) { auto it = map_.find(key); if (it == map_.end()) return -1; cache_.splice(cache_.begin(), cache_, it->second); return it->second->second; } void put(int key, int value) { auto it = map_.find(key); if (it != map_.end()) { it->second->second = value; cache_.splice(cache_.begin(), cache_, it->second); return; } if ((int)cache_.size() == capacity_) { auto& [lruKey, lruVal] = cache_.back(); map_.erase(lruKey); cache_.pop_back(); } cache_.emplace_front(key, value); map_[key] = cache_.begin(); } };

struct LRUCache { capacity: usize, map: HashMap<i32, i32>, order: VecDeque<i32>, } impl LRUCache { fn new(capacity: usize) -> Self { LRUCache { capacity, map: HashMap::new(), order: VecDeque::new() } } fn get(&mut self, key: i32) -> Option<i32> { if let Some(&value) = self.map.get(&key) { self.move_to_back(key); Some(value) } else { None } } fn put(&mut self, key: i32, value: i32) { if self.map.contains_key(&key) { self.map.insert(key, value); self.move_to_back(key); } else { if self.map.len() >= self.capacity { if let Some(lru_key) = self.order.pop_front() { self.map.remove(&lru_key); } } self.map.insert(key, value); self.order.push_back(key); } } fn move_to_back(&mut self, key: i32) { if let Some(pos) = self.order.iter().position(|&k| k == key) { self.order.remove(pos); } self.order.push_back(key); } }
