🔮 Graph Representations

A graph G = (V, E) consists of vertices V connected by edges E. Graphs can be directed/undirected, weighted/unweighted, cyclic/acyclic.

Adjacency List vs Matrix

PropertyAdjacency ListAdjacency Matrix
SpaceO(V + E)O(V^2)
Check edge (u,v)O(degree(u))O(1)
Iterate neighborsO(degree)O(V)
Best forSparse graphs (most real-world)Dense graphs

🔍 BFS -- Breadth-First Search

BFS explores the graph level by level using a queue. It finds the shortest path in unweighted graphs.

Graph: 0 -- 1 -- 2 | | 3 -- 4 -- 5 BFS from 0: Level 0: [0] Level 1: [1, 3] Level 2: [2, 4] Level 3: [5] Distances: {0:0, 1:1, 3:1, 2:2, 4:2, 5:3}
GraphRepresentations.java
int[] shortestDistances(int src) { int[] dist = new int[V]; Arrays.fill(dist, -1); dist[src] = 0; Queue<Integer> queue = new LinkedList<>(); queue.offer(src); while (!queue.isEmpty()) { int u = queue.poll(); for (int v : adj.get(u)) { if (dist[v] == -1) { dist[v] = dist[u] + 1; queue.offer(v); } } } return dist; }
graphs.cpp
std::vector<int> bfs(int src) const { std::vector<int> dist(n_, -1); std::queue<int> q; dist[src] = 0; q.push(src); while (!q.empty()) { int u = q.front(); q.pop(); for (auto [v, w] : adj_[u]) { if (dist[v] == -1) { dist[v] = dist[u] + 1; q.push(v); } } } return dist; }
graphs.py
def bfs(graph: Graph, source: int) -> Tuple[List[int], Dict[int, int]]: visited: Set[int] = {source} queue: deque[int] = deque([source]) order: List[int] = [] distances: Dict[int, int] = {source: 0} while queue: node = queue.popleft() order.append(node) for neighbor in graph.adj[node]: if neighbor not in visited: visited.add(neighbor) distances[neighbor] = distances[node] + 1 queue.append(neighbor) return order, distances
graphs.rs
fn bfs(&self, start: usize) -> (Vec<usize>, Vec<i32>) { let mut visited = vec![false; self.num_vertices]; let mut distances = vec![-1i32; self.num_vertices]; let mut order = Vec::new(); let mut queue = VecDeque::new(); visited[start] = true; distances[start] = 0; queue.push_back(start); while let Some(u) = queue.pop_front() { order.push(u); for &v in &self.adj[u] { if !visited[v] { visited[v] = true; distances[v] = distances[u] + 1; queue.push_back(v); } } } (order, distances) }

🔮 DFS -- Depth-First Search

DFS explores as deep as possible before backtracking. Uses recursion or an explicit stack. Time: O(V + E).

GraphRepresentations.java
List<Integer> dfs(int src) { List<Integer> result = new ArrayList<>(); boolean[] visited = new boolean[V]; dfsHelper(src, visited, result); return result; } void dfsHelper(int u, boolean[] visited, List<Integer> result) { visited[u] = true; result.add(u); for (int v : adj.get(u)) if (!visited[v]) dfsHelper(v, visited, result); }

📌 Connected Components

Count the number of connected components by running DFS/BFS from each unvisited vertex. Each new start increments the component count.

GraphRepresentations.java
int countComponents() { boolean[] visited = new boolean[V]; int count = 0; for (int i = 0; i < V; i++) { if (!visited[i]) { dfsForComponent(i, visited); count++; } } return count; }

🔄 Cycle Detection

Undirected Graph

During DFS, if we encounter a visited neighbor that is NOT the parent of the current node, there is a cycle (a back edge).

Directed Graph (3-state coloring)

Use 3 states: WHITE (unvisited), GRAY (in the current DFS stack), BLACK (fully processed). A back edge to a GRAY node indicates a cycle.

TopologicalSort.java -- Directed Cycle Detection
// States: 0=unvisited, 1=in-stack (being processed), 2=done boolean hasCycle() { int[] state = new int[V]; for (int i = 0; i < V; i++) if (state[i] == 0 && dfsDetectCycle(i, state)) return true; return false; } boolean dfsDetectCycle(int u, int[] state) { state[u] = 1; // mark as in-stack for (int v : adj.get(u)) { if (state[v] == 1) return true; // back edge -- cycle if (state[v] == 0 && dfsDetectCycle(v, state)) return true; } state[u] = 2; // fully processed return false; }
graphs.cpp
bool dfs_cycle_directed(int u, std::vector<int>& color) const { color[u] = 1; // gray for (auto [v, w] : adj_[u]) { if (color[v] == 1) return true; // back edge => cycle if (color[v] == 0 && dfs_cycle_directed(v, color)) return true; } color[u] = 2; // black return false; }
graphs.py
def has_cycle_directed(graph: Graph) -> bool: WHITE, GRAY, BLACK = 0, 1, 2 color: Dict[int, int] = {v: WHITE for v in graph.vertices} def _dfs(node: int) -> bool: color[node] = GRAY for neighbor in graph.adj[node]: if color.get(neighbor, WHITE) == GRAY: return True # back edge => cycle if color.get(neighbor, WHITE) == WHITE: if _dfs(neighbor): return True color[node] = BLACK return False for v in graph.vertices: if color[v] == WHITE and _dfs(v): return True return False

🎨 Bipartite Check

A graph is bipartite (2-colorable) if vertices can be split into two sets such that no edge connects vertices in the same set. Equivalent to having no odd-length cycles.

Key Insight
Use BFS with alternating colors. If any neighbor has the same color as the current node, the graph is NOT bipartite.
GraphRepresentations.java
boolean isBipartite() { int[] color = new int[V]; // 0=uncolored, 1=red, -1=blue for (int i = 0; i < V; i++) { if (color[i] == 0) { Queue<Integer> queue = new LinkedList<>(); queue.offer(i); color[i] = 1; while (!queue.isEmpty()) { int u = queue.poll(); for (int v : adj.get(u)) { if (color[v] == 0) { color[v] = -color[u]; queue.offer(v); } else if (color[v] == color[u]) return false; } } } } return true; }
graphs.rs
fn is_bipartite(&self) -> bool { let mut color = vec![-1i32; self.num_vertices]; for start in 0..self.num_vertices { if color[start] != -1 { continue; } let mut queue = VecDeque::new(); color[start] = 0; queue.push_back(start); while let Some(u) = queue.pop_front() { for &v in &self.adj[u] { match color[v] { -1 => { color[v] = 1 - color[u]; queue.push_back(v); } c if c == color[u] => { return false; } _ => {} } } } } true }

📋 Topological Sort (DFS)

A topological ordering of a DAG is a linear ordering of vertices such that for every edge u→v, u appears before v. Only possible for Directed Acyclic Graphs.

DFS Approach

Run DFS. Push each node to a stack AFTER all descendants are processed. Pop stack for the topological order.

TopologicalSort.java
List<Integer> topoSortDFS() { boolean[] visited = new boolean[V]; Deque<Integer> stack = new ArrayDeque<>(); for (int i = 0; i < V; i++) if (!visited[i]) dfsTopo(i, visited, stack); List<Integer> result = new ArrayList<>(); while (!stack.isEmpty()) result.add(stack.pop()); return result; } void dfsTopo(int u, boolean[] visited, Deque<Integer> stack) { visited[u] = true; for (int v : adj.get(u)) if (!visited[v]) dfsTopo(v, visited, stack); stack.push(u); // push AFTER processing all descendants }
graphs.py
def topological_sort_dfs(graph: Graph) -> Optional[List[int]]: WHITE, GRAY, BLACK = 0, 1, 2 color = {v: WHITE for v in graph.vertices} result: List[int] = [] def _dfs(node: int) -> bool: color[node] = GRAY for neighbor in graph.adj[node]: if color.get(neighbor, WHITE) == GRAY: return False if color.get(neighbor, WHITE) == WHITE: if not _dfs(neighbor): return False color[node] = BLACK result.append(node) # post-order return True for v in sorted(graph.vertices): if color[v] == WHITE: if not _dfs(v): return None result.reverse() return result

⚡ Kahn's Algorithm (BFS Topological Sort)

Process nodes with in-degree 0 first (no dependencies). Remove their edges, enqueue newly zero-indegree nodes. If not all nodes processed, a cycle exists!

DAG: 5→2, 5→0, 4→0, 4→1, 2→3, 3→1 In-degrees: {0:2, 1:2, 2:1, 3:1, 4:0, 5:0} Queue: [4, 5] → process 4 → [5] → process 5 → [0, 2] → ... Result: [4, 5, 0, 2, 3, 1] (one valid topological order)
TopologicalSort.java
List<Integer> topoSortKahn() { int[] indegree = new int[V]; for (int u = 0; u < V; u++) for (int v : adj.get(u)) indegree[v]++; Queue<Integer> queue = new LinkedList<>(); for (int i = 0; i < V; i++) if (indegree[i] == 0) queue.offer(i); List<Integer> result = new ArrayList<>(); while (!queue.isEmpty()) { int u = queue.poll(); result.add(u); for (int v : adj.get(u)) { indegree[v]--; if (indegree[v] == 0) queue.offer(v); } } if (result.size() != V) return null; // cycle detected return result; }
graphs.cpp
std::optional<std::vector<int>> topo_sort_kahn() const { std::vector<int> indegree(n_, 0); for (int u = 0; u < n_; ++u) for (auto [v, w] : adj_[u]) ++indegree[v]; std::queue<int> q; for (int i = 0; i < n_; ++i) if (indegree[i] == 0) q.push(i); std::vector<int> order; while (!q.empty()) { int u = q.front(); q.pop(); order.push_back(u); for (auto [v, w] : adj_[u]) if (--indegree[v] == 0) q.push(v); } if (static_cast<int>(order.size()) != n_) return std::nullopt; return order; }
graphs.rs
fn topological_sort(&self) -> Option<Vec<usize>> { let mut in_degree = vec![0usize; self.num_vertices]; for u in 0..self.num_vertices { for &v in &self.adj[u] { in_degree[v] += 1; } } let mut queue: VecDeque<usize> = VecDeque::new(); for (v, &deg) in in_degree.iter().enumerate() { if deg == 0 { queue.push_back(v); } } let mut order = Vec::with_capacity(self.num_vertices); while let Some(u) = queue.pop_front() { order.push(u); for &v in &self.adj[u] { in_degree[v] -= 1; if in_degree[v] == 0 { queue.push_back(v); } } } if order.len() == self.num_vertices { Some(order) } else { None } }
