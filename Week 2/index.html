🔌 If-Else Conditional Statements

Conditional statements allow your program to make decisions. The if block executes when the condition is true; the else block runs when it is false.

Relational Operators

OperatorMeaningExample
<Less thana < b
>Greater thana > b
==Equal toa == b
!=Not equal toa != b
<=Less than or equala <= b
>=Greater than or equala >= b

Implementation

If_else.java
public static void main(String args[]) { int a = 10, b = 15; if (a > b) { System.out.print("a "); // Skipped: 10 is not > 15 } else { System.out.print("b "); // Executes: a is NOT greater } System.out.print("is greater"); // Always executes (outside if-else) } // Output: b is greater
control_flow.cpp
int x = 42; if (x > 100) cout << "Big" << endl; else if (x > 10) cout << "Medium" << endl; // this runs else cout << "Small" << endl; // Ternary operator (compact if-else) string status = (x % 2 == 0) ? "even" : "odd"; cout << x << " is " << status << endl;
control_flow.py
# Python uses elif (not else if), indentation defines blocks x = 42 if x > 100: print("Big") elif x > 10: print("Medium") # this runs else: print("Small") # Ternary expression status = "even" if x % 2 == 0 else "odd" print(f"{x} is {status}")
control_flow.rs
// In Rust, if/else is an EXPRESSION (returns a value!) let x = 42; let category = if x > 100 { "Big" } else if x > 10 { "Medium" } else { "Small" }; println!("{} is {}", x, category); let status = if x % 2 == 0 { "even" } else { "odd" }; println!("{} is {}", x, status);
Key Insight
Code written outside the if-else block always executes regardless of the condition. Be mindful of what is inside vs. outside the braces.

🔄 While Loop

The while loop repeats a block of code as long as a specified condition remains true. Be careful with the condition -- using assignment (=) instead of comparison (==) leads to subtle bugs.

Tricky Example: Assignment vs Comparison

While_Loop.java
public static void main(String[] args) { int x = 5; int y = 5; while ((x = 5) == y) { // Assignment! x is reset to 5 each time System.out.println("Hello"); x++; // x becomes 6, but resets to 5 on next check y++; // y becomes 6, so (5 == 6) is false } } // Output: prints "Hello" exactly ONCE // After 1st iteration: y=6, condition (x=5)==6 is false
control_flow.cpp
// While loop: Multiplication Table int n = 7, i = 1; while (i <= 10) { cout << n << " x " << i << " = " << n * i << endl; i++; } // Do-while: executes at least once int count = 0; do { cout << "Count: " << count << endl; count++; } while (count < 3);
control_flow.py
# While loop: Multiplication Table n, i = 7, 1 while i <= 10: print(f"{n} x {i} = {n * i}") i += 1 # Python has no ++ operator # Python has no do-while; use while True + break instead count = 0 while True: print(f"Count: {count}") count += 1 if count >= 3: break
control_flow.rs
// While loop let n = 7; let mut i = 1; while i <= 10 { println!("{} x {} = {}", n, i, n * i); i += 1; // Rust has no ++ operator } // Rust's 'loop' keyword = infinite loop with break let mut count = 0; let result = loop { count += 1; if count == 5 { break count * 10; } // returns value! }; println!("Loop returned: {}", result); // 50
Common Pitfall
Using = (assignment) instead of == (comparison) inside a loop condition can create hard-to-find bugs. Java allows assignment in conditions, which resets the variable every iteration.

🌡 Fahrenheit to Celsius Table

A practical while loop example: generate a temperature conversion table using the formula C = (F - 32) * 5 / 9.

Implementation

Fahrenheit_to_Celsius_Table.java
import java.util.Scanner; public class F_C_Table { public static void main(String[] args) { int S, E, W; // Start, End, Step (Fahrenheit) Scanner p = new Scanner(System.in); S = p.nextInt(); // Read start value E = p.nextInt(); // Read end value W = p.nextInt(); // Read step size while (E >= S) { int a = ((S - 32) * 5) / 9; // Convert F to C (integer math) System.out.println(S + " " + a); S = S + W; // Move to next Fahrenheit value } } }
control_flow.cpp
// F to C table with formatted output printf("%5s | %8s\n", "F", "C"); for (int f = 32; f <= 212; f += 20) { double c = (f - 32) * 5.0 / 9.0; printf("%5d | %8.2f\n", f, c); }
control_flow.py
print(f"{'F':>5} | {'C':>8}") print("-" * 16) for f in range(32, 213, 20): c = (f - 32) * 5 / 9 print(f"{f:>5} | {c:>8.2f}")
control_flow.rs
println!("{:>5} | {:>8}", "F", "C"); let mut f = 32; while f <= 212 { let c = (f as f64 - 32.0) * 5.0 / 9.0; println!("{:>5} | {:>8.2}", f, c); f += 20; }

🔢 Sum of Even & Odd Digits

A classic digit-extraction problem: separate the digits of a number into even and odd categories, then compute their sums. This technique uses num % 10 to extract the last digit and num / 10 to remove it.

Step-by-Step Visualization

Number: 13245 Step 1: last = 13245 % 10 = 5 (odd) → oddSum = 5, num = 1324 Step 2: last = 1324 % 10 = 4 (even) → evenSum = 4, num = 132 Step 3: last = 132 % 10 = 2 (even) → evenSum = 6, num = 13 Step 4: last = 13 % 10 = 3 (odd) → oddSum = 8, num = 1 Step 5: last = 1 % 10 = 1 (odd) → oddSum = 9, num = 0 Result: evenSum = 6, oddSum = 9

Implementation

Sum_of_Even_Odd.java
import java.util.Scanner; public class Main { public static void main(String[] args) { Scanner s = new Scanner(System.in); int num = s.nextInt(); int evenSum = 0, oddSum = 0; while (num > 0) { int last = num % 10; // Extract last digit if (last % 2 == 0) evenSum += last; // Even digit else oddSum += last; // Odd digit num = num / 10; // Remove last digit } System.out.println(evenSum + " " + oddSum); } }
control_flow.cpp
pair sumEvenOddDigits(int n) { int evenSum = 0, oddSum = 0; n = abs(n); while (n > 0) { int digit = n % 10; if (digit % 2 == 0) evenSum += digit; else oddSum += digit; n /= 10; } return {evenSum, oddSum}; } auto [e, o] = sumEvenOddDigits(1234); // C++17 structured bindings
control_flow.py
def sum_even_odd_digits(n: int) -> tuple[int, int]: even_sum, odd_sum = 0, 0 n = abs(n) while n > 0: digit = n % 10 if digit % 2 == 0: even_sum += digit else: odd_sum += digit n //= 10 # integer division return even_sum, odd_sum e, o = sum_even_odd_digits(1234) print(f"even={e}, odd={o}") # even=6, odd=4
control_flow.rs
fn sum_even_odd_digits(mut n: i32) -> (i32, i32) { let (mut even_sum, mut odd_sum) = (0, 0); n = n.abs(); while n > 0 { let digit = n % 10; if digit % 2 == 0 { even_sum += digit; } else { odd_sum += digit; } n /= 10; } (even_sum, odd_sum) } let (e, o) = sum_even_odd_digits(1234);
Key Insight
The pattern num % 10 (extract last digit) + num / 10 (remove last digit) is a fundamental technique used in dozens of number-theory problems. Master this pattern early.

🔎 Finding Factors

A factor of n is any integer i where n % i == 0. We only need to check up to n/2 since no factor (other than n itself) can exceed n/2.

Implementation

Factors.java
import java.util.Scanner; public class Solution { public static void main(String[] args) { Scanner s = new Scanner(System.in); int n = s.nextInt(); int i = 2; // Start from 2 (excluding 1) while (i <= n / 2) { // No factor > n/2 exists (except n) if (n % i == 0) { System.out.print(i + " "); } i += 1; } } }
control_flow.cpp
// Optimized: O(sqrt(n)) instead of O(n) vector factors(int n) { vector result; for (int i = 1; i * i <= n; i++) { if (n % i == 0) { result.push_back(i); if (i != n / i) result.push_back(n / i); } } sort(result.begin(), result.end()); return result; }
control_flow.py
def factors(n: int) -> list[int]: """Find all factors of n. Time: O(sqrt(n))""" result = [] for i in range(1, int(n**0.5) + 1): if n % i == 0: result.append(i) if i != n // i: result.append(n // i) return sorted(result) print(f"Factors of 12: {factors(12)}") # [1, 2, 3, 4, 6, 12]
control_flow.rs
fn factors(n: i32) -> Vec { let mut result = Vec::new(); let mut i = 1; while i * i <= n { if n % i == 0 { result.push(i); if i != n / i { result.push(n / i); } } i += 1; } result.sort(); result }

⚡ Find Power (x^n)

Compute x raised to the power n through repeated multiplication. This brute-force O(n) approach multiplies x by itself n times.

Implementation

Find_Power.java
import java.util.Scanner; public class FindPower { public static void main(String[] args) { Scanner s = new Scanner(System.in); int x = s.nextInt(); // base int n = s.nextInt(); // exponent long ans = 1; // Start with 1 (x^0 = 1) for (int i = 0; i < n; i++) { // Multiply x, n times ans *= x; } System.out.println(x + "^" + n + " = " + ans); } } // Time: O(n) -- See Week 5 for O(log n) fast exponentiation
control_flow.cpp
long long power(int x, int n) { long long result = 1; for (int i = 0; i < n; i++) result *= x; return result; } cout << "2^10 = " << power(2, 10) << endl; // 1024 cout << "3^5 = " << power(3, 5) << endl; // 243
control_flow.py
def power(x: int, n: int) -> int: result = 1 for _ in range(n): result *= x return result print(f"2^10 = {power(2, 10)}") # 1024 # Python also has built-in: 2**10 = 1024
control_flow.rs
fn power(x: i64, n: u32) -> i64 { let mut result: i64 = 1; for _ in 0..n { result *= x; } result } println!("2^10 = {}", power(2, 10)); // 1024 // Rust also has: 2_i64.pow(10)
Key Insight
The result uses long (not int) because exponentiation grows very quickly -- 2^31 already exceeds the int range. Always consider overflow when dealing with repeated multiplication.
