🔍 Binary Search

Binary search halves the search space at each step by comparing the target with the middle element. Prerequisite: the array must be sorted.

Step-by-Step Visualization

Array: [1, 3, 5, 7, 9, 11, 13, 15] Target: 7 Step 1: lo=0, hi=7, mid=3 → arr[3]=7 == target → Found at index 3! Another example, Target: 11 Step 1: lo=0, hi=7, mid=3 → arr[3]=7 < 11 → lo=4 Step 2: lo=4, hi=7, mid=5 → arr[5]=11 == target → Found at index 5!
Integer Overflow Pitfall

BAD: mid = (low + high) / 2 -- can overflow if low + high > Integer.MAX_VALUE
GOOD: mid = low + (high - low) / 2 -- always safe

BinarySearch.java
// Standard binary search -- O(log n) time, O(1) space static int binarySearch(int[] arr, int target) { int low = 0, high = arr.length - 1; while (low <= high) { int mid = low + (high - low) / 2; if (arr[mid] == target) return mid; else if (arr[mid] < target) low = mid + 1; else high = mid - 1; } return -1; } // Recursive binary search -- O(log n) time, O(log n) stack static int binarySearchRec(int[] arr, int target, int low, int high) { if (low > high) return -1; int mid = low + (high - low) / 2; if (arr[mid] == target) return mid; else if (arr[mid] < target) return binarySearchRec(arr, target, mid + 1, high); else return binarySearchRec(arr, target, low, mid - 1); }
searching.cpp

/// Standard binary search. Returns Some(index) or None. /// Time: O(log n), Space: O(1) fn binary_search(arr: &[i32], target: i32) -> Option<usize> { if arr.is_empty() { return None; } let mut left = 0usize; let mut right = arr.len() - 1; while left <= right { let mid = left + (right - left) / 2; match arr[mid].cmp(&target) { std::cmp::Ordering::Equal => return Some(mid), std::cmp::Ordering::Less => left = mid + 1, std::cmp::Ordering::Greater => { if mid == 0 { break; } right = mid - 1; } } } None }

📍 First & Last Occurrence

When the array has duplicates, standard binary search returns an arbitrary match. To find the first (leftmost) or last (rightmost) occurrence, keep searching even after finding a match.

BinarySearch.java
// Find FIRST occurrence -- O(log n) static int firstOccurrence(int[] arr, int target) { int low = 0, high = arr.length - 1, result = -1; while (low <= high) { int mid = low + (high - low) / 2; if (arr[mid] == target) { result = mid; high = mid - 1; // keep searching LEFT } else if (arr[mid] < target) low = mid + 1; else high = mid - 1; } return result; } // Find LAST occurrence -- O(log n) static int lastOccurrence(int[] arr, int target) { int low = 0, high = arr.length - 1, result = -1; while (low <= high) { int mid = low + (high - low) / 2; if (arr[mid] == target) { result = mid; low = mid + 1; // keep searching RIGHT } else if (arr[mid] < target) low = mid + 1; else high = mid - 1; } return result; } // Count occurrences = last - first + 1 static int countOccurrences(int[] arr, int target) { int first = firstOccurrence(arr, target); if (first == -1) return 0; return lastOccurrence(arr, target) - first + 1; } // [1,2,2,2,3,4,4,5]: count(2) = 3, count(4) = 2
searching.cpp
int firstOccurrence(const vector<int>& arr, int target) { int lo = 0, hi = (int)arr.size() - 1, result = -1; while (lo <= hi) { int mid = lo + (hi - lo) / 2; if (arr[mid] == target) { result = mid; hi = mid - 1; } else if (arr[mid] < target) lo = mid + 1; else hi = mid - 1; } return result; } int lastOccurrence(const vector<int>& arr, int target) { int lo = 0, hi = (int)arr.size() - 1, result = -1; while (lo <= hi) { int mid = lo + (hi - lo) / 2; if (arr[mid] == target) { result = mid; lo = mid + 1; } else if (arr[mid] < target) lo = mid + 1; else hi = mid - 1; } return result; }

searching.rs
fn first_occurrence(arr: &[i32], target: i32) -> Option<usize> { let (mut left, mut right) = (0, arr.len()); let mut result: Option<usize> = None; while left < right { let mid = left + (right - left) / 2; match arr[mid].cmp(&target) { std::cmp::Ordering::Equal => { result = Some(mid); right = mid; } std::cmp::Ordering::Less => left = mid + 1, std::cmp::Ordering::Greater => right = mid, } } result } fn last_occurrence(arr: &[i32], target: i32) -> Option<usize> { let (mut left, mut right) = (0, arr.len()); let mut result: Option<usize> = None; while left < right { let mid = left + (right - left) / 2; match arr[mid].cmp(&target) { std::cmp::Ordering::Equal => { result = Some(mid); left = mid + 1; } std::cmp::Ordering::Less => left = mid + 1, std::cmp::Ordering::Greater => right = mid, } } result }

🔄 Search in Rotated Sorted Array

A rotated sorted array like [4, 5, 6, 7, 0, 1, 2] was sorted then rotated at some pivot. At each step of binary search, one half is always sorted. Determine which half the target lies in and discard the other.

Array: [4, 5, 6, 7, 0, 1, 2] Target: 0 Step 1: lo=0, hi=6, mid=3 → arr[3]=7 Left [4,5,6,7] is sorted. Target 0 not in [4,7] → go right Step 2: lo=4, hi=6, mid=5 → arr[5]=1 Left [0,1] is sorted... but arr[lo]=0 <= 0 < 1 → go left Step 3: lo=4, hi=4, mid=4 → arr[4]=0 → Found at index 4!
BinarySearch.java
// Search in rotated sorted array -- O(log n) static int searchRotated(int[] arr, int target) { int low = 0, high = arr.length - 1; while (low <= high) { int mid = low + (high - low) / 2; if (arr[mid] == target) return mid; if (arr[low] <= arr[mid]) { // left half is sorted if (arr[low] <= target && target < arr[mid]) high = mid - 1; else low = mid + 1; } else { // right half is sorted if (arr[mid] < target && target <= arr[high]) low = mid + 1; else high = mid - 1; } } return -1; } // [4,5,6,7,0,1,2], target=0 --> 4 // [4,5,6,7,0,1,2], target=3 --> -1
searching.cpp
int searchRotated(const vector<int>& arr, int target) { int lo = 0, hi = (int)arr.size() - 1; while (lo <= hi) { int mid = lo + (hi - lo) / 2; if (arr[mid] == target) return mid; if (arr[lo] <= arr[mid]) { if (arr[lo] <= target && target < arr[mid]) hi = mid - 1; else lo = mid + 1; } else { if (arr[mid] < target && target <= arr[hi]) lo = mid + 1; else hi = mid - 1; } } return -1; }

searching.rs
fn search_rotated(arr: &[i32], target: i32) -> Option<usize> { if arr.is_empty() { return None; } let (mut left, mut right) = (0, arr.len() - 1); while left <= right { let mid = left + (right - left) / 2; if arr[mid] == target { return Some(mid); } if arr[left] <= arr[mid] { if arr[left] <= target && target < arr[mid] { if mid == 0 { break; } right = mid - 1; } else { left = mid + 1; } } else { if arr[mid] < target && target <= arr[right] { left = mid + 1; } else { if mid == 0 { break; } right = mid - 1; } } } None }

⚡ Binary Search on the Answer

Binary search is not just for finding elements in arrays. It can be applied to any monotonic function over a range of possible answers.

The Universal Template

Pattern: "Find the minimum/maximum value X such that condition(X) is true." lo = minimum possible answer hi = maximum possible answer while (lo < hi) { mid = lo + (hi - lo) / 2 if (condition(mid)) hi = mid // mid might be the answer, try smaller else lo = mid + 1 // mid is too small, go bigger } return lo // lo == hi is the answer
Key Insight

Binary search on the answer works whenever the condition function is monotonic: once it becomes true, it stays true for all larger values (or vice versa). This creates a clean partition of the search space into "no" and "yes" regions.

√ Problem 1: Integer Square Root

Find the largest integer x such that x*x <= n. Binary search over [1, n/2].

BinarySearchOnAnswer.java
// Integer Square Root -- O(log n) time, O(1) space static int sqrtInt(int n) { if (n < 2) return n; int lo = 1, hi = n / 2; while (lo < hi) { int mid = lo + (hi - lo + 1) / 2; if ((long) mid * mid <= n) lo = mid; else hi = mid - 1; } return lo; } // sqrtInt(16) = 4, sqrtInt(8) = 2, sqrtInt(100) = 10
searching.cpp
// Use mid <= n / mid to avoid overflow long long intSqrt(long long n) { if (n < 0) return -1; long long lo = 0, hi = n, result = 0; while (lo <= hi) { long long mid = lo + (hi - lo) / 2; if (mid <= n / mid) { result = mid; lo = mid + 1; } else hi = mid - 1; } return result; }
searching.py
def integer_sqrt(n): """floor(sqrt(n)) using binary search. O(log n)""" if n < 2: return n lo, hi, result = 1, n // 2, 0 while lo <= hi: mid = lo + (hi - lo) // 2 if mid * mid == n: return mid elif mid * mid < n: result = mid lo = mid + 1 else: hi = mid - 1 return result
searching.rs
fn integer_sqrt(n: u64) -> u64 { if n < 2 { return n; } let (mut left, mut right, mut result) = (1u64, n / 2 + 1, 1u64); while left <= right { let mid = left + (right - left) / 2; if mid <= n / mid { // avoids overflow result = mid; left = mid + 1; } else { right = mid - 1; } } result }

🍌 Problem 2: Koko Eating Bananas

Koko has piles of bananas and H hours. At speed k (bananas/hour), each pile takes ceil(pile/k) hours. Find the minimum speed k to eat all bananas within H hours.

Search space: [1, max(piles)]. Condition: totalHours(k) <= H

BinarySearchOnAnswer.java
// O(n * log(max_pile)) time static boolean canFinish(int[] piles, int h, int speed) { long hours = 0; for (int pile : piles) { hours += (pile + speed - 1) / speed; // ceil(pile / speed) } return hours <= h; } static int minEatingSpeed(int[] piles, int h) { int lo = 1; int hi = Arrays.stream(piles).max().getAsInt(); while (lo < hi) { int mid = lo + (hi - lo) / 2; if (canFinish(piles, h, mid)) hi = mid; // can finish, try slower else lo = mid + 1; // too slow, speed up } return lo; } // piles=[3,6,7,11], H=8 --> 4 // piles=[30,11,23,4,20], H=5 --> 30
searching.cpp
int minEatingSpeed(const vector<int>& piles, int h) { int lo = 1, hi = *max_element(piles.begin(), piles.end()); int result = hi; while (lo <= hi) { int mid = lo + (hi - lo) / 2; long long hours = 0; for (int p : piles) hours += (p + mid - 1) / mid; if (hours <= h) { result = mid; hi = mid - 1; } else lo = mid + 1; } return result; }
searching.py
import math def koko_bananas(piles, h): """O(n * log(max_pile)) time""" def hours_needed(speed): return sum(math.ceil(p / speed) for p in piles) lo, hi = 1, max(piles) while lo < hi: mid = lo + (hi - lo) // 2 if hours_needed(mid) <= h: hi = mid # try slower else: lo = mid + 1 # need faster return lo
searching.rs
fn min_eating_speed(piles: &[i32], h: i32) -> i32 { let (mut left, mut right) = (1, *piles.iter().max().unwrap()); while left < right { let mid = left + (right - left) / 2; let hours: i64 = piles.iter() .map(|&p| ((p as i64) + (mid as i64) - 1) / (mid as i64)) .sum(); if hours <= h as i64 { right = mid; } else { left = mid + 1; } } left }

🚢 Problem 3: Ship Packages in D Days

Find the minimum ship capacity to ship all packages (in order) within D days. Greedily load packages until the next one would exceed capacity, then start a new day.

Search space: [max(weights), sum(weights)]. Condition: daysNeeded(capacity) <= D

BinarySearchOnAnswer.java
// O(n * log(sum_weights)) time static boolean canShip(int[] weights, int days, int capacity) { int currentLoad = 0, daysNeeded = 1; for (int w : weights) { if (w > capacity) return false; if (currentLoad + w > capacity) { daysNeeded++; currentLoad = 0; } currentLoad += w; } return daysNeeded <= days; } static int minShipCapacity(int[] weights, int days) { int lo = Arrays.stream(weights).max().getAsInt(); int hi = Arrays.stream(weights).sum(); while (lo < hi) { int mid = lo + (hi - lo) / 2; if (canShip(weights, days, mid)) hi = mid; else lo = mid + 1; } return lo; } // weights=[1..10], D=5 --> 15 // weights=[3,2,2,4,1,4], D=3 --> 6
searching.cpp
int shipWithinDays(const vector<int>& weights, int D) { int lo = *max_element(weights.begin(), weights.end()); int hi = accumulate(weights.begin(), weights.end(), 0); int result = hi; while (lo <= hi) { int mid = lo + (hi - lo) / 2; int days = 1, load = 0; for (int w : weights) { if (load + w > mid) { ++days; load = 0; } load += w; } if (days <= D) { result = mid; hi = mid - 1; } else lo = mid + 1; } return result; }
searching.py
def ship_packages(weights, days): """O(n * log(sum - max)) time""" def days_needed(capacity): d, current_load = 1, 0 for w in weights: if current_load + w > capacity: d += 1 current_load = 0 current_load += w return d lo, hi = max(weights), sum(weights) while lo < hi: mid = lo + (hi - lo) // 2 if days_needed(mid) <= days: hi = mid else: lo = mid + 1 return lo
searching.rs
fn min_ship_capacity(weights: &[i32], days: i32) -> i32 { let mut left = *weights.iter().max().unwrap(); let mut right: i32 = weights.iter().sum(); while left < right { let mid = left + (right - left) / 2; let mut days_needed = 1; let mut current_load = 0; for &w in weights { if current_load + w > mid { days_needed += 1; current_load = 0; } current_load += w; } if days_needed <= days { right = mid; } else { left = mid + 1; } } left }

📚 Summary

  • Binary search halves the search space at every step, giving O(log n) time on sorted data.
  • Always use mid = low + (high - low) / 2 to avoid integer overflow.
  • First/last occurrence variants continue searching even after finding a match, using the result variable pattern.
  • Rotated sorted array search leverages the fact that one half is always sorted.
  • Binary search on the answer is a powerful paradigm: apply it whenever you have a monotonic condition over a continuous range of possible answers.
  • The key template: define a condition(mid) function, binary search over the answer space, and return the boundary where the condition flips.
  • Problems like Koko bananas and ship packages combine BS on answer with a greedy O(n) feasibility check, yielding O(n log M) total time.
