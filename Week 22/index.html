Dijkstra's Algorithm

Dijkstra's algorithm finds the shortest path from a single source to all other vertices in a graph with non-negative edge weights. It is the go-to algorithm for GPS navigation, network routing (OSPF protocol), and game pathfinding.

Algorithm: Greedy + Priority Queue

Dijkstra works by greedily processing the nearest unvisited vertex. It maintains tentative distances and relaxes edges to find shorter paths.

Dijkstra Step-by-Step: Source = 0

Graph:  0 --4-- 1 --1-- 3      Step 1: Process 0 (dist=0)
        |       |       |        dist = [0, 4, 1, INF, INF]
        1       2       3
        |       |       |      Step 2: Process 2 (dist=1)
        2 --5-- 3       4        dist = [0, 3, 1, 6, INF]

                                Step 3: Process 1 (dist=3)
 Priority Queue progression:      dist = [0, 3, 1, 4, INF]
   [(0,0)]
   [(1,2), (4,1)]              Step 4: Process 3 (dist=4)
   [(3,1), (6,3)]                dist = [0, 3, 1, 4, 7]
   [(4,3)]
   [(7,4)]                     Step 5: Process 4 (dist=7)
                                 DONE. Final: [0, 3, 1, 4, 7]

Dijkstra processes vertices in order of increasing distance from source. Each vertex is processed exactly once.

Key Insight

Dijkstra fails with negative edges because it assumes once a vertex is settled, its distance is final. A negative edge could later provide a shorter path to an already-settled vertex. Use Bellman-Ford for negative edges.

// Dijkstra's Algorithm (Java)
static int[] dijkstra(List<int[]>[] adj, int src, int V) {
    int[] dist = new int[V];
    Arrays.fill(dist, INF);
    dist[src] = 0;
    // Min-heap: [distance, vertex]
    PriorityQueue<int[]> pq = new PriorityQueue<>(
        Comparator.comparingInt(a -> a[0]));
    pq.offer(new int[]{0, src});

    while (!pq.isEmpty()) {
        int[] curr = pq.poll();
        int d = curr[0], u = curr[1];
        if (d > dist[u]) continue; // stale entry
        for (int[] edge : adj[u]) {
            int v = edge[0], w = edge[1];
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.offer(new int[]{dist[v], v});
            }
        }
    }
    return dist;
}
// Dijkstra — single-source shortest paths (C++17)
std::vector<long long> dijkstra(
    const AdjList& adj, int src) {
    int n = adj.size();
    std::vector<long long> dist(n, LLONG_MAX);
    dist[src] = 0;

    using P = std::pair<long long, int>;
    std::priority_queue<P, std::vector<P>,
                        std::greater<>> pq;
    pq.push({0, src});

    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[u]) continue;
        for (auto [v, w] : adj[u]) {
            if (dist[u] + w < dist[v]) {
                dist[v] = dist[u] + w;
                pq.push({dist[v], v});
            }
        }
    }
    return dist;
}
# Dijkstra with path reconstruction (Python)
def dijkstra(adj, source, vertices):
    dist = {v: float('inf') for v in vertices}
    parent = {v: None for v in vertices}
    dist[source] = 0.0
    heap = [(0.0, source)]

    while heap:
        d, u = heapq.heappop(heap)
        if d > dist[u]:
            continue  # stale entry
        for v, w in adj[u]:
            new_dist = dist[u] + w
            if new_dist < dist[v]:
                dist[v] = new_dist
                parent[v] = u
                heapq.heappush(heap, (new_dist, v))
    return dist, parent
// Dijkstra (Rust) — uses BinaryHeap with Reverse for min-heap
fn dijkstra(adj: &[Vec<(usize, i64)>], src: usize) -> Vec<i64> {
    let n = adj.len();
    let mut dist = vec![i64::MAX; n];
    dist[src] = 0;

    let mut heap: BinaryHeap<Reverse<(i64, usize)>> =
        BinaryHeap::new();
    heap.push(Reverse((0, src)));

    while let Some(Reverse((d, u))) = heap.pop() {
        if d > dist[u] { continue; }
        for &(v, weight) in &adj[u] {
            let new_dist = dist[u] + weight;
            if new_dist < dist[v] {
                dist[v] = new_dist;
                heap.push(Reverse((new_dist, v)));
            }
        }
    }
    dist
}

Bellman-Ford Algorithm

Bellman-Ford handles negative edge weights and detects negative-weight cycles. It relaxes all edges V-1 times. After V-1 iterations, if any edge can still be relaxed, a negative cycle exists.

Why V-1 Iterations?

In a graph with V vertices, the shortest path between any two vertices uses at most V-1 edges (since a simple path visits each vertex at most once). After k iterations of relaxing all edges, the algorithm has found all shortest paths using at most k edges. So V-1 iterations are sufficient for correctness.

Key Insight

Bellman-Ford's negative cycle detection is used in finance: if currency exchange rates form a negative cycle, there's an arbitrage opportunity (you can make money by cycling through currencies). The algorithm can find such cycles in O(V*E).

// Bellman-Ford (Java)
static int[] bellmanFord(int V, int[][] edges, int src) {
    int[] dist = new int[V];
    Arrays.fill(dist, INF);
    dist[src] = 0;

    // Relax all edges V-1 times
    for (int i = 0; i < V - 1; i++) {
        for (int[] e : edges) {
            int u = e[0], v = e[1], w = e[2];
            if (dist[u] != INF && dist[u] + w < dist[v])
                dist[v] = dist[u] + w;
        }
    }
    // Check for negative weight cycles
    for (int[] e : edges) {
        if (dist[e[0]] != INF &&
            dist[e[0]] + e[2] < dist[e[1]]) {
            System.out.println("Negative cycle!");
            return null;
        }
    }
    return dist;
}
// Bellman-Ford (C++17) — returns nullopt on negative cycle
std::optional<std::vector<long long>>
bellman_ford(int n, const std::vector<Edge>& edges, int src) {
    std::vector<long long> dist(n, LLONG_MAX);
    dist[src] = 0;

    for (int i = 0; i < n - 1; ++i) {
        for (auto [u, v, w] : edges) {
            if (dist[u] != LLONG_MAX &&
                dist[u] + w < dist[v])
                dist[v] = dist[u] + w;
        }
    }
    for (auto [u, v, w] : edges)
        if (dist[u] != LLONG_MAX &&
            dist[u] + w < dist[v])
            return std::nullopt;
    return dist;
}
# Bellman-Ford with early termination (Python)
def bellman_ford(edges, source, vertices):
    dist = {v: float('inf') for v in vertices}
    parent = {v: None for v in vertices}
    dist[source] = 0.0
    n = len(vertices)

    for _ in range(n - 1):
        updated = False
        for u, v, w in edges:
            if dist[u] + w < dist[v]:
                dist[v] = dist[u] + w
                parent[v] = u
                updated = True
        if not updated:
            break  # early termination

    # Negative cycle check
    for u, v, w in edges:
        if dist[u] + w < dist[v]:
            return None, parent
    return dist, parent
// Bellman-Ford (Rust)
fn bellman_ford(edges: &[(usize, usize, i64)],
                n: usize, src: usize) -> Option<Vec<i64>> {
    let mut dist = vec![i64::MAX; n];
    dist[src] = 0;

    for _ in 0..n - 1 {
        for &(u, v, w) in edges {
            if dist[u] != i64::MAX &&
               dist[u] + w < dist[v] {
                dist[v] = dist[u] + w;
            }
        }
    }
    for &(u, v, w) in edges {
        if dist[u] != i64::MAX &&
           dist[u] + w < dist[v] {
            return None; // Negative cycle
        }
    }
    Some(dist)
}

Floyd-Warshall Algorithm

Floyd-Warshall computes all-pairs shortest paths using dynamic programming. The key recurrence: dist[i][j] = min(dist[i][j], dist[i][k] + dist[k][j]) for each intermediate vertex k.

The DP Intuition

Think of it as: "Can I find a shorter path from i to j by going through vertex k?" We try every possible intermediate vertex. The ordering (k outermost) is critical -- after processing k, we have found all shortest paths using vertices {0..k} as intermediaries.

Key Insight

Floyd-Warshall detects negative cycles by checking if dist[i][i] < 0 after completion. If the shortest path from a vertex to itself is negative, the graph contains a negative cycle. The algorithm's O(V^3) time is optimal for dense graphs since there are O(V^2) pairs to consider.

// Floyd-Warshall — all-pairs shortest paths (Java)
static int[][] floydWarshall(int V, int[][] graph) {
    int[][] dist = new int[V][V];
    for (int[] row : dist) Arrays.fill(row, INF);
    for (int i = 0; i < V; i++) dist[i][i] = 0;
    for (int[] e : graph)
        dist[e[0]][e[1]] = Math.min(dist[e[0]][e[1]], e[2]);

    for (int k = 0; k < V; k++)
        for (int i = 0; i < V; i++)
            for (int j = 0; j < V; j++)
                if (dist[i][k] != INF && dist[k][j] != INF)
                    dist[i][j] = Math.min(dist[i][j],
                                          dist[i][k] + dist[k][j]);
    return dist;
}
// Floyd-Warshall (C++17)
std::vector<std::vector<long long>>
floyd_warshall(int n, const std::vector<Edge>& edges) {
    std::vector dist(n, std::vector<long long>(n, FW_INF));
    for (int i = 0; i < n; ++i) dist[i][i] = 0;
    for (auto [u, v, w] : edges)
        dist[u][v] = std::min(dist[u][v], (long long)w);

    for (int k = 0; k < n; ++k)
        for (int i = 0; i < n; ++i)
            for (int j = 0; j < n; ++j)
                if (dist[i][k] < FW_INF &&
                    dist[k][j] < FW_INF)
                    dist[i][j] = std::min(dist[i][j],
                                 dist[i][k] + dist[k][j]);
    return dist;
}
# Floyd-Warshall with path reconstruction (Python)
def floyd_warshall(n, edges):
    INF = float('inf')
    dist = [[INF] * n for _ in range(n)]
    nxt = [[None] * n for _ in range(n)]

    for i in range(n): dist[i][i] = 0.0
    for u, v, w in edges:
        dist[u][v] = w
        nxt[u][v] = v

    for k in range(n):
        for i in range(n):
            for j in range(n):
                if dist[i][k] + dist[k][j] < dist[i][j]:
                    dist[i][j] = dist[i][k] + dist[k][j]
                    nxt[i][j] = nxt[i][k]
    return dist, nxt
// Floyd-Warshall (Rust)
fn floyd_warshall(adj_matrix: &[Vec<i64>]) -> Vec<Vec<i64>> {
    let n = adj_matrix.len();
    let mut dist = adj_matrix.to_vec();

    for k in 0..n {
        for i in 0..n {
            for j in 0..n {
                if dist[i][k] != i64::MAX &&
                   dist[k][j] != i64::MAX {
                    let through_k = dist[i][k] + dist[k][j];
                    if through_k < dist[i][j] {
                        dist[i][j] = through_k;
                    }
                }
            }
        }
    }
    dist
}

Disjoint Set Union (Union-Find)

DSU tracks disjoint sets and supports two operations: find (which set does x belong to?) and union (merge two sets). With path compression and union by rank, both operations run in amortized O(alpha(n)) -- the inverse Ackermann function, which is effectively constant for all practical inputs.

Two Key Optimizations

  • Path Compression: During find(x), make every node on the path point directly to the root. This flattens the tree for future queries.
  • Union by Rank: Always attach the shorter tree under the root of the taller tree. This keeps trees balanced.
// Union-Find / DSU (Java)
static class DSU {
    int[] parent, rank;
    DSU(int n) {
        parent = new int[n]; rank = new int[n];
        for (int i = 0; i < n; i++) parent[i] = i;
    }
    int find(int x) {
        if (parent[x] != x)
            parent[x] = find(parent[x]); // path compression
        return parent[x];
    }
    boolean union(int x, int y) {
        int px = find(x), py = find(y);
        if (px == py) return false;
        // Union by rank
        if (rank[px] < rank[py]) { int t=px; px=py; py=t; }
        parent[py] = px;
        if (rank[px] == rank[py]) rank[px]++;
        return true;
    }
}
// DSU / Union-Find (C++17)
class DSU {
public:
    explicit DSU(int n) : parent_(n), rank_(n, 0) {
        std::iota(parent_.begin(), parent_.end(), 0);
    }
    int find(int x) {
        if (parent_[x] != x)
            parent_[x] = find(parent_[x]);
        return parent_[x];
    }
    bool unite(int x, int y) {
        int px = find(x), py = find(y);
        if (px == py) return false;
        if (rank_[px] < rank_[py]) std::swap(px, py);
        parent_[py] = px;
        if (rank_[px] == rank_[py]) ++rank_[px];
        return true;
    }
};
# Union-Find with path compression and union by rank (Python)
class UnionFind:
    def __init__(self, n):
        self.parent = list(range(n))
        self.rank = [0] * n
        self.count = n  # number of components

    def find(self, x):
        if self.parent[x] != x:
            self.parent[x] = self.find(self.parent[x])
        return self.parent[x]

    def union(self, x, y):
        rx, ry = self.find(x), self.find(y)
        if rx == ry: return False
        if self.rank[rx] < self.rank[ry]:
            rx, ry = ry, rx
        self.parent[ry] = rx
        if self.rank[rx] == self.rank[ry]:
            self.rank[rx] += 1
        self.count -= 1
        return True
// DSU (Rust) — path compression + union by rank
struct DSU {
    parent: Vec<usize>,
    rank: Vec<usize>,
}
impl DSU {
    fn new(n: usize) -> Self {
        DSU { parent: (0..n).collect(), rank: vec![0; n] }
    }
    fn find(&mut self, x: usize) -> usize {
        if self.parent[x] != x {
            self.parent[x] = self.find(self.parent[x]);
        }
        self.parent[x]
    }
    fn union(&mut self, x: usize, y: usize) -> bool {
        let rx = self.find(x);
        let ry = self.find(y);
        if rx == ry { return false; }
        match self.rank[rx].cmp(&self.rank[ry]) {
            Less => self.parent[rx] = ry,
            Greater => self.parent[ry] = rx,
            Equal => {
                self.parent[ry] = rx;
                self.rank[rx] += 1;
            }
        }
        true
    }
}

Kruskal's MST

Kruskal's algorithm builds a Minimum Spanning Tree by sorting all edges by weight and greedily adding them if they don't form a cycle. Cycle detection uses Union-Find.

MST Construction (Kruskal):

Edges sorted by weight:
  (2,4,1)  (1,2,2)  (4,5,3)  (0,1,4)  (0,2,4)  (1,3,6)  ...

Step 1: Add 2-4 (w=1)  Components: {0},{1},{2,4},{3},{5}
Step 2: Add 1-2 (w=2)  Components: {0},{1,2,4},{3},{5}
Step 3: Add 4-5 (w=3)  Components: {0},{1,2,4,5},{3}
Step 4: Add 0-1 (w=4)  Components: {0,1,2,4,5},{3}
Step 5: Add 1-3 (w=6)  Components: {0,1,2,3,4,5}  DONE!

Total MST weight: 1+2+3+4+6 = 16

Kruskal's algorithm: sort edges, add lightest edge that doesn't create a cycle.

// Kruskal's MST (Java)
static int kruskalMST(int V, int[][] edges) {
    Arrays.sort(edges, (a, b) -> a[2] - b[2]);
    DSU dsu = new DSU(V);
    int mstWeight = 0, edgeCount = 0;
    for (int[] e : edges) {
        if (edgeCount == V - 1) break;
        if (dsu.union(e[0], e[1])) {
            mstWeight += e[2];
            edgeCount++;
        }
    }
    return mstWeight;
}
// Kruskal's MST (C++17)
std::pair<long long, std::vector<Edge>>
kruskal_mst(int n, std::vector<Edge> edges) {
    std::sort(edges.begin(), edges.end(),
        [](const Edge& a, const Edge& b) {
            return std::get<2>(a) < std::get<2>(b);
        });
    DSU dsu(n);
    long long total = 0;
    std::vector<Edge> mst_edges;
    for (auto& [u, v, w] : edges) {
        if (dsu.unite(u, v)) {
            total += w;
            mst_edges.emplace_back(u, v, w);
            if (mst_edges.size() == n - 1) break;
        }
    }
    return {total, mst_edges};
}
# Kruskal's MST (Python)
def kruskal_mst(n, edges):
    sorted_edges = sorted(edges, key=lambda e: e[2])
    uf = UnionFind(n)
    mst_weight = 0.0
    mst_edges = []
    for u, v, w in sorted_edges:
        if uf.union(u, v):
            mst_weight += w
            mst_edges.append((u, v, w))
            if len(mst_edges) == n - 1:
                break
    return mst_weight, mst_edges
// Kruskal's MST (Rust)
fn kruskal_mst(n: usize,
    edges: &[(usize, usize, i64)])
    -> (i64, Vec<(usize, usize, i64)>) {
    let mut sorted = edges.to_vec();
    sorted.sort_unstable_by_key(|&(_, _, w)| w);
    let mut dsu = DSU::new(n);
    let mut mst = Vec::new();
    let mut total = 0i64;
    for (u, v, w) in sorted {
        if dsu.union(u, v) {
            total += w;
            mst.push((u, v, w));
            if mst.len() == n - 1 { break; }
        }
    }
    (total, mst)
}

Prim's MST

Prim's grows the MST from a single vertex, greedily adding the cheapest edge connecting a new vertex to the existing MST. Uses a min-heap for efficiency.

Key Insight

Kruskal vs. Prim: Use Kruskal for sparse graphs (E is close to V) since it's dominated by sorting E edges. Use Prim for dense graphs (E is close to V^2) since its heap-based approach handles adjacency well. Both produce the same MST weight (MSTs are unique if all edge weights are distinct).

// Prim's MST (Java)
static int primMST(int V, List<int[]>[] adj) {
    boolean[] inMST = new boolean[V];
    int[] key = new int[V];
    Arrays.fill(key, Integer.MAX_VALUE);
    key[0] = 0;
    PriorityQueue<int[]> pq = new PriorityQueue<>(
        Comparator.comparingInt(a -> a[0]));
    pq.offer(new int[]{0, 0});
    int mstWeight = 0;

    while (!pq.isEmpty()) {
        int[] curr = pq.poll();
        int w = curr[0], u = curr[1];
        if (inMST[u]) continue;
        inMST[u] = true;
        mstWeight += w;
        for (int[] edge : adj[u]) {
            int v = edge[0], ew = edge[1];
            if (!inMST[v] && ew < key[v]) {
                key[v] = ew;
                pq.offer(new int[]{ew, v});
            }
        }
    }
    return mstWeight;
}
# Prim's MST (Python)
def prim_mst(adj, start, n):
    visited = [False] * n
    mst_weight = 0.0
    mst_edges = []
    heap = [(0.0, -1, start)]

    while heap and len(mst_edges) < n - 1:
        w, frm, to = heapq.heappop(heap)
        if visited[to]: continue
        visited[to] = True
        if frm != -1:
            mst_weight += w
            mst_edges.append((frm, to, w))
        for neighbor, weight in adj[to]:
            if not visited[neighbor]:
                heapq.heappush(heap, (weight, to, neighbor))
    return mst_weight, mst_edges

Summary & Cheat Sheet

Dijkstra

Non-negative weights. O((V+E) log V). Greedy + priority queue. Use for GPS, routing.

Bellman-Ford

Handles negative weights. Detects negative cycles. O(V*E). Use for currency arbitrage.

Floyd-Warshall

All-pairs shortest paths. O(V^3). Simple triple loop DP. Good for dense graphs with small V.

Union-Find

Near-O(1) find/union. Path compression + rank. Used in Kruskal's MST and connectivity queries.

Kruskal's MST

Sort edges + DSU. O(E log E). Better for sparse graphs.

Prim's MST

Priority queue from single vertex. O((V+E) log V). Better for dense graphs.
