👉 Pattern 1: Two Pointers

Use two pointers that move toward each other (or in the same direction) to solve problems on sorted arrays or linked lists in O(n) time instead of O(n²).

When to Use

  • Finding pairs in a sorted array (Two Sum II)
  • Removing duplicates in-place
  • Container with most water, trapping rain water
  • Palindrome checking
Two Sum in Sorted Array: target = 9 arr = [1, 2, 4, 5, 7, 8, 11] L R Step 1: arr[L] + arr[R] = 1 + 11 = 12 > 9 → move R left L R Step 2: arr[L] + arr[R] = 1 + 8 = 9 = target → FOUND! [0, 5]

def two_sum_sorted(nums, target):
    """Two Sum II - Input Array Is Sorted (LC 167)"""
    left, right = 0, len(nums) - 1
    while left < right:
        total = nums[left] + nums[right]
        if total == target:
            return [left, right]
        elif total < target:
            left += 1
        else:
            right -= 1
    return []

def three_sum(nums):
    """3Sum - Find all unique triplets that sum to zero (LC 15)"""
    nums.sort()
    result = []
    for i in range(len(nums) - 2):
        if i > 0 and nums[i] == nums[i-1]:
            continue  # skip duplicates
        left, right = i + 1, len(nums) - 1
        while left < right:
            total = nums[i] + nums[left] + nums[right]
            if total == 0:
                result.append([nums[i], nums[left], nums[right]])
                while left < right and nums[left] == nums[left+1]:
                    left += 1
                while left < right and nums[right] == nums[right-1]:
                    right -= 1
                left += 1
                right -= 1
            elif total < 0:
                left += 1
            else:
                right -= 1
    return result

def container_with_most_water(height):
    """Container With Most Water (LC 11)"""
    left, right = 0, len(height) - 1
    max_area = 0
    while left < right:
        area = min(height[left], height[right]) * (right - left)
        max_area = max(max_area, area)
        if height[left] < height[right]:
            left += 1
        else:
            right -= 1
    return max_area

import java.util.*;

public class TwoPointers {
    // Two Sum II (LC 167)
    public int[] twoSumSorted(int[] nums, int target) {
        int left = 0, right = nums.length - 1;
        while (left < right) {
            int sum = nums[left] + nums[right];
            if (sum == target) return new int[]{left, right};
            else if (sum < target) left++;
            else right--;
        }
        return new int[]{};
    }

    // 3Sum (LC 15)
    public List<List<Integer>> threeSum(int[] nums) {
        Arrays.sort(nums);
        List<List<Integer>> result = new ArrayList<>();
        for (int i = 0; i < nums.length - 2; i++) {
            if (i > 0 && nums[i] == nums[i-1]) continue;
            int left = i + 1, right = nums.length - 1;
            while (left < right) {
                int sum = nums[i] + nums[left] + nums[right];
                if (sum == 0) {
                    result.add(Arrays.asList(nums[i], nums[left], nums[right]));
                    while (left < right && nums[left] == nums[left+1]) left++;
                    while (left < right && nums[right] == nums[right-1]) right--;
                    left++; right--;
                } else if (sum < 0) left++;
                else right--;
            }
        }
        return result;
    }

    // Container With Most Water (LC 11)
    public int maxArea(int[] height) {
        int left = 0, right = height.length - 1, maxArea = 0;
        while (left < right) {
            int area = Math.min(height[left], height[right]) * (right - left);
            maxArea = Math.max(maxArea, area);
            if (height[left] < height[right]) left++;
            else right--;
        }
        return maxArea;
    }
}

            else { right -= 1; }

🔧 Pattern 2: Sliding Window

Maintain a window (subarray/substring) that expands or contracts to find an optimal solution. Converts O(n²) brute force into O(n).

Two Variants

  • Fixed-size window: Max sum subarray of size K
  • Variable-size window: Smallest subarray with sum ≥ target
Longest Substring Without Repeating Chars: "abcabcbb" Window: [a]bcabcbb len=1, max=1 Window: [ab]cabcbb len=2, max=2 Window: [abc]abcbb len=3, max=3 Window: a[bca]bcbb 'a' repeats, shrink left, len=3, max=3 Window: ab[cab]cbb 'b' repeats, shrink left, len=3, max=3 Window: abc[abc]bb 'c' repeats, shrink left, len=3, max=3 Window: abca[bc]bb 'a' removed, shrink, len=2 Window: abcab[cb]b shrink, len=2 Window: abcabc[b]b shrink, len=1 Answer: 3 ("abc")
sliding_window.py
def length_of_longest_substring(s):
    """Longest Substring Without Repeating Characters (LC 3)"""
    char_index = {}
    left = 0
    max_len = 0
    for right in range(len(s)):
        if s[right] in char_index and char_index[s[right]] >= left:
            left = char_index[s[right]] + 1
        char_index[s[right]] = right
        max_len = max(max_len, right - left + 1)
    return max_len

def min_window(s, t):
    """Minimum Window Substring (LC 76)"""
    from collections import Counter
    need = Counter(t)
    missing = len(t)
    left = 0
    best = (0, float('inf'))

    for right, char in enumerate(s):
        if need[char] > 0:
            missing -= 1
        need[char] -= 1

        while missing == 0:  # valid window
            if right - left < best[1] - best[0]:
                best = (left, right)
            need[s[left]] += 1
            if need[s[left]] > 0:
                missing += 1
            left += 1

    return "" if best[1] == float('inf') else s[best[0]:best[1]+1]

def max_sum_subarray_k(nums, k):
    """Maximum sum subarray of size k (fixed window)"""
    window_sum = sum(nums[:k])
    max_sum = window_sum
    for i in range(k, len(nums)):
        window_sum += nums[i] - nums[i - k]
        max_sum = max(max_sum, window_sum)
    return max_sum

import java.util.*;

public class SlidingWindow {
    // Longest Substring Without Repeating Characters (LC 3)
    public int lengthOfLongestSubstring(String s) {
        Map<Character, Integer> charIndex = new HashMap<>();
        int left = 0, maxLen = 0;
        for (int right = 0; right < s.length(); right++) {
            char c = s.charAt(right);
            if (charIndex.containsKey(c) && charIndex.get(c) >= left) {
                left = charIndex.get(c) + 1;
            }
            charIndex.put(c, right);
            maxLen = Math.max(maxLen, right - left + 1);
        }
        return maxLen;
    }

    // Minimum Window Substring (LC 76)
    public String minWindow(String s, String t) {
        int[] need = new int[128];
        for (char c : t.toCharArray()) need[c]++;
        int missing = t.length(), left = 0, bestL = 0, bestLen = Integer.MAX_VALUE;

        for (int right = 0; right < s.length(); right++) {
            if (need[s.charAt(right)]-- > 0) missing--;
            while (missing == 0) {
                if (right - left + 1 < bestLen) {
                    bestLen = right - left + 1;
                    bestL = left;
                }
                if (++need[s.charAt(left++)] > 0) missing++;
            }
        }
        return bestLen == Integer.MAX_VALUE ? "" : s.substring(bestL, bestL + bestLen);
    }
}

use std::collections::HashMap;

// Longest Substring Without Repeating Characters (LC 3)
fn length_of_longest_substring(s: &str) -> usize {
    let mut char_index: HashMap<char, usize> = HashMap::new();
    let mut left = 0;
    let mut max_len = 0;
    let chars: Vec<char> = s.chars().collect();

    for right in 0..chars.len() {
        if let Some(&prev) = char_index.get(&chars[right]) {
            if prev >= left { left = prev + 1; }
        }
        char_index.insert(chars[right], right);
        max_len = max_len.max(right - left + 1);
    }
    max_len
}

// Max sum subarray of size k
fn max_sum_subarray_k(nums: &[i32], k: usize) -> i32 {
    let mut window_sum: i32 = nums[..k].iter().sum();
    let mut max_sum = window_sum;
    for i in k..nums.len() {
        window_sum += nums[i] - nums[i - k];
        max_sum = max_sum.max(window_sum);
    }
    max_sum
}

🐇 Pattern 3: Fast & Slow Pointers (Floyd's)

Two pointers move at different speeds. The fast pointer moves 2 steps while the slow pointer moves 1 step. If there's a cycle, they will meet. Also used to find the middle of a linked list.

Applications

  • Cycle detection in linked lists
  • Finding the start of a cycle
  • Finding the middle of a linked list
  • Happy number detection
Linked List Cycle Detection: 1 → 2 → 3 → 4 → 5 ↑ | | ↓ 8 ← 7 ← 6 Step 1: slow=1, fast=1 Step 2: slow=2, fast=3 Step 3: slow=3, fast=5 Step 4: slow=4, fast=7 Step 5: slow=5, fast=3 Step 6: slow=6, fast=5 Step 7: slow=7, fast=7 → MEET! Cycle exists Finding cycle start: Reset slow to head, keep fast at meeting point Both move 1 step at a time → they meet at cycle start (node 3)

def has_cycle(head):
    """Linked List Cycle (LC 141)"""
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
        if slow == fast:
            return True
    return False

def detect_cycle_start(head):
    """Linked List Cycle II (LC 142) - Find cycle start"""
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
        if slow == fast:
            # Reset slow to head, both move 1 step
            slow = head
            while slow != fast:
                slow = slow.next
                fast = fast.next
            return slow  # cycle start
    return None

def find_middle(head):
    """Middle of Linked List (LC 876)"""
    slow = fast = head
    while fast and fast.next:
        slow = slow.next
        fast = fast.next.next
    return slow

def is_happy(n):
    """Happy Number (LC 202)"""
    def get_next(num):
        total = 0
        while num > 0:
            digit = num % 10
            total += digit * digit
            num //= 10
        return total

    slow = n
    fast = get_next(n)
    while fast != 1 and slow != fast:
        slow = get_next(slow)
        fast = get_next(get_next(fast))
    return fast == 1

// Find Duplicate Number (LC 287) - Floyd's on array
fn find_duplicate(nums: &[i32]) -> i32 {
    let mut slow = nums[0] as usize;
    let mut fast = nums[0] as usize;

    loop {
        slow = nums[slow] as usize;
        fast = nums[nums[fast] as usize] as usize;
        if slow == fast { break; }
    }

    slow = nums[0] as usize;
    while slow != fast {
        slow = nums[slow] as usize;
        fast = nums[fast] as usize;
    }
    slow as i32
}

📅 Pattern 4: Merge Intervals

Sort intervals by start time, then merge overlapping ones. This pattern applies to any problem involving ranges, scheduling, or overlapping periods.

Input intervals: [[1,3], [2,6], [8,10], [15,18]] Sort by start: [1,3] [2,6] [8,10] [15,18] Merge process: [1,3] + [2,6] → overlap (3 ≥ 2) → merge to [1,6] [1,6] + [8,10] → no overlap (6 < 8) → keep [1,6], start [8,10] [8,10] + [15,18] → no overlap → keep [8,10], start [15,18] Result: [[1,6], [8,10], [15,18]]
merge_intervals.py
def merge(intervals):
    """Merge Intervals (LC 56)"""
    intervals.sort(key=lambda x: x[0])
    merged = [intervals[0]]
    for start, end in intervals[1:]:
        if start <= merged[-1][1]:
            merged[-1][1] = max(merged[-1][1], end)
        else:
            merged.append([start, end])
    return merged

def insert(intervals, new_interval):
    """Insert Interval (LC 57)"""
    result = []
    i = 0
    # Add all intervals before new_interval
    while i < len(intervals) and intervals[i][1] < new_interval[0]:
        result.append(intervals[i])
        i += 1
    # Merge overlapping intervals
    while i < len(intervals) and intervals[i][0] <= new_interval[1]:
        new_interval[0] = min(new_interval[0], intervals[i][0])
        new_interval[1] = max(new_interval[1], intervals[i][1])
        i += 1
    result.append(new_interval)
    # Add remaining
    result.extend(intervals[i:])
    return result

def min_meeting_rooms(intervals):
    """Meeting Rooms II (LC 253)"""
    import heapq
    if not intervals:
        return 0
    intervals.sort(key=lambda x: x[0])
    heap = [intervals[0][1]]  # end times
    for start, end in intervals[1:]:
        if start >= heap[0]:
            heapq.heappop(heap)
        heapq.heappush(heap, end)
    return len(heap)

import java.util.*;

public class MergeIntervals {
    // Merge Intervals (LC 56)
    public int[][] merge(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> a[0] - b[0]);
        List<int[]> merged = new ArrayList<>();
        merged.add(intervals[0]);
        for (int i = 1; i < intervals.length; i++) {
            int[] last = merged.get(merged.size() - 1);
            if (intervals[i][0] <= last[1]) {
                last[1] = Math.max(last[1], intervals[i][1]);
            } else {
                merged.add(intervals[i]);
            }
        }
        return merged.toArray(new int[0][]);
    }

    // Meeting Rooms II (LC 253)
    public int minMeetingRooms(int[][] intervals) {
        Arrays.sort(intervals, (a, b) -> a[0] - b[0]);
        PriorityQueue<Integer> heap = new PriorityQueue<>();
        heap.offer(intervals[0][1]);
        for (int i = 1; i < intervals.length; i++) {
            if (intervals[i][0] >= heap.peek())
                heap.poll();
            heap.offer(intervals[i][1]);
        }
        return heap.size();
    }
}

    heap.push(intervals[0][1]);
    for (int i = 1; i < (int)intervals.size(); i++) {
        if (intervals[i][0] >= heap.top()) heap.pop();
        heap.push(intervals[i][1]);
    }
    return heap.size();
}

// Merge Intervals (LC 56)
fn merge(intervals: &mut Vec<Vec<i32>>) -> Vec<Vec<i32>> {
    intervals.sort();
    let mut merged: Vec<Vec<i32>> = vec![intervals[0].clone()];
    for interval in intervals.iter().skip(1) {
        let last = merged.last_mut().unwrap();
        if interval[0] <= last[1] {
            last[1] = last[1].max(interval[1]);
        } else {
            merged.push(interval.clone());
        }
    }
    merged
}

// Insert Interval (LC 57)
fn insert(intervals: Vec<Vec<i32>>, new_interval: Vec<i32>) -> Vec<Vec<i32>> {
    let mut result = Vec::new();
    let mut new_int = new_interval;
    let mut i = 0;

    while i < intervals.len() && intervals[i][1] < new_int[0] {
        result.push(intervals[i].clone());
        i += 1;
    }
    while i < intervals.len() && intervals[i][0] <= new_int[1] {
        new_int[0] = new_int[0].min(intervals[i][0]);
        new_int[1] = new_int[1].max(intervals[i][1]);
        i += 1;
    }
    result.push(new_int);
    while i < intervals.len() {
        result.push(intervals[i].clone());
        i += 1;
    }
    result
}

🏆 Pattern 5: Top-K Elements

Use a min-heap of size K to efficiently find the K largest elements, or a max-heap for the K smallest. This gives O(n log k) instead of O(n log n) from sorting.

Key Insight

For "K largest": maintain a min-heap of size K. The root is always the smallest among the K largest. For "K smallest": use a max-heap of size K.

Find 3 largest from: [3, 1, 5, 12, 2, 11] Min-heap (size 3): Add 3: [3] Add 1: [1, 3] Add 5: [1, 3, 5] Add 12: [3, 5, 12] (evict 1, since 12 > min=1) Add 2: [3, 5, 12] (skip 2, since 2 < min=3) Add 11: [5, 11, 12] (evict 3, since 11 > min=3) Result: heap contains [5, 11, 12] = top 3 largest Kth largest = heap root = 5
top_k.py
import heapq
from collections import Counter

def find_kth_largest(nums, k):
    """Kth Largest Element in an Array (LC 215)"""
    heap = []
    for num in nums:
        heapq.heappush(heap, num)
        if len(heap) > k:
            heapq.heappop(heap)
    return heap[0]

def top_k_frequent(nums, k):
    """Top K Frequent Elements (LC 347)"""
    count = Counter(nums)
    # Min-heap of size k based on frequency
    return heapq.nlargest(k, count.keys(), key=count.get)

def k_closest_points(points, k):
    """K Closest Points to Origin (LC 973)"""
    # Max-heap (negate distance) of size k
    heap = []
    for x, y in points:
        dist = -(x*x + y*y)  # negate for max-heap
        if len(heap) < k:
            heapq.heappush(heap, (dist, x, y))
        elif dist > heap[0][0]:
            heapq.heapreplace(heap, (dist, x, y))
    return [[x, y] for _, x, y in heap]

def find_median(nums):
    """Find Median from Data Stream (LC 295)"""
    small = []  # max-heap (negated)
    large = []  # min-heap

    for num in nums:
        heapq.heappush(small, -num)
        # Balance: move largest from small to large
        heapq.heappush(large, -heapq.heappop(small))
        # Keep small >= large in size
        if len(large) > len(small):
            heapq.heappush(small, -heapq.heappop(large))

    if len(small) > len(large):
        return -small[0]
    return (-small[0] + large[0]) / 2.0

import java.util.*;

public class TopK {
    // Kth Largest Element (LC 215)
    public int findKthLargest(int[] nums, int k) {
        PriorityQueue<Integer> heap = new PriorityQueue<>();
        for (int num : nums) {
            heap.offer(num);
            if (heap.size() > k) heap.poll();
        }
        return heap.peek();
    }

    // Top K Frequent Elements (LC 347)
    public int[] topKFrequent(int[] nums, int k) {
        Map<Integer, Integer> count = new HashMap<>();
        for (int n : nums) count.merge(n, 1, Integer::sum);

        PriorityQueue<int[]> heap = new PriorityQueue<>((a,b) -> a[1] - b[1]);
        for (var e : count.entrySet()) {
            heap.offer(new int[]{e.getKey(), e.getValue()});
            if (heap.size() > k) heap.poll();
        }
        return heap.stream().mapToInt(a -> a[0]).toArray();
    }

    // Find Median from Data Stream (LC 295)
    PriorityQueue<Integer> small = new PriorityQueue<>(Collections.reverseOrder());
    PriorityQueue<Integer> large = new PriorityQueue<>();

    public void addNum(int num) {
        small.offer(num);
        large.offer(small.poll());
        if (large.size() > small.size())
            small.offer(large.poll());
    }

    public double findMedian() {
        if (small.size() > large.size()) return small.peek();
        return (small.peek() + large.peek()) / 2.0;
    }
}

int findKthLargest(vector<int>& nums, int k) {
    priority_queue<int, vector<int>, greater<int>> heap;

        heap.push(num);
        if ((int)heap.size() > k) heap.pop();
    }
    return heap.top();
}

// Top K Frequent Elements (LC 347)
vector<int> topKFrequent(vector<int>& nums, int k) {
    unordered_map<int, int> count;
    for (int n : nums) count[n]++;

    auto cmp = [](pair<int,int>& a, pair<int,int>& b) {
        return a.second > b.second;
    };
    priority_queue<pair<int,int>, vector<pair<int,int>>, decltype(cmp)> heap(cmp);

    for (auto& [num, freq] : count) {
        heap.push({num, freq});
        if ((int)heap.size() > k) heap.pop();
    }
    vector<int> result;
    while (!heap.empty()) {
        result.push_back(heap.top().first);
        heap.pop();
    }
    return result;
}

use std::collections::{BinaryHeap, HashMap};
use std::cmp::Reverse;

// Kth Largest Element (LC 215)
fn find_kth_largest(nums: &[i32], k: usize) -> i32 {
    let mut heap: BinaryHeap<Reverse<i32>> = BinaryHeap::new();
    for &num in nums {
        heap.push(Reverse(num));
        if heap.len() > k { heap.pop(); }
    }
    heap.peek().unwrap().0
}

// Top K Frequent Elements (LC 347)
fn top_k_frequent(nums: &[i32], k: usize) -> Vec<i32> {
    let mut count: HashMap<i32, usize> = HashMap::new();
    for &n in nums { *count.entry(n).or_insert(0) += 1; }

    let mut heap: BinaryHeap<Reverse<(usize, i32)>> = BinaryHeap::new();
    for (&num, &freq) in &count {
        heap.push(Reverse((freq, num)));
        if heap.len() > k { heap.pop(); }
    }
    heap.into_iter().map(|Reverse((_, n))| n).collect()
}
Pattern Recognition
Whenever you see "K largest", "K smallest", "K most frequent", or "K closest", think heap. Use a min-heap of size K for K largest, and a max-heap of size K for K smallest.
