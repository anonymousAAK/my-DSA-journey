🔍 Z-Algorithm

The Z-Algorithm computes the Z-array for a string S: Z[i] is the length of the longest substring starting at position i that matches a prefix of S. By concatenating the pattern P with the text T as P$T, any Z[i] == len(P) indicates a match.

Text: a a b x a a b Z-array: [7 1 0 0 3 1 0] Pattern match using "aab$aabxaab": P = "aab", T = "aabxaab" Concat: aab$aabxaab Z: [- 1 0 0 3 1 0 0 3 1 0] ^ ^ Z[i]=3=len(P) → match at position i - len(P) - 1
StringAlgorithms.java
static int[] zFunction(String s) { int n = s.length(); int[] z = new int[n]; int l = 0, r = 0; for (int i = 1; i < n; i++) { if (i < r) z[i] = Math.min(r - i, z[i - l]); while (i + z[i] < n && s.charAt(z[i]) == s.charAt(i + z[i])) z[i]++; if (i + z[i] > r) { l = i; r = i + z[i]; } } return z; } static List<Integer> zSearch(String text, String pattern) { String concat = pattern + "$" + text; int[] z = zFunction(concat); List<Integer> result = new ArrayList<>(); for (int i = pattern.length() + 1; i < concat.length(); i++) if (z[i] == pattern.length()) result.add(i - pattern.length() - 1); return result; }

🔢 Rabin-Karp (Rolling Hash)

Rabin-Karp uses a rolling hash to compare pattern and text windows in O(1) per slide. If hashes match, verify character-by-character. Excellent for multi-pattern search since you can check multiple pattern hashes simultaneously.

Pattern: "abc" hash(P) = 294 Text: "x a b c d a b c y" |---| hash("xab") = 311 ≠ 294 |---| hash("abc") = 294 = 294 → verify match! |---| hash("bcd") = 297 ≠ 294 |---| hash("cda") = 298 ≠ 294 |---| hash("dab") = 295 ≠ 294 |---| hash("abc") = 294 = 294 → verify match! |---| hash("bcy") = 320 ≠ 294 Rolling update: hash(s[i+1..i+m]) = (hash - s[i]*base^(m-1)) * base + s[i+m]
StringAlgorithms.java
static List<Integer> rabinKarp(String text, String pattern) { long MOD = 1_000_000_007, BASE = 31; int n = text.length(), m = pattern.length(); List<Integer> result = new ArrayList<>(); if (m > n) return result; long patHash = 0, textHash = 0, power = 1; for (int i = 0; i < m; i++) { patHash = (patHash * BASE + pattern.charAt(i)) % MOD; textHash = (textHash * BASE + text.charAt(i)) % MOD; if (i > 0) power = power * BASE % MOD; } for (int i = 0; i <= n - m; i++) { if (patHash == textHash) { if (text.substring(i, i + m).equals(pattern)) result.add(i); } if (i + m < n) { textHash = (textHash - text.charAt(i) * power % MOD + MOD) % MOD; textHash = (textHash * BASE + text.charAt(i + m)) % MOD; } } return result; }

fn rabin_karp(text: &str, pattern: &str) -> Vec<usize> { let (n, m) = (text.len(), pattern.len()); let (t, p) = (text.as_bytes(), pattern.as_bytes()); let (modv, base): (u64, u64) = (1_000_000_007, 31); let mut result = Vec::new(); if m > n { return result; } let mut pat_hash: u64 = 0; let mut text_hash: u64 = 0; let mut power: u64 = 1; for i in 0..m { pat_hash = (pat_hash * base + p[i] as u64) % modv; text_hash = (text_hash * base + t[i] as u64) % modv; if i > 0 { power = power * base % modv; } } for i in 0..=n - m { if pat_hash == text_hash && &t[i..i + m] == p { result.push(i); } if i + m < n { text_hash = (text_hash + modv - t[i] as u64 * power % modv) % modv; text_hash = (text_hash * base + t[i + m] as u64) % modv; } } result }

🌳 Trie-Based Pattern Matching

A Trie (prefix tree) stores a set of patterns in a tree structure. Each path from root to a marked node represents a pattern. Searching for a pattern of length m takes O(m) time regardless of how many patterns are stored.

Patterns: {"app", "apple", "apt", "bat"} (root) / \ a b / \ p a / \ \ p t* t* | l | e* * = end of word Search "apt": root → a → p → t* → FOUND Search "api": root → a → p → i → NOT FOUND (no 'i' child)
StringAlgorithms.java
class Trie { int[][] children; boolean[] isEnd; int cnt; Trie(int maxNodes) { children = new int[maxNodes][26]; isEnd = new boolean[maxNodes]; for (int[] row : children) Arrays.fill(row, -1); cnt = 1; // root is node 0 } void insert(String word) { int node = 0; for (char c : word.toCharArray()) { int idx = c - 'a'; if (children[node][idx] == -1) children[node][idx] = cnt++; node = children[node][idx]; } isEnd[node] = true; } boolean search(String word) { int node = 0; for (char c : word.toCharArray()) { int idx = c - 'a'; if (children[node][idx] == -1) return false; node = children[node][idx]; } return isEnd[node]; } // Find all patterns present in text List<int[]> searchInText(String text, String[] patterns) { for (String p : patterns) insert(p); List<int[]> matches = new ArrayList<>(); for (int i = 0; i < text.length(); i++) { int node = 0; for (int j = i; j < text.length(); j++) { int idx = text.charAt(j) - 'a'; if (children[node][idx] == -1) break; node = children[node][idx]; if (isEnd[node]) matches.add(new int[]{i, j}); } } return matches; } }

📜 Suffix Array Construction

A suffix array is a sorted array of all suffixes of a string, represented by their starting indices. It enables O(m log n) pattern search and is the foundation for many advanced string algorithms. Construction can be done in O(n log n) using prefix doubling.

String: "banana$" All suffixes: Sorted suffixes: 0: banana$ 6: $ 1: anana$ 5: a$ 2: nana$ 3: ana$ 3: ana$ 1: anana$ 4: na$ 0: banana$ 5: a$ 4: na$ 6: $ 2: nana$ Suffix Array: [6, 5, 3, 1, 0, 4, 2] Binary search for "ana": found at SA positions 2,3 → indices 3, 1
StringAlgorithms.java
// O(n log^2 n) suffix array construction static int[] buildSuffixArray(String s) { int n = s.length(); Integer[] sa = new Integer[n]; int[] rank = new int[n], tmp = new int[n]; for (int i = 0; i < n; i++) { sa[i] = i; rank[i] = s.charAt(i); } for (int k = 1; k < n; k *= 2) { final int kk = k; final int[] rr = rank.clone(); Comparator<Integer> cmp = (a, b) -> { if (rr[a] != rr[b]) return rr[a] - rr[b]; int ra = a + kk < n ? rr[a + kk] : -1; int rb = b + kk < n ? rr[b + kk] : -1; return ra - rb; }; Arrays.sort(sa, cmp); tmp[sa[0]] = 0; for (int i = 1; i < n; i++) tmp[sa[i]] = tmp[sa[i - 1]] + (cmp.compare(sa[i], sa[i - 1]) != 0 ? 1 : 0); System.arraycopy(tmp, 0, rank, 0, n); } int[] result = new int[n]; for (int i = 0; i < n; i++) result[i] = sa[i]; return result; }

💎 Manacher's Algorithm

Finds the longest palindromic substring in O(n) time. The key insight: palindromes mirror around their center, so previously computed palindrome lengths help us skip redundant checks.

Input: "abacaba" Transform (insert '#' between chars): #a#b#a#c#a#b#a# Palindrome radii (p[i] = radius of longest palindrome centered at i): # a # b # a # c # a # b # a # 0 1 0 1 0 1 0 7 0 1 0 1 0 1 0 Center at 'c' (index 7): p[7] = 7 Longest palindromic substring: "abacaba" (length = p[7] = 7)
StringAlgorithms.java
static String manacher(String s) { // Transform: "abc" -> "#a#b#c#" StringBuilder sb = new StringBuilder("#"); for (char c : s.toCharArray()) { sb.append(c); sb.append('#'); } String t = sb.toString(); int n = t.length(); int[] p = new int[n]; int center = 0, right = 0; for (int i = 0; i < n; i++) { int mirror = 2 * center - i; if (i < right) p[i] = Math.min(right - i, p[mirror]); while (i + p[i] + 1 < n && i - p[i] - 1 >= 0 && t.charAt(i + p[i] + 1) == t.charAt(i - p[i] - 1)) p[i]++; if (i + p[i] > right) { center = i; right = i + p[i]; } } int maxLen = 0, maxCenter = 0; for (int i = 0; i < n; i++) { if (p[i] > maxLen) { maxLen = p[i]; maxCenter = i; } } int start = (maxCenter - maxLen) / 2; return s.substring(start, start + maxLen); }

🔑 String Hashing Techniques

Polynomial hashing maps a string to an integer: hash(s) = s[0]*p^(n-1) + s[1]*p^(n-2) + ... + s[n-1]. With prefix hashes, any substring hash can be computed in O(1), enabling fast string comparison.

Double Hashing
Use two different (base, mod) pairs to reduce collision probability. The chance of two different strings having the same hash with one pair is ~1/MOD. With two pairs it drops to ~1/MOD^2, making collisions virtually impossible.
StringAlgorithms.java
class StringHasher { long[] h, pw; long MOD = 1_000_000_007, BASE = 31; StringHasher(String s) { int n = s.length(); h = new long[n + 1]; pw = new long[n + 1]; pw[0] = 1; for (int i = 0; i < n; i++) { h[i + 1] = (h[i] * BASE + s.charAt(i) - 'a' + 1) % MOD; pw[i + 1] = pw[i] * BASE % MOD; } } // Get hash of s[l..r] (inclusive) long getHash(int l, int r) { return (h[r + 1] - h[l] * pw[r - l + 1] % MOD + MOD * MOD) % MOD; } // Compare two substrings in O(1) boolean equal(int l1, int r1, int l2, int r2) { return getHash(l1, r1) == getHash(l2, r2); } }
