Segment Tree

A Segment Tree is a binary tree where each node stores aggregate information (sum, min, max) about a range of the underlying array. Leaf nodes correspond to individual elements, and internal nodes store the merged result of their children. This allows both range queries and point updates in O(log n).

How It Works: An Analogy

Think of a corporate hierarchy. Each manager (internal node) knows the total sales (sum) of their division. If one salesperson updates their numbers, the change ripples up through their chain of managers -- but only their chain, not the entire company. This is why updates are O(log n): only O(log n) nodes on the path from leaf to root need updating.

              [36]                  Array: [1, 3, 5, 7, 9, 11]
            /      \
         [9]        [27]            Node stores sum of its range
        /    \      /    \
      [4]   [5]  [16]   [11]       Leaf nodes = array elements
      / \    |    / \      |
    [1] [3] [5] [7] [9]  [11]

    Query sum(1,3) = ?
    Navigate tree: [9].right + [16].left = 5 + 7 = ...
    Actually: recurse, get 3+5+7 = 15   O(log n)

Segment Tree for range sum queries. Each node stores the sum of its children.

Key Operations

Build: Recursively build bottom-up. Each leaf gets an array value; each internal node sums its two children. Time: O(n).

Query(l, r): Three cases at each node: (1) no overlap -- return identity, (2) full overlap -- return stored value, (3) partial overlap -- recurse on both children. Time: O(log n).

Update(idx, val): Walk from root to the leaf at idx, update the leaf, then recalculate sums back up. Time: O(log n).

OperationTimeSpace
BuildO(n)O(4n)
Range QueryO(log n)--
Point UpdateO(log n)--
Key Insight

The tree is stored in a flat array of size 4n. Node i has children at 2i+1 and 2i+2 (0-indexed) or 2i and 2i+1 (1-indexed). This avoids pointer overhead and enables cache-friendly traversal. The 4n size guarantees enough space for all levels even when n is not a power of 2.

// Segment Tree for range sum queries (Java)
static class SegmentTree {
    int[] tree;
    int n;

    SegmentTree(int[] arr) {
        n = arr.length;
        tree = new int[4 * n];
        build(arr, 0, 0, n - 1);
    }

    void build(int[] arr, int node, int start, int end) {
        if (start == end) { tree[node] = arr[start]; return; }
        int mid = (start + end) / 2;
        build(arr, 2*node+1, start, mid);
        build(arr, 2*node+2, mid+1, end);
        tree[node] = tree[2*node+1] + tree[2*node+2];
    }

    // Range sum query [l, r]
    int query(int node, int start, int end, int l, int r) {
        if (r < start || end < l) return 0;           // no overlap
        if (l <= start && end <= r) return tree[node]; // full overlap
        int mid = (start + end) / 2;
        return query(2*node+1, start, mid, l, r)
             + query(2*node+2, mid+1, end, l, r);
    }

    // Point update: set arr[idx] = val
    void update(int node, int start, int end, int idx, int val) {
        if (start == end) { tree[node] = val; return; }
        int mid = (start + end) / 2;
        if (idx <= mid) update(2*node+1, start, mid, idx, val);
        else update(2*node+2, mid+1, end, idx, val);
        tree[node] = tree[2*node+1] + tree[2*node+2];
    }
}
// Segment Tree — range sum queries + point updates (C++17)
class SegmentTree {
public:
    explicit SegmentTree(const std::vector<int>& data)
        : n_(data.size()), tree_(4 * data.size(), 0) {
        if (n_ > 0) build(data, 1, 0, n_ - 1);
    }

    long long range_query(int l, int r) const {
        return query(1, 0, n_ - 1, l, r);
    }

    void point_update(int idx, int delta) {
        update(1, 0, n_ - 1, idx, delta);
    }

private:
    int n_;
    std::vector<long long> tree_;

    void build(const std::vector<int>& data, int node, int s, int e) {
        if (s == e) { tree_[node] = data[s]; return; }
        int mid = (s + e) / 2;
        build(data, 2*node, s, mid);
        build(data, 2*node+1, mid+1, e);
        tree_[node] = tree_[2*node] + tree_[2*node+1];
    }

    long long query(int node, int s, int e, int l, int r) const {
        if (r < s || e < l) return 0;
        if (l <= s && e <= r) return tree_[node];
        int mid = (s + e) / 2;
        return query(2*node, s, mid, l, r)
             + query(2*node+1, mid+1, e, l, r);
    }

    void update(int node, int s, int e, int idx, int delta) {
        if (s == e) { tree_[node] += delta; return; }
        int mid = (s + e) / 2;
        if (idx <= mid) update(2*node, s, mid, idx, delta);
        else            update(2*node+1, mid+1, e, idx, delta);
        tree_[node] = tree_[2*node] + tree_[2*node+1];
    }
};
# Segment Tree — range sum query + point update (Python)
class SegmentTree:
    def __init__(self, arr):
        self._n = len(arr)
        self._tree = [0] * (4 * self._n)
        self._arr = arr[:]
        if self._n > 0:
            self._build(1, 0, self._n - 1)

    def _build(self, node, start, end):
        if start == end:
            self._tree[node] = self._arr[start]; return
        mid = (start + end) // 2
        self._build(2 * node, start, mid)
        self._build(2 * node + 1, mid + 1, end)
        self._tree[node] = self._tree[2*node] + self._tree[2*node+1]

    def query(self, l, r):
        return self._query(1, 0, self._n - 1, l, r)

    def _query(self, node, start, end, l, r):
        if r < start or end < l: return 0
        if l <= start and end <= r: return self._tree[node]
        mid = (start + end) // 2
        return (self._query(2*node, start, mid, l, r)
              + self._query(2*node+1, mid+1, end, l, r))

    def update(self, idx, val):
        self._arr[idx] = val
        self._update(1, 0, self._n - 1, idx, val)

    def _update(self, node, start, end, idx, val):
        if start == end:
            self._tree[node] = val; return
        mid = (start + end) // 2
        if idx <= mid: self._update(2*node, start, mid, idx, val)
        else: self._update(2*node+1, mid+1, end, idx, val)
        self._tree[node] = self._tree[2*node] + self._tree[2*node+1]
// Segment Tree — Rust
struct SegmentTree {
    tree: Vec<i64>,
    n: usize,
}

impl SegmentTree {
    fn build(data: &[i64]) -> Self {
        let n = data.len();
        let mut tree = vec![0i64; 4 * n];
        if n > 0 { Self::build_helper(&mut tree, data, 0, 0, n-1); }
        SegmentTree { tree, n }
    }

    fn query(&self, ql: usize, qr: usize) -> i64 {
        self.query_helper(0, 0, self.n - 1, ql, qr)
    }

    fn update(&mut self, idx: usize, val: i64) {
        self.update_helper(0, 0, self.n - 1, idx, val);
    }

    fn query_helper(&self, node: usize, lo: usize, hi: usize,
                     ql: usize, qr: usize) -> i64 {
        if qr < lo || hi < ql { return 0; }
        if ql <= lo && hi <= qr { return self.tree[node]; }
        let mid = lo + (hi - lo) / 2;
        self.query_helper(2*node+1, lo, mid, ql, qr)
          + self.query_helper(2*node+2, mid+1, hi, ql, qr)
    }
}

Fenwick Tree (Binary Indexed Tree)

A Fenwick Tree (BIT) is a simpler, more space-efficient alternative to Segment Trees for prefix sum queries and point updates. It exploits binary representation of indices -- each index i is responsible for a range of elements determined by its lowest set bit.

The Bit Trick: i & (-i)

The expression i & (-i) isolates the lowest set bit of i. This single operation drives both traversal directions:

  • Query (walk down): Subtract the lowest set bit: i -= i & (-i)
  • Update (walk up): Add the lowest set bit: i += i & (-i)
Index (1-based):  1    2    3    4    5    6    7    8
Binary:          001  010  011  100  101  110  111  1000
Lowest set bit:   1    2    1    4    1    2    1    8
Responsibility:  [1]  [1-2] [3] [1-4] [5] [5-6] [7] [1-8]

                    BIT[8]: covers [1..8]
                   /
            BIT[4]: [1..4]      BIT[6]: [5..6]
           /           \          |
     BIT[2]: [1..2]  BIT[3]: [3]  BIT[5]: [5]  BIT[7]: [7]
        |
     BIT[1]: [1]

prefix_sum(7): BIT[7] + BIT[6] + BIT[4] = walk: 111->110->100
update(3):     BIT[3] -> BIT[4] -> BIT[8] = walk: 011->100->1000

Each BIT index stores the sum of a range determined by its lowest set bit. Queries and updates follow bit patterns.

OperationTimeSpace
Build from arrayO(n log n) naive / O(n) optimizedO(n)
Prefix SumO(log n)--
Point UpdateO(log n)--
Range Sum [l, r]O(log n)--
Key Insight

BIT uses half the space of a Segment Tree (O(n) vs O(4n)) and has smaller constants. Use BIT when you only need prefix sums + point updates. Use Segment Tree when you need range min/max, lazy propagation, or more complex operations.

// Binary Indexed Tree / Fenwick Tree (Java)
static class BIT {
    int[] bit;
    int n;

    BIT(int n) { this.n = n; bit = new int[n + 1]; } // 1-indexed

    BIT(int[] arr) {
        n = arr.length;
        bit = new int[n + 1];
        for (int i = 0; i < n; i++) update(i + 1, arr[i]);
    }

    // Add delta to index i (1-indexed)
    void update(int i, int delta) {
        for (; i <= n; i += i & (-i)) bit[i] += delta;
    }

    // Prefix sum [1..i]
    int prefixSum(int i) {
        int sum = 0;
        for (; i > 0; i -= i & (-i)) sum += bit[i];
        return sum;
    }

    // Range sum [l, r] (1-indexed)
    int rangeSum(int l, int r) {
        return prefixSum(r) - prefixSum(l - 1);
    }
}
// Fenwick Tree (BIT) — 1-indexed (C++17)
class FenwickTree {
public:
    explicit FenwickTree(int n) : n_(n), tree_(n + 1, 0) {}

    explicit FenwickTree(const std::vector<int>& data)
        : n_(data.size()), tree_(data.size() + 1, 0) {
        for (int i = 0; i < n_; ++i) update(i + 1, data[i]);
    }

    void update(int i, int delta) {
        for (; i <= n_; i += i & (-i)) tree_[i] += delta;
    }

    long long prefix_sum(int i) const {
        long long s = 0;
        for (int x = i; x > 0; x -= x & (-x)) s += tree_[x];
        return s;
    }

    long long range_sum(int l, int r) const {
        return prefix_sum(r) - prefix_sum(l - 1);
    }
};
# Fenwick Tree / Binary Indexed Tree (Python)
class FenwickTree:
    def __init__(self, n):
        self._n = n
        self._tree = [0] * (n + 1)  # 1-indexed

    @classmethod
    def from_array(cls, arr):
        """Build Fenwick tree from array in O(n)."""
        bit = cls(len(arr))
        for i in range(len(arr)):
            bit._tree[i + 1] = arr[i]
        for i in range(1, len(arr) + 1):
            j = i + (i & -i)
            if j <= len(arr):
                bit._tree[j] += bit._tree[i]
        return bit

    def update(self, idx, delta):
        i = idx + 1
        while i <= self._n:
            self._tree[i] += delta
            i += i & (-i)

    def prefix_sum(self, idx):
        s, i = 0, idx + 1
        while i > 0:
            s += self._tree[i]
            i -= i & (-i)
        return s

    def range_sum(self, l, r):
        if l == 0: return self.prefix_sum(r)
        return self.prefix_sum(r) - self.prefix_sum(l - 1)
// Fenwick Tree (Rust) — 0-indexed user API, 1-indexed internally
struct FenwickTree {
    tree: Vec<i64>,
    n: usize,
}

impl FenwickTree {
    fn new(n: usize) -> Self {
        FenwickTree { tree: vec![0i64; n + 1], n }
    }

    fn update(&mut self, idx: usize, delta: i64) {
        let mut i = idx + 1;
        while i <= self.n {
            self.tree[i] += delta;
            i += i & i.wrapping_neg();
        }
    }

    fn prefix_sum(&self, idx: usize) -> i64 {
        let mut sum = 0i64;
        let mut i = idx + 1;
        while i > 0 {
            sum += self.tree[i];
            i -= i & i.wrapping_neg();
        }
        sum
    }

    fn range_sum(&self, l: usize, r: usize) -> i64 {
        if l == 0 { self.prefix_sum(r) }
        else { self.prefix_sum(r) - self.prefix_sum(l - 1) }
    }
}

Trie (Prefix Tree)

A Trie is a tree-shaped data structure for storing strings where each node represents a single character. A path from root to a marked node spells a complete word. Tries excel at prefix-based operations: autocomplete, spell checking, and IP routing (longest prefix match).

Key Insight

Tries trade space for time. For English lowercase letters, each node has up to 26 children (array-based) or uses a hash map (more flexible, slightly slower). The choice between array vs. hash map depends on alphabet size and density. For competitive programming with lowercase English, array is preferred for speed.

// Trie with insert, search, prefix, autocomplete, delete (Java)
static class TrieNode {
    TrieNode[] children = new TrieNode[26];
    boolean isEnd = false;
    int wordCount = 0;
}

static class TrieImpl {
    TrieNode root = new TrieNode();

    void insert(String word) {
        TrieNode curr = root;
        for (char c : word.toCharArray()) {
            int idx = c - 'a';
            if (curr.children[idx] == null)
                curr.children[idx] = new TrieNode();
            curr = curr.children[idx];
            curr.wordCount++;
        }
        curr.isEnd = true;
    }

    boolean search(String word) {
        TrieNode curr = root;
        for (char c : word.toCharArray()) {
            int idx = c - 'a';
            if (curr.children[idx] == null) return false;
            curr = curr.children[idx];
        }
        return curr.isEnd;
    }

    boolean startsWith(String prefix) {
        TrieNode curr = root;
        for (char c : prefix.toCharArray()) {
            int idx = c - 'a';
            if (curr.children[idx] == null) return false;
            curr = curr.children[idx];
        }
        return true;
    }

    List<String> autocomplete(String prefix) {
        List<String> result = new ArrayList<>();
        TrieNode curr = root;
        for (char c : prefix.toCharArray()) {
            if (curr.children[c-'a'] == null) return result;
            curr = curr.children[c-'a'];
        }
        dfsCollect(curr, new StringBuilder(prefix), result);
        return result;
    }
}
// Trie — prefix tree for lowercase English letters (C++17)
class Trie {
public:
    Trie() : root_(std::make_unique<Node>()) {}

    void insert(const std::string& word) {
        Node* cur = root_.get();
        for (char ch : word) {
            int idx = ch - 'a';
            if (!cur->children[idx])
                cur->children[idx] = std::make_unique<Node>();
            cur = cur->children[idx].get();
        }
        cur->is_end = true;
    }

    bool search(const std::string& word) const {
        const Node* node = find_node(word);
        return node != nullptr && node->is_end;
    }

    bool starts_with(const std::string& prefix) const {
        return find_node(prefix) != nullptr;
    }

private:
    static constexpr int ALPHA = 26;
    struct Node {
        std::unique_ptr<Node> children[ALPHA]{};
        bool is_end = false;
    };
    std::unique_ptr<Node> root_;

    const Node* find_node(const std::string& s) const {
        const Node* cur = root_.get();
        for (char ch : s) {
            int idx = ch - 'a';
            if (!cur->children[idx]) return nullptr;
            cur = cur->children[idx].get();
        }
        return cur;
    }
};

class _TrieNode:
    __slots__ = ("children", "is_end")
    def __init__(self):
        self.children = {}
        self.is_end = False

class Trie:
    def __init__(self):
        self._root = _TrieNode()

    def insert(self, word):
        node = self._root
        for ch in word:
            if ch not in node.children:
                node.children[ch] = _TrieNode()
            node = node.children[ch]
        node.is_end = True

    def search(self, word):
        node = self._find_node(word)
        return node is not None and node.is_end

    def starts_with(self, prefix):
        return self._find_node(prefix) is not None

    def autocomplete(self, prefix):
        node = self._find_node(prefix)
        if node is None: return []
        results = []
        self._collect(node, list(prefix), results)
        return results

    def delete(self, word):
        return self._delete(self._root, word, 0)

    def _find_node(self, prefix):
        node = self._root
        for ch in prefix:
            if ch not in node.children: return None
            node = node.children[ch]
        return node
// Trie using HashMap for children (Rust)
use std::collections::HashMap;

struct TrieNode {
    children: HashMap<char, TrieNode>,
    is_end: bool,
}

struct Trie { root: TrieNode }

impl Trie {
    fn new() -> Self {
        Trie { root: TrieNode {
            children: HashMap::new(), is_end: false
        }}
    }

    fn insert(&mut self, word: &str) {
        let mut node = &mut self.root;
        for ch in word.chars() {
            node = node.children
                .entry(ch)
                .or_insert_with(|| TrieNode {
                    children: HashMap::new(),
                    is_end: false,
                });
        }
        node.is_end = true;
    }

    fn search(&self, word: &str) -> bool {
        match self.find_node(word) {
            Some(node) => node.is_end,
            None => false,
        }
    }

    fn starts_with(&self, prefix: &str) -> bool {
        self.find_node(prefix).is_some()
    }
}

Bonus: Binary Trie (Maximum XOR)

A binary Trie stores numbers bit by bit (MSB to LSB). For each query number, we greedily choose the opposite bit at each level to maximize XOR. This yields the maximum XOR pair in O(n * log(max_val)).

Key Insight

To maximize XOR, at each bit position choose the path that gives the opposite bit. If we want bit b to be 1 in the XOR result, we need the two numbers to differ at that position. Walking from MSB to LSB greedily ensures we maximize the highest-order bits first.

// Maximum XOR of two elements using binary Trie
static int maxXOR(int[] nums) {
    int max = Arrays.stream(nums).max().getAsInt();
    int bits = 32 - Integer.numberOfLeadingZeros(max);
    int[][] trie = new int[32 * nums.length][2];
    int size = 1;
    // Build binary trie
    for (int x : nums) {
        int node = 0;
        for (int b = bits - 1; b >= 0; b--) {
            int bit = (x >> b) & 1;
            if (trie[node][bit] == 0) trie[node][bit] = size++;
            node = trie[node][bit];
        }
    }
    // Query: for each number, try to pick opposite bits
    int result = 0;
    for (int x : nums) {
        int node = 0, cur = 0;
        for (int b = bits - 1; b >= 0; b--) {
            int bit = (x >> b) & 1;
            int want = 1 - bit; // opposite bit
            if (trie[node][want] != 0) {
                cur |= (1 << b);
                node = trie[node][want];
            } else node = trie[node][bit];
        }
        result = Math.max(result, cur);
    }
    return result;
}

Summary & Cheat Sheet

Segment Tree

Build O(n), Query/Update O(log n). Stores aggregates in binary tree. Supports range queries (sum/min/max) + point updates. Space: O(4n).

Fenwick Tree

Uses i & (-i) bit trick. Simpler than Segment Tree. Best for prefix sum + point update. Space: O(n). Not suitable for range min/max.

Trie

Tree where paths spell words. O(L) insert/search/prefix. Applications: autocomplete, spell check, IP routing, XOR queries.

When to Use What

BIT: prefix sums only. Segment Tree: range min/max, lazy propagation. Trie: string prefix operations. Binary Trie: XOR optimization.
