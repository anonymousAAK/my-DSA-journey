Greedy Algorithms

Make the locally optimal choice at each step to reach a globally optimal solution. Learn when greedy works, when it fails, and how to prove correctness.

Intermediate ~8 hours estimated 6 classic problems

🎯 Learning Objectives

  • Understand the greedy paradigm: greedy choice property + optimal substructure
  • Solve Activity Selection by sorting by finish time
  • Differentiate Fractional Knapsack (greedy) from 0/1 Knapsack (DP)
  • Recognize when greedy fails (e.g., non-canonical coin systems)
  • Apply the merge-and-sweep technique for interval problems

💡 Greedy Fundamentals

A greedy algorithm builds a solution by making the locally optimal choice at each step. Unlike DP, it never reconsiders past decisions.

When Does Greedy Work?

  • Greedy Choice Property: A global optimum can be reached by making locally optimal choices
  • Optimal Substructure: The optimal solution contains optimal solutions to subproblems

Greedy vs Dynamic Programming

AspectGreedyDP
ApproachOne choice at each step, never look backTry all subproblems, combine results
GuaranteeOnly if greedy choice property holdsAlways optimal (if recurrence is correct)
TimeUsually O(n log n)Usually O(n*W) or O(n^2)
ExampleFractional Knapsack0/1 Knapsack

Proving Correctness

Exchange Argument
The most common proof technique: assume an optimal solution that differs from the greedy solution, then show you can "exchange" a non-greedy choice for the greedy one without making the solution worse.

📅 Activity Selection

Given activities with start and end times, select the maximum number of non-overlapping activities. The key insight: always pick the activity that finishes earliest.

Why Earliest Finish?

Activities sorted by end time: |--A--| end=2 ← pick (finishes earliest) |----B----| end=4 |--------C--------| end=6 |-------D------| end=7 |--E--| end=9 ← pick (starts after A ends) |----F---| end=10 Selected: A, D or E, ... → maximizes count
GreedyAlgorithms.java
static int activitySelection(int[] start, int[] end) { int n = start.length; // Sort by end time Integer[] idx = new Integer[n]; for (int i = 0; i < n; i++) idx[i] = i; Arrays.sort(idx, (a, b) -> end[a] - end[b]); int count = 1; int lastEnd = end[idx[0]]; for (int i = 1; i < n; i++) { if (start[idx[i]] >= lastEnd) { count++; lastEnd = end[idx[i]]; } } return count; }
greedy.cpp
std::vector<std::pair<int,int>> activity_selection( std::vector<std::pair<int,int>> activities) { // Sort by finish time. std::sort(activities.begin(), activities.end(), [](const auto& a, const auto& b) { return a.second < b.second; }); std::vector<std::pair<int,int>> selected; int last_finish = -1; for (const auto& [start, finish] : activities) { if (start >= last_finish) { selected.emplace_back(start, finish); last_finish = finish; } } return selected; }
greedy.py
def activity_selection(activities: List[Tuple[int, int]]) -> List[Tuple[int, int]]: """Select max non-overlapping activities. Greedy: always pick the activity that finishes earliest.""" sorted_acts = sorted(activities, key=lambda x: x[1]) selected = [] last_finish = -1 for start, finish in sorted_acts: if start >= last_finish: selected.append((start, finish)) last_finish = finish return selected
greedy.rs
fn activity_selection(activities: &[(i32, i32)]) -> Vec<usize> { if activities.is_empty() { return Vec::new(); } // Create indices sorted by finish time let mut indices: Vec<usize> = (0..activities.len()).collect(); indices.sort_unstable_by_key(|&i| activities[i].1); let mut selected = vec![indices[0]]; let mut last_finish = activities[indices[0]].1; for &i in &indices[1..] { if activities[i].0 >= last_finish { selected.push(i); last_finish = activities[i].1; } } selected }

🎒 Fractional Knapsack

Unlike 0/1 Knapsack, items can be broken into fractions. Sort by value-to-weight ratio and greedily take the best items first.

Why Greedy Works Here (but not for 0/1)

Key Difference
Since we can take fractions, we can always improve by replacing a lower-ratio portion with a higher-ratio one. In 0/1 Knapsack, we can't split items, so this exchange doesn't apply — DP is required.
GreedyAlgorithms.java
static double fractionalKnapsack(int[] weights, int[] values, int W) { int n = weights.length; Integer[] idx = new Integer[n]; for (int i = 0; i < n; i++) idx[i] = i; // Sort by value/weight ratio (descending) Arrays.sort(idx, (a, b) -> Double.compare( (double)values[b]/weights[b], (double)values[a]/weights[a])); double totalValue = 0; int remaining = W; for (int i : idx) { if (remaining <= 0) break; int take = Math.min(weights[i], remaining); totalValue += take * (double)values[i] / weights[i]; remaining -= take; } return totalValue; }
greedy.cpp
double fractional_knapsack( std::vector<std::pair<double,double>> items, double capacity) { // items[i] = {weight, value} // Sort by value/weight ratio descending. std::sort(items.begin(), items.end(), [](const auto& a, const auto& b) { return (a.second / a.first) > (b.second / b.first); }); double total_value = 0.0; for (const auto& [w, v] : items) { if (capacity <= 0) break; double take = std::min(capacity, w); total_value += take * (v / w); capacity -= take; } return total_value; }
greedy.py
def fractional_knapsack( items: List[Tuple[float, float]], capacity: float ) -> float: """items = [(weight, value), ...]. Sort by value/weight ratio descending; take greedily.""" sorted_items = sorted(items, key=lambda x: x[1] / x[0], reverse=True) total_value = 0.0 remaining = capacity for weight, value in sorted_items: if remaining <= 0: break take = min(weight, remaining) total_value += take * (value / weight) remaining -= take return total_value
greedy.rs
fn fractional_knapsack(items: &[(f64, f64)], capacity: f64) -> f64 { let mut indices: Vec<usize> = (0..items.len()).collect(); indices.sort_unstable_by(|&a, &b| { let ratio_a = items[a].1 / items[a].0; let ratio_b = items[b].1 / items[b].0; ratio_b.partial_cmp(&ratio_a).unwrap() }); let mut remaining = capacity; let mut total_value = 0.0; for &i in &indices { let (weight, value) = items[i]; if remaining <= 0.0 { break; } if weight <= remaining { total_value += value; remaining -= weight; } else { total_value += value * (remaining / weight); remaining = 0.0; } } total_value }

💰 Minimum Coins (Greedy)

For canonical coin systems (like US denominations: 1, 5, 10, 25), always picking the largest denomination that fits gives the optimal result.

Greedy Fails for Non-Canonical Systems!
For coins {1, 3, 4} and target 6: Greedy picks 4+1+1 = 3 coins. But DP finds 3+3 = 2 coins. Always verify the greedy choice property before using greedy on coin problems. Use DP (Week 18: Coin Change) for arbitrary denominations.
GreedyAlgorithms.java
static Map<Integer, Integer> minCoinsGreedy(int[] coins, int target) { Arrays.sort(coins); Map<Integer, Integer> result = new LinkedHashMap<>(); for (int i = coins.length - 1; i >= 0 && target > 0; i--) { if (coins[i] <= target) { int count = target / coins[i]; result.put(coins[i], count); target -= coins[i] * count; } } return result; }
greedy.py
def minimum_coins_greedy(coins: List[int], amount: int) -> List[int]: """Greedy: always take the largest denomination that fits. Only optimal for canonical coin systems (e.g. US coins).""" sorted_coins = sorted(coins, reverse=True) result = [] remaining = amount for coin in sorted_coins: while remaining >= coin: result.append(coin) remaining -= coin if remaining != 0: return [] # not possible with greedy return result

🔗 Merge Intervals

Given a list of intervals, merge all overlapping ones. Sort by start time, then sweep through, extending the current interval when overlap is detected.

Walk-Through

Input: [1,3] [2,6] [8,10] [15,18] Sort by start (already sorted): [1,3] [2,6] [8,10] [15,18] Merge step: current = [1,3] [2,6]: 2 <= 3 → overlap! extend to [1,6] [8,10]: 8 > 6 → no overlap. push [1,6], current = [8,10] [15,18]: 15 > 10 → no overlap. push [8,10], current = [15,18] push [15,18] Result: [1,6] [8,10] [15,18]
GreedyAlgorithms.java
static int[][] mergeIntervals(int[][] intervals) { if (intervals.length <= 1) return intervals; Arrays.sort(intervals, (a, b) -> a[0] - b[0]); // sort by start List<int[]> merged = new ArrayList<>(); int[] current = intervals[0]; for (int i = 1; i < intervals.length; i++) { if (intervals[i][0] <= current[1]) { // overlapping current[1] = Math.max(current[1], intervals[i][1]); } else { merged.add(current); current = intervals[i]; } } merged.add(current); return merged.toArray(new int[0][]); }
greedy.cpp
std::vector<std::pair<int,int>> merge_intervals( std::vector<std::pair<int,int>> intervals) { if (intervals.empty()) return {}; std::sort(intervals.begin(), intervals.end()); std::vector<std::pair<int,int>> merged; merged.push_back(intervals[0]); for (std::size_t i = 1; i < intervals.size(); ++i) { auto& back = merged.back(); if (intervals[i].first <= back.second) { back.second = std::max(back.second, intervals[i].second); } else { merged.push_back(intervals[i]); } } return merged; }
greedy.py
def merge_intervals(intervals: List[List[int]]) -> List[List[int]]: if not intervals: return [] intervals.sort(key=lambda x: x[0]) merged = [intervals[0]] for start, end in intervals[1:]: if start <= merged[-1][1]: merged[-1][1] = max(merged[-1][1], end) else: merged.append([start, end]) return merged
greedy.rs
fn merge_intervals(intervals: &[(i32, i32)]) -> Vec<(i32, i32)> { if intervals.is_empty() { return Vec::new(); } let mut sorted: Vec<(i32, i32)> = intervals.to_vec(); sorted.sort_unstable_by_key(|&(start, _)| start); let mut merged: Vec<(i32, i32)> = vec![sorted[0]]; for &(start, end) in &sorted[1..] { let last = merged.last_mut().unwrap(); if start <= last.1 { last.1 = last.1.max(end); } else { merged.push((start, end)); } } merged }

🏢 Minimum Meeting Rooms

Given meeting intervals, find the minimum number of rooms needed so no two overlapping meetings share a room. Uses a min-heap of end times.

Algorithm

  • Sort meetings by start time
  • Maintain a min-heap tracking end times of currently active meetings
  • For each meeting: if it starts after the earliest-ending meeting, reuse that room (pop from heap)
  • Always push the new meeting's end time. The heap size = rooms needed
GreedyAlgorithms.java
static int minMeetingRooms(int[][] intervals) { if (intervals.length == 0) return 0; Arrays.sort(intervals, (a, b) -> a[0] - b[0]); PriorityQueue<Integer> ends = new PriorityQueue<>(); // min-heap ends.offer(intervals[0][1]); for (int i = 1; i < intervals.length; i++) { if (intervals[i][0] >= ends.peek()) ends.poll(); // room becomes free ends.offer(intervals[i][1]); } return ends.size(); }
greedy.cpp
int min_meeting_rooms(std::vector<std::pair<int,int>> meetings) { if (meetings.empty()) return 0; std::sort(meetings.begin(), meetings.end()); // Min-heap of end times std::priority_queue<int, std::vector<int>, std::greater<>> pq; pq.push(meetings[0].second); for (std::size_t i = 1; i < meetings.size(); ++i) { if (pq.top() <= meetings[i].first) pq.pop(); // reuse room pq.push(meetings[i].second); } return static_cast<int>(pq.size()); }
greedy.py
def min_meeting_rooms(intervals: List[List[int]]) -> int: if not intervals: return 0 intervals.sort(key=lambda x: x[0]) rooms = [] # min-heap of end times for start, end in intervals: if rooms and rooms[0] <= start: heapq.heappop(rooms) # free that room heapq.heappush(rooms, end) return len(rooms)
greedy.rs
fn min_meeting_rooms(meetings: &[(i32, i32)]) -> usize { if meetings.is_empty() { return 0; } let mut sorted: Vec<(i32, i32)> = meetings.to_vec(); sorted.sort_unstable_by_key(|&(start, _)| start); // BinaryHeap is max-heap; use Reverse for min-heap let mut heap: BinaryHeap<Reverse<i32>> = BinaryHeap::new(); for &(start, end) in &sorted { if let Some(&Reverse(earliest_end)) = heap.peek() { if start >= earliest_end { heap.pop(); } } heap.push(Reverse(end)); } heap.len() }

💼 Job Sequencing with Deadlines

Each job has a deadline and profit, and takes 1 unit of time. Schedule jobs to maximize total profit. Greedy: sort by profit descending, assign each job to the latest available slot before its deadline.

greedy.cpp
struct Job { char id; int deadline; int profit; }; std::pair<int, std::string> job_sequencing(std::vector<Job> jobs) { // Sort by profit descending. std::sort(jobs.begin(), jobs.end(), [](const Job& a, const Job& b) { return a.profit > b.profit; }); int max_deadline = 0; for (const auto& j : jobs) max_deadline = std::max(max_deadline, j.deadline); std::vector<char> slot(max_deadline + 1, '\0'); int total_profit = 0; for (const auto& j : jobs) { for (int t = j.deadline; t >= 1; --t) { if (slot[t] == '\0') { slot[t] = j.id; total_profit += j.profit; break; } } } std::string sequence; for (int t = 1; t <= max_deadline; ++t) if (slot[t] != '\0') sequence.push_back(slot[t]); return {total_profit, sequence}; }
greedy.py
def job_sequencing(jobs): """jobs = [(job_id, deadline, profit), ...]. Greedy: sort by profit desc, assign to latest free slot.""" sorted_jobs = sorted(jobs, key=lambda x: x[2], reverse=True) max_deadline = max(job[1] for job in jobs) if jobs else 0 slots = [None] * (max_deadline + 1) total_profit = 0 for job_id, deadline, profit in sorted_jobs: for t in range(deadline, 0, -1): if slots[t] is None: slots[t] = job_id total_profit += profit break return total_profit, [s for s in slots if s is not None]

When Greedy Fails
If you can't prove the greedy choice property, the problem likely needs DP. Classic traps: 0/1 Knapsack, non-canonical coin change, longest path in general graphs. When in doubt, try a counterexample with a small input.
