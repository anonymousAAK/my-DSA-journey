📍 Point and Line Representation

The foundation of computational geometry: a point is (x, y), a vector from A to B is (B.x - A.x, B.y - A.y), and a line segment is defined by two endpoints. Using floating point can introduce errors, so prefer integer arithmetic and cross products over slopes.

Point: P = (x, y) Vector: AB = B - A = (B.x - A.x, B.y - A.y) Line through P with direction d: P + t*d (parametric form) Line segment AB: points between A and B (t in [0, 1]) B (4,3) / / ← vector AB = (3, 2) / A (1,1) Key: avoid division and floating point — use cross products instead!

✖ Cross Product and Orientation Test

The 2D cross product of vectors u and v is: u × v = u.x * v.y - u.y * v.x. Its sign tells us the orientation of the turn from u to v.

Cross product of AB and AC: cross = (B.x - A.x) * (C.y - A.y) - (B.y - A.y) * (C.x - A.x) cross > 0: Counter-clockwise (left turn) cross < 0: Clockwise (right turn) cross = 0: Collinear C C / \ / CCW (cross > 0) \ CW (cross < 0) / \ A ──────── B A ──────── B
Geometry.java
static long cross(long[] O, long[] A, long[] B) { return (A[0] - O[0]) * (B[1] - O[1]) - (A[1] - O[1]) * (B[0] - O[0]); } // Returns: +1 = CCW, -1 = CW, 0 = collinear static int orientation(long[] A, long[] B, long[] C) { long v = cross(A, B, C); if (v > 0) return 1; // counter-clockwise if (v < 0) return -1; // clockwise return 0; // collinear } // Check if point P is on segment AB static boolean onSegment(long[] A, long[] B, long[] P) { return cross(A, B, P) == 0 && Math.min(A[0], B[0]) <= P[0] && P[0] <= Math.max(A[0], B[0]) && Math.min(A[1], B[1]) <= P[1] && P[1] <= Math.max(A[1], B[1]); }

struct Point { long long x, y; }; long long cross(Point O, Point A, Point B) { return (A.x - O.x) * (B.y - O.y) - (A.y - O.y) * (B.x - O.x); } int orientation(Point A, Point B, Point C) { long long v = cross(A, B, C); if (v > 0) return 1; // CCW if (v < 0) return -1; // CW return 0; // collinear } bool onSegment(Point A, Point B, Point P) { return cross(A, B, P) == 0 && min(A.x, B.x) <= P.x && P.x <= max(A.x, B.x) && min(A.y, B.y) <= P.y && P.y <= max(A.y, B.y); }

⭐ Convex Hull

The convex hull of a set of points is the smallest convex polygon enclosing all points. Think of stretching a rubber band around pins — it snaps to the outermost points.

Points: Convex Hull: * C C / \ / \ / * D / \ * \ / \ A *E \ A D \ / \ / \ *F / \ / \ / \ / * B B Hull vertices: A, B, D, C (counter-clockwise) Interior points E, F are excluded.

🔄 Graham Scan

Sort points by polar angle from the lowest point, then process them maintaining a stack of hull vertices. If a right turn is detected, pop the last point. Time: O(n log n).

Geometry.java
static List<long[]> grahamScan(long[][] points) { int n = points.length; // Find lowest point (then leftmost) int pivot = 0; for (int i = 1; i < n; i++) if (points[i][1] < points[pivot][1] || (points[i][1] == points[pivot][1] && points[i][0] < points[pivot][0])) pivot = i; long[] tmp = points[0]; points[0] = points[pivot]; points[pivot] = tmp; long[] O = points[0]; // Sort by polar angle relative to O Arrays.sort(points, 1, n, (a, b) -> { long v = cross(O, a, b); if (v != 0) return v > 0 ? -1 : 1; long da = (a[0]-O[0])*(a[0]-O[0]) + (a[1]-O[1])*(a[1]-O[1]); long db = (b[0]-O[0])*(b[0]-O[0]) + (b[1]-O[1])*(b[1]-O[1]); return Long.compare(da, db); }); Deque<long[]> stack = new ArrayDeque<>(); stack.push(points[0]); for (int i = 1; i < n; i++) { while (stack.size() > 1) { long[] top = stack.pop(); if (cross(stack.peek(), top, points[i]) > 0) { stack.push(top); break; } } stack.push(points[i]); } return new ArrayList<>(stack); }

📈 Andrew's Monotone Chain

An alternative convex hull algorithm that builds the upper and lower hulls separately. Sort points by x-coordinate, sweep left-to-right for the lower hull and right-to-left for the upper hull. Often preferred for its simplicity.

Geometry.java
static List<long[]> monotoneChain(long[][] points) { int n = points.length; Arrays.sort(points, (a, b) -> a[0] != b[0] ? Long.compare(a[0], b[0]) : Long.compare(a[1], b[1])); long[][] hull = new long[2 * n][]; int k = 0; // Lower hull for (int i = 0; i < n; i++) { while (k >= 2 && cross(hull[k-2], hull[k-1], points[i]) <= 0) k--; hull[k++] = points[i]; } // Upper hull int lower = k + 1; for (int i = n - 2; i >= 0; i--) { while (k >= lower && cross(hull[k-2], hull[k-1], points[i]) <= 0) k--; hull[k++] = points[i]; } return Arrays.asList(Arrays.copyOf(hull, k - 1)); }
geometry.cpp
vector<Point> monotoneChain(vector<Point> pts) { int n = pts.size(); sort(pts.begin(), pts.end(), [](Point a, Point b) { return a.x != b.x ? a.x < b.x : a.y < b.y; }); vector<Point> hull(2 * n); int k = 0; // Lower hull for (int i = 0; i < n; ++i) { while (k >= 2 && cross(hull[k-2], hull[k-1], pts[i]) <= 0) --k; hull[k++] = pts[i]; } // Upper hull int lower = k + 1; for (int i = n - 2; i >= 0; --i) { while (k >= lower && cross(hull[k-2], hull[k-1], pts[i]) <= 0) --k; hull[k++] = pts[i]; } hull.resize(k - 1); return hull; }
geometry.py
def monotone_chain(points): points = sorted(points) lower = [] for p in points: while len(lower) >= 2 and cross(lower[-2], lower[-1], p) <= 0: lower.pop() lower.append(p) upper = [] for p in reversed(points): while len(upper) >= 2 and cross(upper[-2], upper[-1], p) <= 0: upper.pop() upper.append(p) return lower[:-1] + upper[:-1]
geometry.rs
fn monotone_chain(pts: &mut Vec<Point>) -> Vec<Point> { pts.sort_by(|a, b| a.x.cmp(&b.x).then(a.y.cmp(&b.y))); let n = pts.len(); let mut hull = Vec::with_capacity(2 * n); // Lower hull for i in 0..n { while hull.len() >= 2 && cross(hull[hull.len()-2], hull[hull.len()-1], pts[i]) <= 0 { hull.pop(); } hull.push(pts[i]); } // Upper hull let lower = hull.len() + 1; for i in (0..n-1).rev() { while hull.len() >= lower && cross(hull[hull.len()-2], hull[hull.len()-1], pts[i]) <= 0 { hull.pop(); } hull.push(pts[i]); } hull.pop(); hull }

✚ Line Segment Intersection

Two segments AB and CD intersect if and only if the orientations of their endpoints satisfy specific conditions. This uses the cross product orientation test — no division or floating point needed.

Segments AB and CD intersect if: 1. A and B are on opposite sides of line CD (orientation(C,D,A) != orientation(C,D,B)) AND 2. C and D are on opposite sides of line AB (orientation(A,B,C) != orientation(A,B,D)) Special case: collinear points require overlap check. A C ─────── D \ / \ / ← Intersection point \ / \ / \ / B
Geometry.java
static boolean segmentsIntersect(long[] A, long[] B, long[] C, long[] D) { int d1 = sign(cross(C, D, A)); int d2 = sign(cross(C, D, B)); int d3 = sign(cross(A, B, C)); int d4 = sign(cross(A, B, D)); if (((d1 > 0 && d2 < 0) || (d1 < 0 && d2 > 0)) && ((d3 > 0 && d4 < 0) || (d3 < 0 && d4 > 0))) return true; // Check collinear cases if (d1 == 0 && onSegment(C, D, A)) return true; if (d2 == 0 && onSegment(C, D, B)) return true; if (d3 == 0 && onSegment(A, B, C)) return true; if (d4 == 0 && onSegment(A, B, D)) return true; return false; } static int sign(long x) { return x > 0 ? 1 : (x < 0 ? -1 : 0); }

📏 Closest Pair of Points

Find the two points with minimum Euclidean distance. The naive O(n²) approach checks all pairs. The divide and conquer approach achieves O(n log n) by splitting points, solving recursively, and cleverly checking the strip near the dividing line.

Divide and Conquer strategy: 1. Sort points by x-coordinate 2. Split into left and right halves at midpoint M 3. Recursively find closest pair in each half 4. d = min(d_left, d_right) 5. Check points within strip of width 2d around M 6. For each point in strip, check at most 7 neighbors (sorted by y) Left Half | Right Half | * * | * * | * * * * | * * | * | <- d -><- d -> Strip region Key insight: at most O(n) comparisons needed in the strip!
Geometry.java
static double closestPair(long[][] pts) { Arrays.sort(pts, (a, b) -> Long.compare(a[0], b[0])); return solve(pts, 0, pts.length - 1); } static double solve(long[][] pts, int lo, int hi) { if (hi - lo < 3) { double best = Double.MAX_VALUE; for (int i = lo; i <= hi; i++) for (int j = i + 1; j <= hi; j++) best = Math.min(best, dist(pts[i], pts[j])); return best; } int mid = (lo + hi) / 2; long midX = pts[mid][0]; double d = Math.min(solve(pts, lo, mid), solve(pts, mid + 1, hi)); // Collect strip points List<long[]> strip = new ArrayList<>(); for (int i = lo; i <= hi; i++) if (Math.abs(pts[i][0] - midX) < d) strip.add(pts[i]); strip.sort((a, b) -> Long.compare(a[1], b[1])); for (int i = 0; i < strip.size(); i++) for (int j = i + 1; j < strip.size() && (strip.get(j)[1] - strip.get(i)[1]) < d; j++) d = Math.min(d, dist(strip.get(i), strip.get(j))); return d; } static double dist(long[] a, long[] b) { double dx = a[0] - b[0], dy = a[1] - b[1]; return Math.sqrt(dx * dx + dy * dy); }

📐 Shoelace Formula (Polygon Area)

The Shoelace formula computes the area of a simple polygon given its vertices in order: Area = |∑(x_i * y_{i+1} - x_{i+1} * y_i)| / 2. The sign tells orientation: positive for counter-clockwise, negative for clockwise.

Polygon vertices (CCW): (1,0), (4,0), (4,3), (2,5), (0,3) Area = | (1*0 - 4*0) + (4*3 - 4*0) + (4*5 - 2*3) + (2*3 - 0*5) + (0*0 - 1*3) | / 2 = | 0 + 12 + 14 + 6 + (-3) | / 2 = | 29 | / 2 = 14.5 square units (2,5) / \ / \ (0,3) (4,3) | | | | (1,0)---(4,0)
Geometry.java
// Returns 2 * signed area (positive if CCW) static long signedArea2(long[][] polygon) { long area = 0; int n = polygon.length; for (int i = 0; i < n; i++) { int j = (i + 1) % n; area += polygon[i][0] * polygon[j][1]; area -= polygon[j][0] * polygon[i][1]; } return area; } static double polygonArea(long[][] polygon) { return Math.abs(signedArea2(polygon)) / 2.0; } // Bonus: check if point P is inside convex polygon static boolean isInsideConvex(long[][] hull, long[] P) { int n = hull.length; for (int i = 0; i < n; i++) { int j = (i + 1) % n; if (cross(hull[i], hull[j], P) < 0) return false; } return true; }

Pro Tip
Floating-point precision is the biggest pitfall in computational geometry. Use integer arithmetic wherever possible (cross products, signed areas). When you must use floating point, add an epsilon tolerance (e.g., 1e-9) to comparisons, and prefer squared distances over actual distances to avoid square roots.
