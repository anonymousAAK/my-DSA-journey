➕ Array Sum

The simplest array problem: iterate through all elements and accumulate a running sum. This O(n) single-pass technique is the foundation for many array algorithms.

ReturnArraySum.java
// Iterative approach -- O(n) time, O(1) space public static int sum(int[] arr) { int sum = 0; for (int val : arr) { sum += val; } return sum; } // Recursive approach -- O(n) time, O(n) stack space public static int sumRecursive(int[] arr, int index) { if (index == arr.length) return 0; return arr[index] + sumRecursive(arr, index + 1); }
arrays.cpp
// Using STL accumulate -- O(n) time, O(1) space #include <numeric> int arraySum(const vector<int>& arr) { return accumulate(arr.begin(), arr.end(), 0); }
arrays.py
# Python built-in -- O(n) time, O(1) space total = sum(arr) # Manual loop def array_sum(arr): total = 0 for val in arr: total += val return total
arrays.rs
// Rust iterator sum -- O(n) time, O(1) space let total: i64 = arr.iter().map(|&x| x as i64).sum();

🔍 Linear Search

Sequentially check each element until a match is found or the array ends. This is the simplest search algorithm -- no prerequisites on array order.

CaseTime
Best (target is first)O(1)
AverageO(n/2) = O(n)
Worst (not found)O(n)
LinearSearch.java
// Return index of first occurrence, or -1 if not found public static int linearSearch(int[] arr, int target) { for (int i = 0; i < arr.length; i++) { if (arr[i] == target) return i; } return -1; } // Count all occurrences public static int countOccurrences(int[] arr, int target) { int count = 0; for (int val : arr) { if (val == target) count++; } return count; } // Find minimum and maximum in one pass -- O(n) public static int[] findMinMax(int[] arr) { int min = arr[0], max = arr[0]; for (int val : arr) { if (val < min) min = val; if (val > max) max = val; } return new int[]{min, max}; }
arrays.cpp
// Linear search -- O(n) time, O(1) space int linearSearch(const vector<int>& arr, int target) { for (int i = 0; i < (int)arr.size(); ++i) { if (arr[i] == target) return i; } return -1; } // Using STL find auto it = find(arr.begin(), arr.end(), target); int idx = (it != arr.end()) ? distance(arr.begin(), it) : -1;
arrays.py
def linear_search(arr, target): for i, val in enumerate(arr): if val == target: return i return -1 # Python built-in (raises ValueError if not found) idx = arr.index(target)
arrays.rs
// Using iterator position -- O(n) time fn linear_search(arr: &[i32], target: i32) -> Option<usize> { arr.iter().position(|&x| x == target) }
Linear vs Binary Search

Linear search works on any array (sorted or unsorted). Binary search requires a sorted array but runs in O(log n). For small arrays (n < 30), linear search can be faster due to cache effects.

🔄 Array Reverse & Rotate

Reverse: Two-Pointer Technique

Swap elements at the left and right pointers, then move inward. Time: O(n), Space: O(1).

Left Rotate by K: Reversal Algorithm

Input: [1, 2, 3, 4, 5], k=2 Step 1: Reverse first k elements: [2, 1, 3, 4, 5] Step 2: Reverse remaining n-k: [2, 1, 5, 4, 3] Step 3: Reverse entire array: [3, 4, 5, 1, 2] Time: O(n), Space: O(1) -- much better than naive O(n*k) shifting!
ArrayReverseAndRotate.java
// Reverse a subarray from index l to r (inclusive) static void reverse(int[] arr, int l, int r) { while (l < r) { int temp = arr[l]; arr[l] = arr[r]; arr[r] = temp; l++; r--; } } // Left rotate by k positions using reversal algorithm // Time: O(n), Space: O(1) static void leftRotate(int[] arr, int k) { int n = arr.length; k = k % n; // handle k >= n if (k == 0) return; reverse(arr, 0, k - 1); reverse(arr, k, n - 1); reverse(arr, 0, n - 1); } // Right rotate = left rotate by (n - k) static void rightRotate(int[] arr, int k) { leftRotate(arr, arr.length - (k % arr.length)); }
arrays.cpp
void reverseArray(vector<int>& arr, int left, int right) { while (left < right) { swap(arr[left++], arr[right--]); } } // Rotate right by k positions -- O(n) time, O(1) space void rotateRight(vector<int>& arr, int k) { int n = arr.size(); if (n == 0) return; k = ((k % n) + n) % n; reverseArray(arr, 0, n - 1); reverseArray(arr, 0, k - 1); reverseArray(arr, k, n - 1); }
arrays.py
def reverse_array(arr): """Reverse in-place using two pointers. Time: O(n), Space: O(1)""" left, right = 0, len(arr) - 1 while left < right: arr[left], arr[right] = arr[right], arr[left] left += 1 right -= 1 return arr def rotate_array(arr, k): """Rotate right by k using reversal algorithm. Time: O(n), Space: O(1)""" n = len(arr) k %= n def _reverse(lo, hi): while lo < hi: arr[lo], arr[hi] = arr[hi], arr[lo] lo += 1; hi -= 1 _reverse(0, n - 1) _reverse(0, k - 1) _reverse(k, n - 1) return arr
arrays.rs
/// Rotate right by k using reversal algorithm. Time: O(n), Space: O(1) fn rotate_array(arr: &mut [i32], k: usize) { let n = arr.len(); if n == 0 { return; } let k = k % n; if k == 0 { return; } arr.reverse(); arr[..k].reverse(); arr[k..].reverse(); }

📈 Prefix Sum & Range Queries

A prefix sum array P where P[i] = arr[0] + arr[1] + ... + arr[i]. After O(n) preprocessing, any range sum query can be answered in O(1):

sum(arr[l..r]) = P[r] - P[l-1]

Array: [3, -1, 2, 4, -3, 7] Prefix: [3, 2, 4, 8, 5, 12] Query: sum(arr[1..4]) = P[4] - P[0] = 5 - 3 = 2 Verify: -1 + 2 + 4 + (-3) = 2 ✔
PrefixSumAndKadane.java
// Build prefix sum array -- O(n) static int[] buildPrefix(int[] arr) { int n = arr.length; int[] prefix = new int[n]; prefix[0] = arr[0]; for (int i = 1; i < n; i++) { prefix[i] = prefix[i - 1] + arr[i]; } return prefix; } // Range sum query [l, r] using prefix -- O(1) static int rangeSum(int[] prefix, int l, int r) { if (l == 0) return prefix[r]; return prefix[r] - prefix[l - 1]; }
arrays.cpp
// Build prefix sum -- O(n) time, O(n) space // prefix[i] = arr[0] + ... + arr[i-1], prefix[0] = 0 vector<long long> buildPrefixSum(const vector<int>& arr) { int n = arr.size(); vector<long long> prefix(n + 1, 0); for (int i = 0; i < n; ++i) { prefix[i + 1] = prefix[i] + arr[i]; } return prefix; } // Range sum [l, r] inclusive -- O(1) long long rangeSum(const vector<long long>& prefix, int l, int r) { return prefix[r + 1] - prefix[l]; }
arrays.py
def build_prefix_sum(arr): """Build prefix-sum array. Time: O(n), Space: O(n)""" prefix = [0] * (len(arr) + 1) for i in range(len(arr)): prefix[i + 1] = prefix[i] + arr[i] return prefix def range_sum(prefix, left, right): """Return sum(arr[left..right]) in O(1).""" return prefix[right + 1] - prefix[left]
arrays.rs
fn build_prefix_sum(arr: &[i32]) -> Vec<i64> { let mut prefix = Vec::with_capacity(arr.len() + 1); prefix.push(0i64); for &val in arr { let last = *prefix.last().unwrap(); prefix.push(last + val as i64); } prefix } fn range_sum_query(prefix: &[i64], left: usize, right: usize) -> i64 { prefix[right + 1] - prefix[left] }

⚡ Kadane's Algorithm (Maximum Subarray Sum)

Find the contiguous subarray with the maximum sum. This is a classic Dynamic Programming problem solved in linear time.

Algorithm

At each position, decide: extend the current subarray, or start fresh?

currentSum = max(arr[i], currentSum + arr[i])

Array: [-2, 1, -3, 4, -1, 2, 1, -5, 4] The subarray [4, -1, 2, 1] has the maximum sum = 6
PrefixSumAndKadane.java
// Kadane's Algorithm -- O(n) time, O(1) space static int maxSubarraySum(int[] arr) { int maxSum = arr[0]; int currentSum = arr[0]; for (int i = 1; i < arr.length; i++) { currentSum = Math.max(arr[i], currentSum + arr[i]); maxSum = Math.max(maxSum, currentSum); } return maxSum; } // Extended: also return start and end indices static int[] maxSubarrayWithIndices(int[] arr) { int maxSum = arr[0], currentSum = arr[0]; int start = 0, end = 0, tempStart = 0; for (int i = 1; i < arr.length; i++) { if (arr[i] > currentSum + arr[i]) { currentSum = arr[i]; tempStart = i; } else { currentSum += arr[i]; } if (currentSum > maxSum) { maxSum = currentSum; start = tempStart; end = i; } } return new int[]{maxSum, start, end}; }
arrays.cpp
// Kadane's -- returns {maxSum, startIndex, endIndex} // Time: O(n) Space: O(1) tuple<long long, int, int> kadane(const vector<int>& arr) { long long maxSum = arr[0], currentSum = arr[0]; int start = 0, end = 0, tempStart = 0; for (int i = 1; i < (int)arr.size(); ++i) { if (currentSum + arr[i] < arr[i]) { currentSum = arr[i]; tempStart = i; } else { currentSum += arr[i]; } if (currentSum > maxSum) { maxSum = currentSum; start = tempStart; end = i; } } return {maxSum, start, end}; }
arrays.py
def kadane(arr): """Return (max_sum, start_index, end_index). Time: O(n), Space: O(1)""" max_sum = current_sum = arr[0] start = end = temp_start = 0 for i in range(1, len(arr)): if current_sum + arr[i] < arr[i]: current_sum = arr[i] temp_start = i else: current_sum += arr[i] if current_sum > max_sum: max_sum = current_sum start = temp_start end = i return (max_sum, start, end)
arrays.rs
/// Kadane's algorithm. Returns (max_sum, start, end). /// Time: O(n), Space: O(1) fn kadanes_algorithm(arr: &[i32]) -> (i64, usize, usize) { let mut max_sum = arr[0] as i64; let mut current_sum = arr[0] as i64; let (mut start, mut end, mut temp_start) = (0, 0, 0); for i in 1..arr.len() { if current_sum + (arr[i] as i64) < arr[i] as i64 { current_sum = arr[i] as i64; temp_start = i; } else { current_sum += arr[i] as i64; } if current_sum > max_sum { max_sum = current_sum; start = temp_start; end = i; } } (max_sum, start, end) }

🇳🇱 Dutch National Flag (3-Way Partition)

Sort an array containing only 0s, 1s, and 2s in-place in a single pass.

Dijkstra's Three-Pointer Algorithm

Three pointers: low, mid, high arr[0..low-1] = all 0s (sorted) arr[low..mid-1] = all 1s (sorted) arr[mid..high] = unsorted (being processed) arr[high+1..n-1] = all 2s (sorted) Rules: arr[mid] == 0 → swap(low, mid), low++, mid++ arr[mid] == 1 → mid++ arr[mid] == 2 → swap(mid, high), high-- (don't advance mid!)
DutchNationalFlagAndMissing.java
// Dutch National Flag -- O(n) time, O(1) space static void dutchFlag(int[] arr) { int low = 0, mid = 0, high = arr.length - 1; while (mid <= high) { if (arr[mid] == 0) { int t = arr[low]; arr[low] = arr[mid]; arr[mid] = t; low++; mid++; } else if (arr[mid] == 1) { mid++; } else { // arr[mid] == 2 int t = arr[mid]; arr[mid] = arr[high]; arr[high] = t; high--; } } } // [2,0,2,1,1,0] --> [0,0,1,1,2,2]
arrays.cpp
// Dutch National Flag -- O(n) time, O(1) space void dutchNationalFlag(vector<int>& arr) { int low = 0, mid = 0, high = (int)arr.size() - 1; while (mid <= high) { if (arr[mid] == 0) swap(arr[low++], arr[mid++]); else if (arr[mid] == 1) ++mid; else swap(arr[mid], arr[high--]); } }
arrays.py
def dutch_national_flag(arr): """Sort 0s, 1s, 2s in one pass. Time: O(n), Space: O(1)""" low = mid = 0 high = len(arr) - 1 while mid <= high: if arr[mid] == 0: arr[low], arr[mid] = arr[mid], arr[low] low += 1; mid += 1 elif arr[mid] == 1: mid += 1 else: arr[mid], arr[high] = arr[high], arr[mid] high -= 1 return arr
arrays.rs
/// Dutch National Flag -- O(n) time, O(1) space fn dutch_national_flag(arr: &mut [i32]) { if arr.is_empty() { return; } let (mut low, mut mid) = (0usize, 0usize); let mut high = arr.len() - 1; while mid <= high { match arr[mid] { 0 => { arr.swap(low, mid); low += 1; mid += 1; } 1 => { mid += 1; } 2 => { arr.swap(mid, high); if high == 0 { break; } high -= 1; } _ => panic!("Array must contain only 0, 1, or 2"), } } }

🔎 Missing Number & Find Duplicate

Missing Number

Given array [0..n] with one number missing. Two approaches:

  • Sum formula: expected = n*(n+1)/2, subtract actual sum. O(n) time, O(1) space.
  • XOR: XOR of [0..n] XOR all elements. All pairs cancel; only missing remains.

Find Duplicate: Floyd's Cycle Detection

Given n+1 integers where each is in [1..n], one number repeats. Treat the array as a linked list where arr[i] is the "next" pointer. The duplicate creates a cycle; find the cycle entry.

DutchNationalFlagAndMissing.java
// Missing number using XOR -- O(n) time, O(1) space static int missingNumberXOR(int[] arr) { int n = arr.length; int xor = 0; for (int i = 0; i <= n; i++) xor ^= i; for (int x : arr) xor ^= x; return xor; // only missing number remains } // Find duplicate -- Floyd's Cycle Detection // O(n) time, O(1) space static int findDuplicate(int[] arr) { // Phase 1: find intersection point inside cycle int slow = arr[0], fast = arr[0]; do { slow = arr[slow]; fast = arr[arr[fast]]; } while (slow != fast); // Phase 2: find cycle entrance (= duplicate) slow = arr[0]; while (slow != fast) { slow = arr[slow]; fast = arr[fast]; } return slow; }
arrays.cpp
// Missing number using XOR -- O(n) time, O(1) space int missingNumberXOR(const vector<int>& nums) { int n = nums.size(); int xorAll = 0; for (int i = 0; i <= n; ++i) xorAll ^= i; for (int x : nums) xorAll ^= x; return xorAll; } // Floyd's cycle detection -- O(n) time, O(1) space int findDuplicate(const vector<int>& nums) { int slow = nums[0], fast = nums[0]; do { slow = nums[slow]; fast = nums[nums[fast]]; } while (slow != fast); slow = nums[0]; while (slow != fast) { slow = nums[slow]; fast = nums[fast]; } return slow; }
arrays.py
def missing_number_xor(nums): """Find missing number using XOR. Time: O(n), Space: O(1)""" xor = len(nums) for i, v in enumerate(nums): xor ^= i ^ v return xor def find_duplicate(nums): """Floyd's tortoise and hare. Time: O(n), Space: O(1)""" slow = fast = nums[0] while True: slow = nums[slow] fast = nums[nums[fast]] if slow == fast: break slow = nums[0] while slow != fast: slow = nums[slow] fast = nums[fast] return slow
arrays.rs
fn missing_number_xor(nums: &[i32]) -> i32 { let n = nums.len() as i32; let mut xor_all = 0i32; for i in 0..=n { xor_all ^= i; } for &num in nums { xor_all ^= num; } xor_all } fn find_duplicate(nums: &[i32]) -> i32 { let mut slow = nums[0] as usize; let mut fast = nums[0] as usize; loop { slow = nums[slow] as usize; fast = nums[nums[fast] as usize] as usize; if slow == fast { break; } } let mut slow = nums[0] as usize; while slow != fast { slow = nums[slow] as usize; fast = nums[fast] as usize; } slow as i32 }

📚 Summary

  • Two-pointer technique (reverse, rotate) is fundamental for O(1)-space array manipulation.
  • Prefix sum is a powerful preprocessing technique for range queries.
  • Kadane's algorithm elegantly solves maximum subarray in O(n) using a simple DP recurrence.
  • Dutch National Flag (3-way partition) is the key building block for quicksort with many duplicates.
  • XOR tricks and Floyd's cycle detection solve missing/duplicate problems in O(1) extra space.
  • Nearly every algorithm this week runs in O(n) time and O(1) space -- that is the sweet spot for array problems.
