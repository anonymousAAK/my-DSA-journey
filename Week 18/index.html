💡 DP Fundamentals

Dynamic Programming solves problems by breaking them into overlapping subproblems, solving each once and storing the result.

When to Use DP

  • Optimal Substructure: The optimal solution contains optimal solutions to subproblems
  • Overlapping Subproblems: The same subproblems are solved multiple times

Two Approaches

ApproachDirectionImplementationPros
Top-Down (Memoization)Recursion + cacheNatural recursive thinkingOnly solves needed subproblems
Bottom-Up (Tabulation)Iterative, small → largeFill a table from base casesUsually faster, no stack overhead

🚶 Climbing Stairs

Reach stair n taking 1 or 2 steps at a time. How many distinct ways? This is Fibonacci in disguise: dp[i] = dp[i-1] + dp[i-2].

DynamicProgramming.java
static int climbStairs(int n) { if (n <= 2) return n; int prev2 = 1, prev1 = 2; for (int i = 3; i <= n; i++) { int curr = prev1 + prev2; prev2 = prev1; prev1 = curr; } return prev1; }
dynamic_programming.cpp
int climbing_stairs(int n) { if (n <= 2) return n; int a = 1, b = 2; for (int i = 3; i <= n; ++i) { int c = a + b; a = b; b = c; } return b; }
dynamic_programming.py
def climbing_stairs(n: int) -> int: if n <= 1: return 1 prev2, prev1 = 1, 1 for _ in range(2, n + 1): prev2, prev1 = prev1, prev2 + prev1 return prev1
dynamic_programming.rs
fn climbing_stairs(n: u32) -> u64 { if n <= 1 { return 1; } let mut prev2 = 1u64; let mut prev1 = 1u64; for _ in 2..=n { let current = prev1 + prev2; prev2 = prev1; prev1 = current; } prev1 }

🎒 0/1 Knapsack

Given n items with weights and values, and a knapsack of capacity W, maximize total value without exceeding capacity. Each item is either taken or not (0/1).

Recurrence

dp[i][w] = max( dp[i-1][w], // skip item i dp[i-1][w - weights[i]] + values[i] // take item i (if it fits) )
DynamicProgramming.java
static int knapsack(int[] weights, int[] values, int W) { int n = weights.length; int[][] dp = new int[n + 1][W + 1]; for (int i = 1; i <= n; i++) { for (int w = 0; w <= W; w++) { dp[i][w] = dp[i-1][w]; // don't take if (weights[i-1] <= w) dp[i][w] = Math.max(dp[i][w], dp[i-1][w - weights[i-1]] + values[i-1]); } } return dp[n][W]; }
dynamic_programming.cpp
int knapsack_2d(const std::vector<int>& weights, const std::vector<int>& values, int W) { int n = static_cast<int>(weights.size()); std::vector<std::vector<int>> dp(n + 1, std::vector<int>(W + 1, 0)); for (int i = 1; i <= n; ++i) for (int w = 0; w <= W; ++w) { dp[i][w] = dp[i - 1][w]; if (weights[i - 1] <= w) dp[i][w] = std::max(dp[i][w], dp[i - 1][w - weights[i - 1]] + values[i - 1]); } return dp[n][W]; }
dynamic_programming.py
def knapsack_01_2d(weights, values, capacity): n = len(weights) dp = [[0] * (capacity + 1) for _ in range(n + 1)] for i in range(1, n + 1): for w in range(capacity + 1): dp[i][w] = dp[i - 1][w] if weights[i - 1] <= w: dp[i][w] = max(dp[i][w], dp[i - 1][w - weights[i - 1]] + values[i - 1]) return dp[n][capacity]
dynamic_programming.rs
fn knapsack_01(weights: &[usize], values: &[i64], capacity: usize) -> i64 { let n = weights.len(); let mut dp = vec![vec![0i64; capacity + 1]; n + 1]; for i in 1..=n { for w in 0..=capacity { dp[i][w] = dp[i - 1][w]; if weights[i - 1] <= w { let take = dp[i - 1][w - weights[i - 1]] + values[i - 1]; dp[i][w] = dp[i][w].max(take); } } } dp[n][capacity] }

⚡ Space-Optimized Knapsack

Since each row only depends on the previous row, we can use a single 1D array. Key: iterate w in reverse to avoid using an item twice.

Why Reverse?
If we iterate w from left to right, dp[w - weights[i]] might already be updated for the current item, effectively allowing an item to be used multiple times (unbounded knapsack). Reversing prevents this.
DynamicProgramming.java
static int knapsackOptimized(int[] weights, int[] values, int W) { int[] dp = new int[W + 1]; for (int i = 0; i < weights.length; i++) for (int w = W; w >= weights[i]; w--) // REVERSE! dp[w] = Math.max(dp[w], dp[w - weights[i]] + values[i]); return dp[W]; }
dynamic_programming.py
def knapsack_01_optimized(weights, values, capacity): dp = [0] * (capacity + 1) for i in range(len(weights)): for w in range(capacity, weights[i] - 1, -1): dp[w] = max(dp[w], dp[w - weights[i]] + values[i]) return dp[capacity]

📝 Longest Common Subsequence

Find the length of the longest subsequence present in both strings. A subsequence preserves relative order but need not be contiguous.

Recurrence

if s1[i-1] == s2[j-1]: dp[i][j] = dp[i-1][j-1] + 1 // characters match else: dp[i][j] = max(dp[i-1][j], dp[i][j-1]) // skip one or the other
DynamicProgramming.java
static int lcs(String s1, String s2) { int m = s1.length(), n = s2.length(); int[][] dp = new int[m + 1][n + 1]; for (int i = 1; i <= m; i++) for (int j = 1; j <= n; j++) if (s1.charAt(i-1) == s2.charAt(j-1)) dp[i][j] = dp[i-1][j-1] + 1; else dp[i][j] = Math.max(dp[i-1][j], dp[i][j-1]); return dp[m][n]; }
dynamic_programming.cpp
std::pair<int, std::string> lcs(const std::string& a, const std::string& b) { int n = a.size(), m = b.size(); std::vector<std::vector<int>> dp(n + 1, std::vector<int>(m + 1, 0)); for (int i = 1; i <= n; ++i) for (int j = 1; j <= m; ++j) if (a[i-1] == b[j-1]) dp[i][j] = dp[i-1][j-1] + 1; else dp[i][j] = std::max(dp[i-1][j], dp[i][j-1]); // Reconstruct std::string result; int i = n, j = m; while (i > 0 && j > 0) { if (a[i-1] == b[j-1]) { result.push_back(a[i-1]); --i; --j; } else if (dp[i-1][j] > dp[i][j-1]) --i; else --j; } std::reverse(result.begin(), result.end()); return {dp[n][m], result}; }
dynamic_programming.py
def lcs(text1: str, text2: str) -> Tuple[int, str]: m, n = len(text1), len(text2) dp = [[0] * (n + 1) for _ in range(m + 1)] for i in range(1, m + 1): for j in range(1, n + 1): if text1[i-1] == text2[j-1]: dp[i][j] = dp[i-1][j-1] + 1 else: dp[i][j] = max(dp[i-1][j], dp[i][j-1]) return dp[m][n]
dynamic_programming.rs
fn longest_common_subsequence(s1: &str, s2: &str) -> (usize, String) { let a: Vec<char> = s1.chars().collect(); let b: Vec<char> = s2.chars().collect(); let (m, n) = (a.len(), b.len()); let mut dp = vec![vec![0usize; n + 1]; m + 1]; for i in 1..=m { for j in 1..=n { if a[i-1] == b[j-1] { dp[i][j] = dp[i-1][j-1] + 1; } else { dp[i][j] = dp[i-1][j].max(dp[i][j-1]); } } } (dp[m][n], String::new()) // reconstruction omitted for brevity }

💰 Coin Change

Find the minimum number of coins to make a target amount. dp[a] = min(dp[a - coin] + 1) for all coins where coin <= a.

DynamicProgramming.java
static int coinChange(int[] coins, int target) { int[] dp = new int[target + 1]; Arrays.fill(dp, target + 1); // "impossible" dp[0] = 0; for (int a = 1; a <= target; a++) for (int coin : coins) if (coin <= a) dp[a] = Math.min(dp[a], dp[a - coin] + 1); return dp[target] > target ? -1 : dp[target]; }
dynamic_programming.cpp
int coin_change(const std::vector<int>& coins, int amount) { const int INF = amount + 1; std::vector<int> dp(amount + 1, INF); dp[0] = 0; for (int a = 1; a <= amount; ++a) for (int c : coins) if (c <= a) dp[a] = std::min(dp[a], dp[a - c] + 1); return dp[amount] >= INF ? -1 : dp[amount]; }
dynamic_programming.py
def coin_change(coins: List[int], amount: int) -> int: dp = [float("inf")] * (amount + 1) dp[0] = 0 for a in range(1, amount + 1): for coin in coins: if coin <= a and dp[a - coin] + 1 < dp[a]: dp[a] = dp[a - coin] + 1 return dp[amount] if dp[amount] != float("inf") else -1
dynamic_programming.rs
fn coin_change(coins: &[i32], amount: i32) -> i32 { let amount = amount as usize; let mut dp = vec![i32::MAX; amount + 1]; dp[0] = 0; for a in 1..=amount { for &coin in coins { let c = coin as usize; if c <= a && dp[a - c] != i32::MAX { dp[a] = min(dp[a], dp[a - c] + 1); } } } if dp[amount] == i32::MAX { -1 } else { dp[amount] } }

🔢 Subset Sum

Can we select a subset of nums that sums to target? Boolean DP with reverse iteration (same pattern as 0/1 knapsack).

DynamicProgramming.java
static boolean subsetSum(int[] nums, int target) { boolean[] dp = new boolean[target + 1]; dp[0] = true; // empty subset for (int x : nums) for (int t = target; t >= x; t--) // reverse! dp[t] = dp[t] || dp[t - x]; return dp[target]; }
dynamic_programming.py
def subset_sum(nums: List[int], target: int) -> bool: dp = [False] * (target + 1) dp[0] = True for num in nums: for j in range(target, num - 1, -1): if dp[j - num]: dp[j] = True return dp[target]

📈 Longest Increasing Subsequence (LIS)

Given an integer array, find the length of the longest strictly increasing subsequence. This is one of the most classic DP problems with two well-known approaches.

O(n²) DP Approach

Define dp[i] = length of the LIS ending at index i. For each j < i, if nums[j] < nums[i], we can extend the subsequence ending at j.

nums = [10, 9, 2, 5, 3, 7, 101, 18] dp[0] = 1 (just 10) dp[1] = 1 (just 9) dp[2] = 1 (just 2) dp[3] = 2 (2, 5) dp[4] = 2 (2, 3) dp[5] = 3 (2, 3, 7) or (2, 5, 7) dp[6] = 4 (2, 3, 7, 101) dp[7] = 4 (2, 3, 7, 18) Answer: max(dp) = 4
DynamicProgramming.java
// O(n^2) DP approach static int lisDP(int[] nums) { int n = nums.length; int[] dp = new int[n]; Arrays.fill(dp, 1); // each element is a subsequence of length 1 int maxLen = 1; for (int i = 1; i < n; i++) { for (int j = 0; j < i; j++) { if (nums[j] < nums[i]) dp[i] = Math.max(dp[i], dp[j] + 1); } maxLen = Math.max(maxLen, dp[i]); } return maxLen; }
dynamic_programming.cpp
// O(n^2) DP approach int lis_dp(const std::vector<int>& nums) { int n = static_cast<int>(nums.size()); std::vector<int> dp(n, 1); int max_len = 1; for (int i = 1; i < n; ++i) { for (int j = 0; j < i; ++j) { if (nums[j] < nums[i]) dp[i] = std::max(dp[i], dp[j] + 1); } max_len = std::max(max_len, dp[i]); } return max_len; }
dynamic_programming.py
# O(n^2) DP approach def lis_dp(nums: List[int]) -> int: n = len(nums) dp = [1] * n # each element is an LIS of length 1 for i in range(1, n): for j in range(i): if nums[j] < nums[i]: dp[i] = max(dp[i], dp[j] + 1) return max(dp)
dynamic_programming.rs
// O(n^2) DP approach fn lis_dp(nums: &[i32]) -> usize { let n = nums.len(); let mut dp = vec![1usize; n]; let mut max_len = 1; for i in 1..n { for j in 0..i { if nums[j] < nums[i] { dp[i] = dp[i].max(dp[j] + 1); } } max_len = max_len.max(dp[i]); } max_len }

O(n log n) Binary Search Approach

Maintain a tails array where tails[i] is the smallest tail element for an increasing subsequence of length i+1. For each number, use binary search to find its position.

Key Insight
The tails array is always sorted! So we can binary search for the correct position. If the number is larger than all tails, extend the array. Otherwise, replace the first tail that is >= the number. The final length of tails is the LIS length.
DynamicProgramming.java
// O(n log n) binary search approach static int lisBinarySearch(int[] nums) { int[] tails = new int[nums.length]; int size = 0; for (int x : nums) { int lo = 0, hi = size; while (lo < hi) { int mid = lo + (hi - lo) / 2; if (tails[mid] < x) lo = mid + 1; else hi = mid; } tails[lo] = x; if (lo == size) size++; } return size; }
dynamic_programming.cpp
// O(n log n) binary search approach int lis_binary_search(const std::vector<int>& nums) { std::vector<int> tails; for (int x : nums) { auto it = std::lower_bound(tails.begin(), tails.end(), x); if (it == tails.end()) tails.push_back(x); else *it = x; } return static_cast<int>(tails.size()); }
dynamic_programming.py
# O(n log n) binary search approach from bisect import bisect_left def lis_binary_search(nums: List[int]) -> int: tails = [] for x in nums: pos = bisect_left(tails, x) if pos == len(tails): tails.append(x) else: tails[pos] = x return len(tails)
dynamic_programming.rs
// O(n log n) binary search approach fn lis_binary_search(nums: &[i32]) -> usize { let mut tails: Vec<i32> = Vec::new(); for &x in nums { let pos = tails.partition_point(|&t| t < x); if pos == tails.len() { tails.push(x); } else { tails[pos] = x; } } tails.len() }
Important
The tails array does NOT contain the actual LIS -- it is a working array used to track the smallest possible tails. To reconstruct the actual subsequence, you need to store predecessor indices alongside the binary search.

✏️ Edit Distance

Given two strings word1 and word2, find the minimum number of operations (insert, delete, replace) to convert word1 into word2. Also known as Levenshtein distance.

Recurrence

if word1[i-1] == word2[j-1]: dp[i][j] = dp[i-1][j-1] // characters match, no operation else: dp[i][j] = 1 + min( dp[i-1][j], // delete from word1 dp[i][j-1], // insert into word1 dp[i-1][j-1] // replace in word1 )

2D Table Visualization

For word1 = "horse", word2 = "ros":

"" r o s "" [ 0, 1, 2, 3 ] h [ 1, 1, 2, 3 ] o [ 2, 2, 1, 2 ] r [ 3, 2, 2, 2 ] s [ 4, 3, 3, 2 ] e [ 5, 4, 4, 3 ] Answer: dp[5][3] = 3 Operations: horse → rorse (replace h→r) → rose (delete r) → ros (delete e)

Space Optimization
Edit Distance can be optimized from O(m*n) space to O(min(m,n)) by keeping only two rows (or even one row + a variable for the diagonal). This follows the same pattern as LCS space optimization.

🔨 Matrix Chain Multiplication

Given a sequence of matrices, find the most efficient way to multiply them. The order of multiplication matters because matrix multiplication is associative but the number of scalar multiplications varies dramatically with parenthesization.

Interval DP

This is a classic interval DP problem. Define dp[i][j] = minimum cost to multiply matrices from i to j. We try every possible split point k between i and j.

dp[i][j] = min over k from i to j-1 of: dp[i][k] + dp[k+1][j] + dims[i-1] * dims[k] * dims[j] Base case: dp[i][i] = 0 (single matrix, no multiplication needed) Order: Fill by increasing chain length (j - i)
Why Interval DP?
Unlike linear DP (where we fill left-to-right), interval DP fills by gap length. We first solve all subproblems of length 2, then length 3, and so on. This ensures that when computing dp[i][j], all smaller intervals dp[i][k] and dp[k+1][j] are already solved.
DynamicProgramming.java
// dims[] has length n+1 for n matrices // Matrix i has dimensions dims[i-1] x dims[i] static int matrixChainOrder(int[] dims) { int n = dims.length - 1; // number of matrices int[][] dp = new int[n + 1][n + 1]; // length 1: dp[i][i] = 0 (already initialized) for (int len = 2; len <= n; len++) { for (int i = 1; i <= n - len + 1; i++) { int j = i + len - 1; dp[i][j] = Integer.MAX_VALUE; for (int k = i; k < j; k++) { int cost = dp[i][k] + dp[k+1][j] + dims[i-1] * dims[k] * dims[j]; dp[i][j] = Math.min(dp[i][j], cost); } } } return dp[1][n]; }
dynamic_programming.cpp
int matrix_chain_order(const std::vector<int>& dims) { int n = static_cast<int>(dims.size()) - 1; std::vector<std::vector<int>> dp(n + 1, std::vector<int>(n + 1, 0)); for (int len = 2; len <= n; ++len) { for (int i = 1; i <= n - len + 1; ++i) { int j = i + len - 1; dp[i][j] = INT_MAX; for (int k = i; k < j; ++k) { int cost = dp[i][k] + dp[k + 1][j] + dims[i - 1] * dims[k] * dims[j]; dp[i][j] = std::min(dp[i][j], cost); } } } return dp[1][n]; }
dynamic_programming.py

Matrix i has dimensions dims[i-1] x dims[i] (1-indexed).""" n = len(dims) - 1 dp = [[0] * (n + 1) for _ in range(n + 1)] for length in range(2, n + 1): # chain length for i in range(1, n - length + 2): j = i + length - 1 dp[i][j] = float("inf") for k in range(i, j): # split point cost = dp[i][k] + dp[k + 1][j] + dims[i-1] * dims[k] * dims[j] dp[i][j] = min(dp[i][j], cost) return dp[1][n]
dynamic_programming.rs

Complexity
Time: O(n³) -- three nested loops (length, start, split point). Space: O(n²) for the DP table. This is optimal for the general matrix chain problem.

🏠 House Robber

A robber plans to rob houses along a street. Adjacent houses have security systems connected -- if two adjacent houses are robbed on the same night, the police are alerted. Given an array of non-negative integers representing the money in each house, determine the maximum amount you can rob without alerting the police.

Recurrence

dp[i] = max( dp[i-1], // skip house i dp[i-2] + nums[i] // rob house i (can't rob i-1) ) Base cases: dp[0] = nums[0], dp[1] = max(nums[0], nums[1])

Full DP Array

DynamicProgramming.java
static int rob(int[] nums) { if (nums.length == 0) return 0; if (nums.length == 1) return nums[0]; int[] dp = new int[nums.length]; dp[0] = nums[0]; dp[1] = Math.max(nums[0], nums[1]); for (int i = 2; i < nums.length; i++) { dp[i] = Math.max(dp[i-1], dp[i-2] + nums[i]); } return dp[nums.length - 1]; }
dynamic_programming.cpp

Space-Optimized to O(1)

Since dp[i] only depends on dp[i-1] and dp[i-2], we only need two variables -- exactly like Climbing Stairs.

DynamicProgramming.java
// O(1) space -- only two variables needed static int robOptimized(int[] nums) { int prev2 = 0, prev1 = 0; for (int num : nums) { int curr = Math.max(prev1, prev2 + num); prev2 = prev1; prev1 = curr; } return prev1; }
dynamic_programming.cpp
// O(1) space -- only two variables needed int rob_optimized(const std::vector<int>& nums) { int prev2 = 0, prev1 = 0; for (int num : nums) { int curr = std::max(prev1, prev2 + num); prev2 = prev1; prev1 = curr; } return prev1; }

Pattern Recognition
House Robber is a 1D DP where each state depends on the previous two states. This is the same structure as Climbing Stairs and Fibonacci. Whenever you see this pattern, you can always optimize from O(n) space to O(1) by keeping just two variables.

🗺 Unique Paths

A robot is located at the top-left corner of an m x n grid. It can only move right or down. How many unique paths exist to reach the bottom-right corner?

Recurrence

dp[i][j] = dp[i-1][j] + dp[i][j-1] // from above + from left Base cases: dp[0][j] = 1, dp[i][0] = 1 // only one way along edges Example (3 x 4 grid): [ 1, 1, 1, 1 ] [ 1, 2, 3, 4 ] [ 1, 3, 6, 10 ] Answer: 10 unique paths
DynamicProgramming.java
static int uniquePaths(int m, int n) { int[][] dp = new int[m][n]; // First row and first column are all 1s for (int i = 0; i < m; i++) dp[i][0] = 1; for (int j = 0; j < n; j++) dp[0][j] = 1; for (int i = 1; i < m; i++) for (int j = 1; j < n; j++) dp[i][j] = dp[i-1][j] + dp[i][j-1]; return dp[m-1][n-1]; } // Space-optimized: O(n) using a single row static int uniquePathsOptimized(int m, int n) { int[] dp = new int[n]; Arrays.fill(dp, 1); for (int i = 1; i < m; i++) for (int j = 1; j < n; j++) dp[j] += dp[j-1]; return dp[n-1]; }

Math Shortcut
Unique Paths can also be computed using combinatorics: the answer is C(m+n-2, m-1) since you need exactly (m-1) down moves and (n-1) right moves in any order. However, the DP approach generalizes to grids with obstacles (LeetCode 63).

🧰 DP Problem-Solving Framework

When faced with a DP problem, follow this systematic 6-step approach. This framework works for virtually every DP problem you will encounter in interviews and competitive programming.

Step 1: Identify the State

Ask: What information do I need to uniquely describe a subproblem? This becomes your DP array dimensions.

ProblemStateMeaning
Climbing Stairsdp[i]Ways to reach step i
Knapsackdp[i][w]Max value using first i items with capacity w
LCSdp[i][j]LCS length of first i chars and first j chars
Edit Distancedp[i][j]Min ops to convert word1[0..i] to word2[0..j]
Matrix Chaindp[i][j]Min cost to multiply matrices i through j

Step 2: Define the Recurrence

Express dp[current] in terms of dp[smaller subproblems]. Look for the last decision made and enumerate all choices.

Common patterns: Take or skip: dp[i] = max(dp[i-1], dp[i-2] + val[i]) // House Robber All choices: dp[i] = min(dp[i - coin] + 1) for each coin // Coin Change Two sequences: dp[i][j] depends on dp[i-1][j-1], dp[i-1][j], dp[i][j-1] // LCS, Edit Distance Interval: dp[i][j] = min(dp[i][k] + dp[k+1][j] + cost) // Matrix Chain

Step 3: Establish Base Cases

What are the smallest subproblems you can solve directly?

  • Empty input: dp[0] = 0 (or 1 for counting problems)
  • Single element: dp[1] = value of that element
  • Grid edges: First row/column often has a single path
  • Intervals of length 1: dp[i][i] = 0 (no operation needed)

Step 4: Determine Filling Order

Ensure that when you compute dp[i][j], all subproblems it depends on are already computed.

Filling Order Cheat Sheet
Linear DP: Left to right (dp[0], dp[1], ..., dp[n]).
2D string DP: Row by row, left to right within each row.
Interval DP: By increasing interval length.
0/1 knapsack (1D): Inner loop in REVERSE.
Unbounded knapsack (1D): Inner loop FORWARD.

Step 5: Extract the Answer

The final answer is not always the last cell!

ProblemAnswer Location
Climbing Stairsdp[n]
LISmax(dp[0..n-1]) -- not dp[n-1]!
Knapsackdp[n][W]
Edit Distancedp[m][n]
Coin Changedp[amount] (check for impossible)

Step 6: Optimize Space

After getting a correct solution, look for space optimization opportunities:

  • Row-only dependency: If dp[i] only depends on dp[i-1], keep two rows → O(n) space
  • Two-variable dependency: If dp[i] depends on dp[i-1] and dp[i-2], use two variables → O(1) space
  • Diagonal dependency: For 2D DP, if you need dp[i-1][j-1], store it in a temp variable before overwriting
Interview Tip
In interviews, always start with the unoptimized version. Explain the recurrence clearly, then offer the space optimization as a follow-up. Interviewers value clear thinking over premature optimization. Show that you understand why the optimization works, not just how to write it.
