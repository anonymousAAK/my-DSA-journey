🔘 Consistent Hashing

Consistent hashing distributes data across nodes in a way that minimizes redistribution when nodes are added or removed. Nodes are placed on a virtual ring using hash values. Each key is assigned to the first node encountered clockwise from its hash position.

Why Not Simple Modular Hashing?

With hash(key) % N, changing N (adding/removing a server) remaps almost every key. Consistent hashing ensures only K/N keys need to move on average (K = total keys, N = nodes).

Hash Ring (0 to 2^32 - 1): Node A (hash=100) / ----○---- / | \ / | \ ○ | ○ Node B (hash=300) | | | Node D | | (hash=800) | \ | / \ | / ----○---- \ Node C (hash=600) Key "user:42" (hash=250) → maps to Node B (next clockwise) Key "user:99" (hash=650) → maps to Node D (next clockwise)

Virtual Nodes

To achieve even distribution, each physical node gets multiple virtual nodes (replicas) on the ring. Typically 100-200 virtual nodes per physical node.

consistent_hashing.py
import hashlib
from bisect import bisect_right

class ConsistentHash:
    def __init__(self, nodes=None, replicas=150):
        self.replicas = replicas
        self.ring = {}       # hash -> node
        self.sorted_keys = []
        if nodes:
            for node in nodes:
                self.add_node(node)

    def _hash(self, key):
        return int(hashlib.md5(key.encode()).hexdigest(), 16)

    def add_node(self, node):
        for i in range(self.replicas):
            virtual_key = f"{node}#vn{i}"
            h = self._hash(virtual_key)
            self.ring[h] = node
            self.sorted_keys.append(h)
        self.sorted_keys.sort()

    def remove_node(self, node):
        for i in range(self.replicas):
            virtual_key = f"{node}#vn{i}"
            h = self._hash(virtual_key)
            del self.ring[h]
            self.sorted_keys.remove(h)

    def get_node(self, key):
        if not self.ring:
            return None
        h = self._hash(key)
        idx = bisect_right(self.sorted_keys, h)
        if idx == len(self.sorted_keys):
            idx = 0  # wrap around the ring
        return self.ring[self.sorted_keys[idx]]

ConsistentHash.java
import java.security.MessageDigest;
import java.util.TreeMap;

public class ConsistentHash {
    private final TreeMap<Long, String> ring = new TreeMap<>();
    private final int replicas;

    public ConsistentHash(int replicas) {
        this.replicas = replicas;
    }

    private long hash(String key) {
        try {
            MessageDigest md = MessageDigest.getInstance("MD5");
            byte[] digest = md.digest(key.getBytes());
            return ((long)(digest[0] & 0xFF) << 24)
                 | ((long)(digest[1] & 0xFF) << 16)
                 | ((long)(digest[2] & 0xFF) << 8)
                 | (digest[3] & 0xFF);
        } catch (Exception e) { throw new RuntimeException(e); }
    }

    public void addNode(String node) {
        for (int i = 0; i < replicas; i++) {
            long h = hash(node + "#vn" + i);
            ring.put(h, node);
        }
    }

    public void removeNode(String node) {
        for (int i = 0; i < replicas; i++) {
            long h = hash(node + "#vn" + i);
            ring.remove(h);
        }
    }

    public String getNode(String key) {
        if (ring.isEmpty()) return null;
        long h = hash(key);
        var entry = ring.ceilingEntry(h);
        return (entry != null) ? entry.getValue() : ring.firstEntry().getValue();
    }
}

#include <map>
#include <string>
#include <functional>

class ConsistentHash {
    std::map<size_t, std::string> ring;
    int replicas;
    std::hash<std::string> hasher;

public:
    ConsistentHash(int replicas = 150) : replicas(replicas) {}

    void addNode(const std::string& node) {
        for (int i = 0; i < replicas; i++) {
            size_t h = hasher(node + "#vn" + std::to_string(i));
            ring[h] = node;
        }
    }

    void removeNode(const std::string& node) {
        for (int i = 0; i < replicas; i++) {
            size_t h = hasher(node + "#vn" + std::to_string(i));
            ring.erase(h);
        }
    }

    std::string getNode(const std::string& key) {
        if (ring.empty()) return "";
        size_t h = hasher(key);
        auto it = ring.lower_bound(h);
        if (it == ring.end()) it = ring.begin();
        return it->second;
    }
};

use std::collections::BTreeMap;
use std::collections::hash_map::DefaultHasher;
use std::hash::{Hash, Hasher};

struct ConsistentHash {
    ring: BTreeMap<u64, String>,
    replicas: usize,
}

impl ConsistentHash {
    fn new(replicas: usize) -> Self {
        ConsistentHash { ring: BTreeMap::new(), replicas }
    }

    fn hash_key(key: &str) -> u64 {
        let mut hasher = DefaultHasher::new();
        key.hash(&mut hasher);
        hasher.finish()
    }

    fn add_node(&mut self, node: &str) {
        for i in 0..self.replicas {
            let vkey = format!("{}#vn{}", node, i);
            let h = Self::hash_key(&vkey);
            self.ring.insert(h, node.to_string());
        }
    }

    fn remove_node(&mut self, node: &str) {
        for i in 0..self.replicas {
            let vkey = format!("{}#vn{}", node, i);
            let h = Self::hash_key(&vkey);
            self.ring.remove(&h);
        }
    }

    fn get_node(&self, key: &str) -> Option<&str> {
        if self.ring.is_empty() { return None; }
        let h = Self::hash_key(key);
        let node = self.ring.range(h..).next()
            .or_else(|| self.ring.iter().next())
            .map(|(_, v)| v.as_str());
        node
    }
}
Key Insight
Consistent hashing is used in Amazon DynamoDB, Apache Cassandra, and CDN load balancers. When a node fails, only its keys (roughly 1/N of total) need to be redistributed.

🗃 LRU Cache

An LRU (Least Recently Used) Cache evicts the least recently accessed item when the cache is full. It requires O(1) for both get and put operations, achieved by combining a HashMap with a doubly linked list.

How It Works

HashMap: key → Node pointer Doubly Linked List: Most Recent ←→ ... ←→ Least Recent get(key): 1. Look up node in HashMap → O(1) 2. Move node to front of list → O(1) 3. Return value put(key, value): 1. If key exists: update value, move to front 2. If full: remove tail node, delete from HashMap 3. Insert new node at front, add to HashMap Example (capacity=3): put(1,A) → [1:A] put(2,B) → [2:B, 1:A] put(3,C) → [3:C, 2:B, 1:A] get(1) → [1:A, 3:C, 2:B] (moved 1 to front) put(4,D) → [4:D, 1:A, 3:C] (evicted 2:B)
lru_cache.py
class Node:
    def __init__(self, key=0, val=0):
        self.key = key
        self.val = val
        self.prev = None
        self.next = None

class LRUCache:
    def __init__(self, capacity):
        self.cap = capacity
        self.cache = {}  # key -> Node
        # Sentinel nodes to avoid null checks
        self.head = Node()
        self.tail = Node()
        self.head.next = self.tail
        self.tail.prev = self.head

    def _remove(self, node):
        node.prev.next = node.next
        node.next.prev = node.prev

    def _add_front(self, node):
        node.next = self.head.next
        node.prev = self.head
        self.head.next.prev = node
        self.head.next = node

    def get(self, key):
        if key not in self.cache:
            return -1
        node = self.cache[key]
        self._remove(node)
        self._add_front(node)
        return node.val

    def put(self, key, value):
        if key in self.cache:
            self._remove(self.cache[key])
        node = Node(key, value)
        self._add_front(node)
        self.cache[key] = node
        if len(self.cache) > self.cap:
            lru = self.tail.prev
            self._remove(lru)
            del self.cache[lru.key]

LRUCache.java
import java.util.HashMap;

public class LRUCache {
    class Node {
        int key, val;
        Node prev, next;
        Node(int k, int v) { key = k; val = v; }
    }

    private int capacity;
    private HashMap<Integer, Node> cache = new HashMap<>();
    private Node head = new Node(0, 0);
    private Node tail = new Node(0, 0);

    public LRUCache(int capacity) {
        this.capacity = capacity;
        head.next = tail;
        tail.prev = head;
    }

    private void remove(Node node) {
        node.prev.next = node.next;
        node.next.prev = node.prev;
    }

    private void addFront(Node node) {
        node.next = head.next;
        node.prev = head;
        head.next.prev = node;
        head.next = node;
    }

    public int get(int key) {
        if (!cache.containsKey(key)) return -1;
        Node node = cache.get(key);
        remove(node);
        addFront(node);
        return node.val;
    }

    public void put(int key, int value) {
        if (cache.containsKey(key)) remove(cache.get(key));
        Node node = new Node(key, value);
        addFront(node);
        cache.put(key, node);
        if (cache.size() > capacity) {
            Node lru = tail.prev;
            remove(lru);
            cache.remove(lru.key);
        }
    }
}
lru_cache.cpp
#include <unordered_map>
#include <list>

class LRUCache {
    int capacity;
    std::list<std::pair<int,int>> dll;  // {key, value}
    std::unordered_map<int, std::list<std::pair<int,int>>::iterator> cache;

public:
    LRUCache(int cap) : capacity(cap) {}

    int get(int key) {
        auto it = cache.find(key);
        if (it == cache.end()) return -1;
        dll.splice(dll.begin(), dll, it->second);
        return it->second->second;
    }

    void put(int key, int value) {
        auto it = cache.find(key);
        if (it != cache.end()) {
            it->second->second = value;
            dll.splice(dll.begin(), dll, it->second);
            return;
        }
        if ((int)cache.size() == capacity) {
            int lru_key = dll.back().first;
            dll.pop_back();
            cache.erase(lru_key);
        }
        dll.emplace_front(key, value);
        cache[key] = dll.begin();
    }
};
lru_cache.rs
use std::collections::HashMap;

struct Node {
    key: i32,
    val: i32,
    prev: usize,
    next: usize,
}

struct LRUCache {
    capacity: usize,
    cache: HashMap<i32, usize>,  // key -> index in nodes
    nodes: Vec<Node>,
    head: usize,  // sentinel
    tail: usize,  // sentinel
}

impl LRUCache {
    fn new(capacity: usize) -> Self {
        let mut nodes = Vec::new();
        nodes.push(Node { key: 0, val: 0, prev: 0, next: 1 }); // head
        nodes.push(Node { key: 0, val: 0, prev: 0, next: 1 }); // tail
        LRUCache { capacity, cache: HashMap::new(), nodes, head: 0, tail: 1 }
    }

    fn remove(&mut self, idx: usize) {
        let prev = self.nodes[idx].prev;
        let next = self.nodes[idx].next;
        self.nodes[prev].next = next;
        self.nodes[next].prev = prev;
    }

    fn add_front(&mut self, idx: usize) {
        let front = self.nodes[self.head].next;
        self.nodes[idx].next = front;
        self.nodes[idx].prev = self.head;
        self.nodes[front].prev = idx;
        self.nodes[self.head].next = idx;
    }

    fn get(&mut self, key: i32) -> i32 {
        if let Some(&idx) = self.cache.get(&key) {
            self.remove(idx);
            self.add_front(idx);
            self.nodes[idx].val
        } else {
            -1
        }
    }

    fn put(&mut self, key: i32, value: i32) {
        if let Some(&idx) = self.cache.get(&key) {
            self.nodes[idx].val = value;
            self.remove(idx);
            self.add_front(idx);
        } else {
            if self.cache.len() == self.capacity {
                let lru = self.nodes[self.tail].prev;
                self.remove(lru);
                self.cache.remove(&self.nodes[lru].key);
            }
            let idx = self.nodes.len();
            self.nodes.push(Node { key, val: value, prev: 0, next: 0 });
            self.cache.insert(key, idx);
            self.add_front(idx);
        }
    }
}
Interview Tip
LRU Cache (LeetCode 146) is one of the most frequently asked interview questions. The key insight is that a HashMap alone gives O(1) lookup but not O(1) ordering; a linked list alone gives O(1) reordering but not O(1) lookup. Combining them gives both.

⏱ Rate Limiter

Rate limiters control the rate of requests a client can make. Two common algorithms are Token Bucket and Sliding Window Counter.

Token Bucket Algorithm

A bucket holds tokens (up to a maximum). Tokens are added at a fixed rate. Each request consumes one token. If the bucket is empty, the request is rejected.

Token Bucket (capacity=5, refill=1/sec): t=0s: [●●●●●] 5 tokens (full) t=0s: Request! [●●●●] 4 tokens → ALLOWED t=0s: Request! [●●●] 3 tokens → ALLOWED t=1s: Refill [●●●●] 4 tokens t=1s: Request! [●●●] 3 tokens → ALLOWED ...burst 3 more... t=1s: [] 0 tokens t=1s: Request! [] 0 tokens → REJECTED

Sliding Window Counter

Track request counts in fixed time windows. Use weighted counts from the current and previous window to approximate a sliding window.

rate_limiter.py
import time

class TokenBucket:
    def __init__(self, capacity, refill_rate):
        self.capacity = capacity
        self.tokens = capacity
        self.refill_rate = refill_rate  # tokens per second
        self.last_refill = time.time()

    def allow_request(self):
        now = time.time()
        # Add tokens based on elapsed time
        elapsed = now - self.last_refill
        self.tokens = min(self.capacity,
                         self.tokens + elapsed * self.refill_rate)
        self.last_refill = now

        if self.tokens >= 1:
            self.tokens -= 1
            return True
        return False

class SlidingWindowCounter:
    def __init__(self, window_size, max_requests):
        self.window_size = window_size  # seconds
        self.max_requests = max_requests
        self.prev_count = 0
        self.curr_count = 0
        self.curr_window_start = 0

    def allow_request(self):
        now = time.time()
        curr_window = int(now // self.window_size)

        if curr_window != self.curr_window_start:
            self.prev_count = self.curr_count if curr_window == self.curr_window_start + 1 else 0
            self.curr_count = 0
            self.curr_window_start = curr_window

        # Weighted estimate
        elapsed_ratio = (now % self.window_size) / self.window_size
        estimate = self.prev_count * (1 - elapsed_ratio) + self.curr_count

        if estimate < self.max_requests:
            self.curr_count += 1
            return True
        return False
RateLimiter.java
public class TokenBucket {
    private final int capacity;
    private double tokens;
    private final double refillRate;
    private long lastRefillNanos;

    public TokenBucket(int capacity, double refillRate) {
        this.capacity = capacity;
        this.tokens = capacity;
        this.refillRate = refillRate;
        this.lastRefillNanos = System.nanoTime();
    }

    public synchronized boolean allowRequest() {
        long now = System.nanoTime();
        double elapsed = (now - lastRefillNanos) / 1e9;
        tokens = Math.min(capacity, tokens + elapsed * refillRate);
        lastRefillNanos = now;

        if (tokens >= 1) {
            tokens -= 1;
            return true;
        }
        return false;
    }
}

use std::time::Instant;

struct TokenBucket {
    capacity: f64,
    tokens: f64,
    refill_rate: f64,  // tokens per second
    last_refill: Instant,
}

impl TokenBucket {
    fn new(capacity: u32, refill_rate: f64) -> Self {
        TokenBucket {
            capacity: capacity as f64,
            tokens: capacity as f64,
            refill_rate,
            last_refill: Instant::now(),
        }
    }

        let now = Instant::now();
        let elapsed = now.duration_since(self.last_refill).as_secs_f64();
        self.tokens = (self.tokens + elapsed * self.refill_rate).min(self.capacity);
        self.last_refill = now;

        if self.tokens >= 1.0 {
            self.tokens -= 1.0;
            true
        } else {
            false
        }
    }
}
Design Consideration
In distributed systems, rate limiting across multiple servers requires a shared store (e.g., Redis). Use Redis INCR with TTL for simple counters, or Redis + Lua scripts for atomic token bucket operations.

🔍 Trie-Based Autocomplete

A Trie (prefix tree) stores strings character by character, enabling O(L) prefix lookups where L is the prefix length. For autocomplete, we extend the trie with frequency counts and return top-K suggestions.

Trie for: ["cat", "car", "card", "care", "dog"] (root) / \ c d | | a o / \ | t r g* * | / \ d e * * * = end of word Search "car" → node at 'r' DFS from 'r': "car"*, "card"*, "care"* Return: ["car", "card", "care"]
autocomplete.py
import heapq

class TrieNode:
    def __init__(self):
        self.children = {}
        self.is_end = False
        self.freq = 0

class AutoComplete:
    def __init__(self):
        self.root = TrieNode()

    def insert(self, word, freq=1):
        node = self.root
        for ch in word:
            if ch not in node.children:
                node.children[ch] = TrieNode()
            node = node.children[ch]
        node.is_end = True
        node.freq += freq

    def _find_prefix(self, prefix):
        node = self.root
        for ch in prefix:
            if ch not in node.children:
                return None
            node = node.children[ch]
        return node

    def _dfs(self, node, path, results):
        if node.is_end:
            results.append((node.freq, path))
        for ch, child in node.children.items():
            self._dfs(child, path + ch, results)

    def suggest(self, prefix, k=5):
        node = self._find_prefix(prefix)
        if not node:
            return []
        results = []
        self._dfs(node, prefix, results)
        # Return top-k by frequency
        top_k = heapq.nlargest(k, results, key=lambda x: x[0])
        return [word for freq, word in top_k]

AutoComplete.java
import java.util.*;

public class AutoComplete {
    class TrieNode {
        Map<Character, TrieNode> children = new HashMap<>();
        boolean isEnd;
        int freq;
    }

    private TrieNode root = new TrieNode();

    public void insert(String word, int freq) {
        TrieNode node = root;
        for (char ch : word.toCharArray()) {
            node.children.putIfAbsent(ch, new TrieNode());
            node = node.children.get(ch);
        }
        node.isEnd = true;
        node.freq += freq;
    }

    public List<String> suggest(String prefix, int k) {
        TrieNode node = root;
        for (char ch : prefix.toCharArray()) {
            if (!node.children.containsKey(ch)) return List.of();
            node = node.children.get(ch);
        }
        // Min-heap of (freq, word)
        PriorityQueue<int[]> pq = new PriorityQueue<>((a, b) -> a[0] - b[0]);
        List<String> words = new ArrayList<>();
        List<Integer> freqs = new ArrayList<>();
        dfs(node, new StringBuilder(prefix), words, freqs);

        for (int i = 0; i < words.size(); i++) {
            pq.offer(new int[]{freqs.get(i), i});
            if (pq.size() > k) pq.poll();
        }
        List<String> result = new ArrayList<>();
        while (!pq.isEmpty()) result.add(0, words.get(pq.poll()[1]));
        return result;
    }

    private void dfs(TrieNode node, StringBuilder sb,
                     List<String> words, List<Integer> freqs) {
        if (node.isEnd) { words.add(sb.toString()); freqs.add(node.freq); }
        for (var e : node.children.entrySet()) {
            sb.append(e.getKey());
            dfs(e.getValue(), sb, words, freqs);
            sb.deleteCharAt(sb.length() - 1);
        }
    }
}

#include <string>
#include <vector>
#include <unordered_map>
#include <queue>
#include <algorithm>

struct TrieNode {
    std::unordered_map<char, TrieNode*> children;
    bool is_end = false;
    int freq = 0;
};

class AutoComplete {
    TrieNode* root = new TrieNode();

    void dfs(TrieNode* node, std::string& path,
             std::vector<std::pair<int,std::string>>& results) {
        if (node->is_end)
            results.push_back({node->freq, path});
        for (auto& [ch, child] : node->children) {
            path.push_back(ch);
            dfs(child, path, results);
            path.pop_back();
        }
    }

public:
    void insert(const std::string& word, int freq = 1) {
        TrieNode* node = root;
        for (char ch : word) {
            if (!node->children.count(ch))
                node->children[ch] = new TrieNode();
            node = node->children[ch];
        }
        node->is_end = true;
        node->freq += freq;
    }

    std::vector<std::string> suggest(const std::string& prefix, int k = 5) {
        TrieNode* node = root;
        for (char ch : prefix) {
            if (!node->children.count(ch)) return {};
            node = node->children[ch];
        }
        std::vector<std::pair<int,std::string>> results;
        std::string path = prefix;
        dfs(node, path, results);
        std::sort(results.begin(), results.end(),
                  [](auto& a, auto& b){ return a.first > b.first; });
        std::vector<std::string> top;
        for (int i = 0; i < std::min(k, (int)results.size()); i++)
            top.push_back(results[i].second);
        return top;
    }
};

struct TrieNode {
    children: HashMap<char, TrieNode>,
    is_end: bool,
    freq: i32,
}

impl TrieNode {
    fn new() -> Self {
        TrieNode { children: HashMap::new(), is_end: false, freq: 0 }
    }
}

struct AutoComplete {
    root: TrieNode,
}

impl AutoComplete {
    fn new() -> Self {
        AutoComplete { root: TrieNode::new() }
    }

    fn insert(&mut self, word: &str, freq: i32) {
        let mut node = &mut self.root;
        for ch in word.chars() {
            node = node.children.entry(ch).or_insert_with(TrieNode::new);
        }
        node.is_end = true;
        node.freq += freq;
    }

    fn suggest(&self, prefix: &str, k: usize) -> Vec<String> {
        let mut node = &self.root;
        for ch in prefix.chars() {
            match node.children.get(&ch) {
                Some(child) => node = child,
                None => return vec![],
            }
        }
        let mut results = Vec::new();
        Self::dfs(node, &mut prefix.to_string(), &mut results);
        results.sort_by(|a, b| b.0.cmp(&a.0));
        results.into_iter().take(k).map(|(_, w)| w).collect()
    }

    fn dfs(node: &TrieNode, path: &mut String, results: &mut Vec<(i32, String)>) {
        if node.is_end {
            results.push((node.freq, path.clone()));
        }
        for (&ch, child) in &node.children {
            path.push(ch);
            Self::dfs(child, path, results);
            path.pop();
        }
    }
}
Optimization
For production autocomplete systems, pre-compute top-K suggestions at each trie node during insertion. This trades space for query speed — O(L) lookup instead of O(L + N) where N is the subtree size.

🛡 Merkle Tree

A Merkle tree (hash tree) is a binary tree where each leaf node contains the hash of a data block, and each internal node contains the hash of its children. It enables efficient and secure verification of large data structures.

How Verification Works

H(1234) / \ H(12) H(34) / \ / \ H(D1) H(D2) H(D3) H(D4) | | | | D1 D2 D3 D4 To verify D3 is unchanged: 1. Provide: H(D4) and H(12) (proof path) 2. Compute: H(D3) 3. Compute: H(34) = H(H(D3) + H(D4)) 4. Compute: H(1234) = H(H(12) + H(34)) 5. Compare H(1234) with stored root hash Only O(log n) hashes needed to verify any single block!
merkle_tree.py
import hashlib

class MerkleTree:
    def __init__(self, data_blocks):
        self.leaves = [self._hash(d) for d in data_blocks]
        # Pad to power of 2
        while len(self.leaves) & (len(self.leaves) - 1):
            self.leaves.append(self._hash(""))
        self.tree = [None] * (2 * len(self.leaves))
        self._build()

    def _hash(self, data):
        return hashlib.sha256(data.encode()).hexdigest()

    def _build(self):
        n = len(self.leaves)
        # Fill leaves
        for i in range(n):
            self.tree[n + i] = self.leaves[i]
        # Build up
        for i in range(n - 1, 0, -1):
            self.tree[i] = self._hash(
                self.tree[2 * i] + self.tree[2 * i + 1]
            )

    @property
    def root(self):
        return self.tree[1]

    def get_proof(self, index):
        """Return sibling hashes from leaf to root."""
        proof = []
        n = len(self.leaves)
        idx = n + index
        while idx > 1:
            sibling = idx ^ 1  # flip last bit
            proof.append(("left" if idx % 2 == 1 else "right",
                         self.tree[sibling]))
            idx //= 2
        return proof

    def verify(self, data, index, proof):
        """Verify a data block using its proof path."""
        current = self._hash(data)
        for side, sibling_hash in proof:
            if side == "left":
                current = self._hash(sibling_hash + current)
            else:
                current = self._hash(current + sibling_hash)
        return current == self.root

# Usage
mt = MerkleTree(["block1", "block2", "block3", "block4"])
proof = mt.get_proof(2)  # proof for "block3"
print(mt.verify("block3", 2, proof))  # True
print(mt.verify("tampered", 2, proof))  # False
MerkleTree.java
import java.security.MessageDigest;
import java.util.*;

public class MerkleTree {
    private String[] tree;
    private int leafCount;

    public MerkleTree(String[] dataBlocks) {
        leafCount = 1;
        while (leafCount < dataBlocks.length) leafCount *= 2;
        tree = new String[2 * leafCount];

        for (int i = 0; i < dataBlocks.length; i++)
            tree[leafCount + i] = hash(dataBlocks[i]);
        for (int i = dataBlocks.length; i < leafCount; i++)
            tree[leafCount + i] = hash("");
        for (int i = leafCount - 1; i >= 1; i--)
            tree[i] = hash(tree[2*i] + tree[2*i+1]);
    }

    private String hash(String data) {
        try {
            MessageDigest md = MessageDigest.getInstance("SHA-256");
            byte[] digest = md.digest(data.getBytes());
            StringBuilder sb = new StringBuilder();
            for (byte b : digest) sb.append(String.format("%02x", b));
            return sb.toString();
        } catch (Exception e) { throw new RuntimeException(e); }
    }

    public String getRoot() { return tree[1]; }

    public List<String[]> getProof(int index) {
        List<String[]> proof = new ArrayList<>();
        int idx = leafCount + index;
        while (idx > 1) {
            int sibling = idx ^ 1;
            proof.add(new String[]{
                idx % 2 == 1 ? "left" : "right", tree[sibling]
            });
            idx /= 2;
        }
        return proof;
    }

    public boolean verify(String data, List<String[]> proof) {
        String current = hash(data);
        for (String[] p : proof) {
            if (p[0].equals("left"))
                current = hash(p[1] + current);
            else
                current = hash(current + p[1]);
        }
        return current.equals(getRoot());
    }
}

merkle_tree.rs
use std::collections::hash_map::DefaultHasher;
use std::hash::{Hash, Hasher};

struct MerkleTree {
    tree: Vec<String>,
    leaf_count: usize,
}

impl MerkleTree {
    fn hash_data(data: &str) -> String {
        let mut hasher = DefaultHasher::new();
        data.hash(&mut hasher);
        format!("{:016x}", hasher.finish())
    }

    fn new(blocks: &[&str]) -> Self {
        let mut leaf_count = 1;
        while leaf_count < blocks.len() { leaf_count *= 2; }
        let mut tree = vec![String::new(); 2 * leaf_count];

        for (i, block) in blocks.iter().enumerate() {
            tree[leaf_count + i] = Self::hash_data(block);
        }
        for i in blocks.len()..leaf_count {
            tree[leaf_count + i] = Self::hash_data("");
        }
        for i in (1..leaf_count).rev() {
            let combined = format!("{}{}", tree[2*i], tree[2*i+1]);
            tree[i] = Self::hash_data(&combined);
        }
        MerkleTree { tree, leaf_count }
    }

    fn root(&self) -> &str { &self.tree[1] }

    fn get_proof(&self, index: usize) -> Vec<(&str, &str)> {
        let mut proof = Vec::new();
        let mut idx = self.leaf_count + index;
        while idx > 1 {
            let sibling = idx ^ 1;
            let side = if idx % 2 == 1 { "left" } else { "right" };
            proof.push((side, self.tree[sibling].as_str()));
            idx /= 2;
        }
        proof
    }

    fn verify(&self, data: &str, proof: &[(&str, &str)]) -> bool {
        let mut current = Self::hash_data(data);
        for &(side, hash) in proof {
            current = if side == "left" {
                Self::hash_data(&format!("{}{}", hash, current))
            } else {
                Self::hash_data(&format!("{}{}", current, hash))
            };
        }
        current == self.tree[1]
    }
}
Real-World Usage
Merkle trees are fundamental to Git (verifying repository integrity), Bitcoin (transaction verification in blocks), and distributed databases like Cassandra (anti-entropy repair). They enable verifying data integrity with only O(log n) hash comparisons.
